333/// Errors on the active side with network specs received through RPC call
334#[derive(Debug, PartialEq, Eq, thiserror::Error)]
335pub enum SpecsError {
336    /// Network base58 prefix information is not found neither in results of
337    /// the `system_properties` RPC call, nor in `System` pallet of the metadata
338    /// fetched with `state_getMetadata` RPC call.
339    #[error("no base58 prefix")]
340    NoBase58Prefix,
341
342    /// Network base58 prefix information found through `system_properties` RPC
343    /// call differs from the one from `System` pallet of the metadata fetched
344    /// with `state_getMetadata` RPC call.
345    ///
346    /// Associated data is corresponding base58 prefixes.
347    #[error("base58 prefix mismatch {specs}:{meta}")]
348    Base58PrefixMismatch { specs: u16, meta: u16 },
349
350    /// Network base58 prefix information received through `system_properties`
351    /// RPC call could not be transformed into expected `u16` prefix.
352    ///
353    /// Associated data is base58 prefix as received.
354    #[error("base58 prefix format not supported {value}")]
355    Base58PrefixFormatNotSupported { value: String },
356
357    /// Network decimals information **is not found** in the results if the
358    /// `system_properties` RPC call, but the unit information **is found**.
359    ///
360    /// Associated data is the fetched unit value.
361    #[error("unit no decimals {0}")]
362    UnitNoDecimals(String),
363
364    /// Network decimals information received through `system_properties`
365    /// RPC call could not be transformed into expected `u8` value.
366    ///
367    /// Associated data is decimals information as received.
368    #[error("decimals format not supported {value}")]
369    DecimalsFormatNotSupported { value: String },
370
371    /// Network unit information **is not found** in the results if the
372    /// `system_properties` RPC call, but the decimals information **is found**.
373    ///
374    /// Associated data is the fetched decimals value, could be array too.
375    #[error("decimals no unit {0}")]
376    DecimalsNoUnit(String),
377
378    /// Network unit information received through `system_properties`
379    /// RPC call could not be transformed into expected `String` value.
380    ///
381    /// Associated data is unit information as received.
382    #[error("unit format not supported {value}")]
383    UnitFormatNotSupported { value: String },
384
385    /// An array with more than one element is received for network decimals
386    /// through `system_properties` RPC call. Received units are not an array.
387    #[error("decimals array units not")]
388    DecimalsArrayUnitsNot,
389
390    /// Both the network decimals and network units are received as arrays,
391    /// but the array length is different, i.e. something not straightforward
392    /// is going on with the network.
393    ///
394    /// Associated data are the printed sets as they are received through the
395    /// `system_properties` RPC call.
396    #[error("decimals units array length {decimals} {unit}")]
397    DecimalsUnitsArrayLength { decimals: String, unit: String },
398
399    /// An array with more than one element is received for network units
400    /// through `system_properties` RPC call. Received decimals are not an array.
401    #[error("units array decimals not")]
402    UnitsArrayDecimalsNot,
403
404    /// Unit and decimal override is not allowed, when network has a single
405    /// token.
406    ///
407    /// The only case when the decimals and unit override is permitted is when
408    /// the network has a matching set of decimals and units, and user has to
409    /// select the needed set element manually.
410    ///
411    /// If the network has a single decimals value and a single unit value, i.e.
412    /// the values that would be suitable on their own, and user attempts to
413    /// override it, this error appears.
414    #[error("override ignored single")]
415    OverrideIgnoredSingle,
416
417    /// Unit and decimal override is not allowed, when network has no token.
418    #[error("override ignored none")]
419    OverrideIgnoredNone,
420}

422/// Data received through RPC call is different from the data in hot database
423#[derive(Debug)]
424pub enum Changed {
425    /// Network base58 prefix in hot database (consistent between the metadata
426    /// in `METATREE` and network specs in `SPECSPREPTREE`) is different from
427    /// the one received through new RPC calls (also consistent).
428    ///
429    /// Associated data is the base58 prefix values in question.
430    Base58Prefix { old: u16, new: u16 },
431
432    /// Network genesis hash in hot database is different from the one fetched
433    /// through a new RPC call.
434    ///
435    /// Network genesis hash is encountered in `SPECSTREEPREP` and
436    /// `ADDRESS_BOOK`.
437    ///
438    /// It is possible for a network to change genesis hash, some, especially
439    /// experimental ones, are doing it quite regularly.
440    ///
441    /// If the network has changed the genesis hash, it would be best to remove
442    /// the old entry from the database, and then load the new one. If the
443    /// network is one of the default ones (currently Polkadot, Kusama,
444    /// Westend), the `defaults` crate must be updated as well.
445    ///
446    /// Associated data is the genesis hash values in question.
447    GenesisHash { old: H256, new: H256 },
448
449    /// Network decimals value in
450    /// [`NetworkSpecs`](crate::network_specs::NetworkSpecs)
451    /// stored in `SPECSTREEPREP` tree of the hot database is different from
452    /// the one fetched through a new RPC call.
453    ///
454    /// Network decimals value is expected to be permanent.
455    Decimals { old: u8, new: u8 },
456
457    /// Network decimals value in
458    /// [`NetworkSpecs`](crate::network_specs::NetworkSpecs)
459    /// stored in `SPECSTREEPREP` tree of the hot database has some value,
460    /// freshly fetched specs have no decimals.
461    ///
462    /// Network decimals value is expected to be permanent. Override for no
463    /// decimals at the moment is blocked.
464    DecimalsBecameNone { old: u8 },
465
466    /// Network name is stored in multiple places in the hot database:
467    ///
468    /// - in `name` field of network specs
469    ///   [`NetworkSpecs`](crate::network_specs::NetworkSpecs) stored
470    ///   in `SPECSTREEPREP` tree
471    /// - in `name` field of address book entry
472    ///   [`AddressBookEntry`](crate::metadata::AddressBookEntry) stored in
473    ///   `ADDRESS_BOOK` tree
474    /// - encoded as a part of [`MetaKey`] and inside the network metadata
475    ///   stored in `METATREE` tree
476    ///
477    /// All those entries eventually are produced from network name that is
478    /// part of `Version` constant in `System` pallet of the network metadata.
479    ///
480    /// Network name is expected to be permanent. This error appears if the
481    /// name derived from metadata fetched through a new RPC call is different.
482    ///
483    /// [`MetaKey`]: crate::keyring::MetaKey
484    Name { old: String, new: String },
485
486    /// Network unit value in
487    /// [`NetworkSpecs`](crate::network_specs::NetworkSpecs)
488    /// stored in `SPECSTREEPREP` tree of the hot database is different from
489    /// the one fetched through a new RPC call.
490    ///
491    /// Network unit value is expected to be permanent.
492    Unit { old: String, new: String },
493
494    /// Network unit value in
495    /// [`NetworkSpecs`](crate::network_specs::NetworkSpecs)
496    /// stored in `SPECSTREEPREP` tree of the hot database has some value,
497    /// freshly fetched specs have no unit.
498    ///
499    /// Network unit value is expected to be permanent. Override for no
500    /// unit at the moment is blocked.
501    UnitBecameNone { old: String },
502}
503
504impl std::fmt::Display for Changed {
505    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
506        let (insert, old, new) = match self {
507            Changed::Base58Prefix { old, new } => {
508                ("base58 prefix", old.to_string(), new.to_string())
509            }
510            Changed::GenesisHash { old, new } => {
511                ("genesis hash", hex::encode(old), hex::encode(new))
512            }
513            Changed::Decimals { old, new } => ("decimals value", old.to_string(), new.to_string()),
514            Changed::DecimalsBecameNone { old } => {
515                ("decimals value", old.to_string(), "no value".to_string())
516            }
517            Changed::Name { old, new } => ("name", old.to_string(), new.to_string()),
518            Changed::Unit { old, new } => ("unit", old.to_string(), new.to_string()),
519            Changed::UnitBecameNone { old } => ("unit", old.to_string(), "no value".to_string()),
520        };
521
522        write!(
523            f,
524            "Network {insert} differs from the one in the hot database. Old: {old}. New: {new}."
525        )
526    }
527}
