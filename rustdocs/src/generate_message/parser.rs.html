566
567impl Make {
568    pub fn payload(&self) -> Result<Vec<u8>> {
569        Ok(std::fs::read(self.files_dir.join(&self.payload))?)
570    }
571
572    pub fn crypto(&self) -> Result<Crypto> {
573        if let Some(s) = match (
574            &self.sufficient.sufficient_hex,
575            &self.sufficient.sufficient_file,
576        ) {
577            (Some(hex), None) => Some(unhex(hex)?),
578            (None, Some(path)) => {
579                let sufficient_filename = &self.files_dir.join(path);
580                Some(std::fs::read(sufficient_filename)?)
581            }
582            _ => None,
583        } {
584            let s = <SufficientCrypto>::decode(&mut &s[..])?;
585            return Ok(Crypto::Sufficient { s });
586        }
587        let verifier_public_key = match (
588            self.verifier.verifier_alice,
589            &self.verifier.verifier_hex,
590            &self.verifier.verifier_file,
591        ) {
592            (Some(e), None, None) => return Ok(Crypto::Alice { e }),
593            (None, Some(hex), None) => unhex(hex)?,
594            (None, None, Some(path)) => {
595                let verifier_filename = &self.files_dir.join(path);
596                std::fs::read(verifier_filename)?
597            }
598            f => {
599                if self.signature.signature_file.is_none() && self.signature.signature_hex.is_none()
600                {
601                    return Ok(Crypto::None);
602                } else {
603                    panic!("mutually exclusive flags: {:?}", f);
604                }
605            }
606        };
607
608        let signature = match (
609            &self.signature.signature_hex,
610            &self.signature.signature_file,
611        ) {
612            (Some(hex), None) => unhex(hex)?,
613            (None, Some(path)) => {
614                let signature_filename = &self.files_dir.join(path);
615                std::fs::read(signature_filename)?
616            }
617            f => panic!("mutually exclusive flags: {:?}", f),
618        };
619
620        Ok(Crypto::Sufficient {
621            s: into_sufficient(verifier_public_key, signature, self.crypto.unwrap())?,
622        })
623    }
624}

764/// Overrides for `add-specs` command.
765#[derive(Args, Debug)]
766pub struct Override {
767    /// [`Encryption`] override to specify encryption algorithm used by a new
768    /// network or to add another encryption algorithm in known network.
769    #[arg(long, value_parser = encryption_from_args)]
770    pub encryption: Option<Encryption>,
771
772    /// Network title override, so that user can specify the network title in
773    /// [`NetworkSpecs`](definitions::network_specs::NetworkSpecs)
774    /// that determines under what title the network is displayed in the Vault
775    #[arg(long)]
776    pub title: Option<String>,
777
778    /// Token override to specify decimals used to display balance in
779    /// network transactions.
780    ///
781    /// Token override could be invoked only if:
782    ///
783    /// - network has no database record yet
784    /// - network has multiple decimals and unit values, those were retrieved as
785    ///   arrays of equal size.
786    #[arg(long)]
787    pub token_decimals: Option<u8>,
788
789    /// Token override to specify units used to display balance in
790    /// network transactions.
791    ///
792    /// Token override could be invoked only if:
793    ///
794    /// - network has no database record yet
795    /// - network has multiple decimals and unit values, those were retrieved as
796    ///   arrays of equal size.
797    #[arg(long)]
798    pub token_unit: Option<String>,
799}
800
801impl Override {
802    pub fn token(&self) -> Option<Token> {
803        match (self.token_decimals, self.token_unit.as_ref()) {
804            (Some(d), Some(u)) => Some(Token {
805                decimals: d,
806                unit: u.clone(),
807            }),
808            _ => None,
809        }
810    }
811}
812
813fn encryption_from_args(s: &str) -> std::result::Result<Encryption, &'static str> {
814    match s {
815        "ed25519" => Ok(Encryption::Ed25519),
816        "sr25519" => Ok(Encryption::Sr25519),
817        "ecdsa" => Ok(Encryption::Ecdsa),
818        "ethereum" => Ok(Encryption::Ethereum),
819        _ => {
820            Err("unexpected encryption type, expected `ed25519`, `sr25519`, `ecdsa` or `ethereum`")
821        }
822    }
823}
824
825impl Override {
826    /// Flag to indicate that no overrides were invoked.
827    pub fn all_empty(&self) -> bool {
828        self.encryption.is_none() && self.title.is_none() && self.token().is_none()
829    }
830}

842fn vec_to_pubkey_array<const N: usize>(v: Vec<u8>) -> Result<[u8; N]> {
843    v.try_into()
844        .map_err(|e: Vec<_>| Error::PublicKeyWrongLength(N, e.len()))
845}
846
847fn vec_to_signature_array<const N: usize>(v: Vec<u8>) -> Result<[u8; N]> {
848    v.try_into()
849        .map_err(|e: Vec<_>| Error::SignatureWrongLength(N, e.len()))
850}

852/// Fit public key and signature drafts into [`SufficientCrypto`].
853fn into_sufficient(
854    verifier_public_key: Vec<u8>,
855    signature: Vec<u8>,
856    encryption: Encryption,
857) -> Result<SufficientCrypto> {
858    match encryption {
859        Encryption::Ed25519 => {
860            let into_pubkey = vec_to_pubkey_array(verifier_public_key)?;
861            let public = ed25519::Public::from_raw(into_pubkey);
862            let into_sign = vec_to_signature_array(signature)?;
863            let signature = ed25519::Signature::from_raw(into_sign);
864            Ok(SufficientCrypto::Ed25519 { public, signature })
865        }
866        Encryption::Sr25519 => {
867            let into_pubkey = vec_to_pubkey_array(verifier_public_key)?;
868            let public = sr25519::Public::from_raw(into_pubkey);
869            let into_sign = vec_to_signature_array(signature)?;
870            let signature = sr25519::Signature::from_raw(into_sign);
871            Ok(SufficientCrypto::Sr25519 { public, signature })
872        }
873        Encryption::Ecdsa | Encryption::Ethereum => {
874            let into_pubkey = vec_to_pubkey_array(verifier_public_key)?;
875            let public = ecdsa::Public::from_raw(into_pubkey);
876            let into_sign = vec_to_signature_array(signature)?;
877            let signature = ecdsa::Signature::from_raw(into_sign);
878            Ok(SufficientCrypto::Ecdsa { public, signature })
879        }
880    }
