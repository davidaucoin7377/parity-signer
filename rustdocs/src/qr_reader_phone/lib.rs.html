20pub fn get_payload(line: &str, cleaned: bool) -> Result<Vec<u8>> {
21    let payload = match cleaned {
22        true => line,
23        false => parse_qr_payload(line)?,
24    };
25    Ok(hex::decode(payload)?)
26}

28pub fn get_length(line: &str, cleaned: bool) -> Result<u32> {
29    let payload = get_payload(line, cleaned)?;
30    if let Ok(frame) = RaptorqFrame::try_from(payload.as_ref()) {
31        Ok(frame.total())
32    } else if let Ok(frame) = LegacyFrame::try_from(payload.as_ref()) {
33        Ok(frame.total as u32)
34    } else if let Ok(banana_spilt_qr) = Share::new(payload) {
35        Ok(banana_spilt_qr.required_shards() as u32)
36    } else {
37        Ok(1)
38    }
39}

41pub fn decode_sequence(
42    set: &[String],
43    password: &Option<String>,
44    cleaned: bool,
45) -> Result<DecodeSequenceResult> {
46    let mut out = Ready::NotYet(InProgress::None);
47    let mut final_result: Option<String> = None;
48    for x in set {
49        let payload = get_payload(x, cleaned)?;
50        if let Ready::NotYet(decoding) = out {
51            out = process_decoded_payload(payload, password, decoding)?;
52            match out {
53                Ready::Yes(v) => {
54                    final_result = Some(hex::encode(v));
55                    break;
56                }
57                Ready::BananaSplitPasswordRequest => {
58                    return Ok(DecodeSequenceResult::BBananaSplitRecoveryResult {
59                        b: BananaSplitRecoveryResult::RequestPassword,
60                    });
61                }
62                Ready::BananaSplitReady(s) => {
63                    return Ok(DecodeSequenceResult::BBananaSplitRecoveryResult {
64                        b: BananaSplitRecoveryResult::RecoveredSeed { s },
65                    })
66                }
67                _ => (),
68            }
69        }
70    }
71    match final_result {
72        Some(s) => Ok(decode_payload(&s, ENABLE_DYNAMIC_DERIVATIONS)?),
73        None => Err(Error::UnableToDecode),
74    }
75}
