1//! Types description needed to decode transactions generated with
2//! [`RuntimeMetadata`](https://docs.rs/frame-metadata/15.0.0/frame_metadata/enum.RuntimeMetadata.html)
3//! `V12` and `V13`
4//!
5//! Prior to `V14`, metadata did not contain information on *what* the type used
6//! in the transaction actually is, it had only the type name.
7//! For common types (i.e the ones encountered in network Westend, Polkadot,
8//! Kusama, and Rococo) the type information was found elsewhere, and is a part of
9//! existing Vault defaults.
10//!
11//! [`TypeEntry`] is used to store the type information. Default types set contains [`TypeEntry`]
12//! data for common types, and is stored in `SETTREE` tree of both hot and cold databases
13//! with key `TYPES`.  
14//!
15//! For transaction with runtime metadata `V12` or `V13` to be decoded each of the
16//! encountered types must have a corresponding [`TypeEntry`] or be transformable into a type
17//! having [`TypeEntry`].  
18//!
19//! With `V14` release this module became obsolete, however, since not all networks are yet on `V14`
20//! metadata, it remains here.
21
22use parity_scale_codec::{Decode, Encode};
23
24/// Information each type has to have: type name and description
25#[derive(Decode, Encode, PartialEq, Eq, Clone)]
26pub struct TypeEntry {
27    /// type name, as it appears in the metadata
28    pub name: String,
29
30    /// type description
31    pub description: Description,
32}
33
34/// Type description
35///
36/// Must be available for all types
37#[derive(Decode, Encode, PartialEq, Eq, Clone)]
38pub enum Description {
39    /// type is a type alias
40    Type(String),
41
42    /// type is an enum
43    Enum(Vec<EnumVariant>),
44
45    /// type is a struct
46    Struct(Vec<StructField>),
47}
48
49/// Enum variant description
50///
51/// Must be available for each variant of the enum.
52#[derive(Decode, Encode, PartialEq, Eq, Clone)]
53pub struct EnumVariant {
54    /// enum variant name, will be displayed during the parsing if the variant is encountered
55    pub variant_name: String,
56
57    /// variant type
58    pub variant_type: EnumVariantType,
59}
60
61/// Types of enum variants
62#[derive(Decode, Encode, PartialEq, Eq, Clone)]
63pub enum EnumVariantType {
64    None,
65    Type(String),
66    Struct(Vec<StructField>),
67}
68
69/// Struct field description
70///
71/// Must be available for each field if the struct.
72#[derive(Decode, Encode, PartialEq, Eq, Clone)]
73pub struct StructField {
74    /// struct field name, if it exists, it will be displayed during the parsing if the struct is encountered
75    pub field_name: Option<String>,
76
77    /// field type
78    pub field_type: String,
79}
