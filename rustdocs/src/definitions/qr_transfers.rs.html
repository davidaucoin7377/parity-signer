1//! Information that could be send and received through air-gap as QR codes content  
2//!
3//! All information that Vault could get without a complete wipe must arrive
4//! through the QR codes maintaining the air-gap. In addition to transactions
5//! that could be parsed and signed, this includes:
6//! - adding new networks,  
7//! - updating the metadata of existing networks,  
8//! - updating the types information,  
9//! - bulk-importing the derivations  
10//!
11//! QR codes for adding networks, loading metadata and updating types information
12//! all have similar structure:
13//! - prelude `53xxyy` where `xx` is the encryption type, and `yy` is the message type  
14//! - verifier public key (if the QR code is signed by verifier)  
15//! - content  
16//! - verifier signature (if the QR code is signed by verifier)  
17//!
18//! QR codes for importing derivations are never signed and have structure:
19//! - prelude `53ffde`
20//! - content  
21//!
22//! This module deals with content part of QR codes.  
23

36/// `load_metadata` QR code content  
37///
38/// Messages `load_metadata` are used to update through air-gap the network
39/// metadata for networks already known to the Vault.
40pub struct ContentLoadMeta(Vec<u8>);
41
42#[derive(Decode, Encode)]
43struct DecodedContentLoadMeta {
44    meta: Vec<u8>,
45    genesis_hash: H256,
46}
47
48impl ContentLoadMeta {
49    /// Generate [`ContentLoadMeta`] from metadata `&[u8]` slice and network genesis hash.
50    pub fn generate(meta: &[u8], genesis_hash: &H256) -> Self {
51        Self(
52            DecodedContentLoadMeta {
53                meta: meta.to_vec(),
54                genesis_hash: genesis_hash.to_owned(),
55            }
56            .encode(),
57        )
58    }
59
60    /// Transform `&[u8]` slice into [`ContentLoadMeta`].
61    pub fn from_slice(slice: &[u8]) -> Self {
62        Self(slice.to_vec())
63    }
64
65    /// Get metadata `Vec<u8>` from [`ContentLoadMeta`].
66    pub fn meta(&self) -> Result<Vec<u8>> {
67        Ok(<DecodedContentLoadMeta>::decode(&mut &self.0[..])?.meta)
68    }
69
70    /// Get genesis hash `[u8; 32]` from [`ContentLoadMeta`].
71    pub fn genesis_hash(&self) -> Result<H256> {
72        Ok(<DecodedContentLoadMeta>::decode(&mut &self.0[..])?.genesis_hash)
73    }
74
75    /// Get metadata `Vec<u8>` and genesis hash `[u8; 32]` from [`ContentLoadMeta`] as a tuple.
76    pub fn meta_genhash(&self) -> Result<(Vec<u8>, H256)> {
77        let a = <DecodedContentLoadMeta>::decode(&mut &self.0[..])?;
78        Ok((a.meta, a.genesis_hash))
79    }
80
81    /// Write [`ContentLoadMeta`] into file that could be signed by the verifier.
82    #[cfg(feature = "active")]
83    pub fn write<P>(&self, filename: P) -> Result<()>
84    where
85        P: AsRef<Path>,
86    {
87        Ok(std::fs::write(filename, self.to_sign())?)
88    }
89
90    /// Transform [`ContentLoadMeta`] into `Vec<u8>` that could be signed by the verifier.
91    pub fn to_sign(&self) -> Vec<u8> {
92        self.0.to_vec()
93    }
94
95    /// Transform [`ContentLoadMeta`] into `Vec<u8>` that is concatenated with
96    /// other parts of the QR code.
97    ///
98    /// Currently similar to `.to_sign()` function.
99    pub fn to_transfer(&self) -> Vec<u8> {
100        self.0.to_vec()
101    }
102}

104/// `add_specs` QR code content  
105///
106/// Messages `add_specs` are used to add new networks to the Vault.
107#[derive(Decode, Encode)]
108pub struct ContentAddSpecs(Vec<u8>);
109
110#[derive(Decode, Encode)]
111struct DecodedContentAddSpecs {
112    specs: NetworkSpecs,
113}
114
115impl ContentAddSpecs {

117    pub fn generate(specs: &NetworkSpecs) -> Self {
118        Self(
119            DecodedContentAddSpecs {
120                specs: specs.to_owned(),
121            }
122            .encode(),
123        )
124    }
125
126    /// Transform `&[u8]` slice into [`ContentAddSpecs`].
127    pub fn from_slice(slice: &[u8]) -> Self {
128        Self(slice.to_vec())
129    }
130
131    /// Get network specs [`NetworkSpecs`] from [`ContentAddSpecs`].
132    pub fn specs(&self) -> Result<NetworkSpecs> {
133        Ok(<DecodedContentAddSpecs>::decode(&mut &self.0[..])?.specs)
134    }
135
136    /// Write [`ContentAddSpecs`] into file that could be signed by the verifier.
137    #[cfg(feature = "active")]
138    pub fn write<P>(&self, file_path: P) -> Result<()>
139    where
140        P: AsRef<Path>,
141    {
142        Ok(std::fs::write(file_path, self.to_sign())?)
143    }
144
145    /// Transform [`ContentAddSpecs`] into `Vec<u8>` that could be signed by the verifier.
146    pub fn to_sign(&self) -> Vec<u8> {
147        self.0.to_vec()
148    }
149
150    /// Transform [`ContentAddSpecs`] into `Vec<u8>` that is concatenated with
151    /// other parts of the QR code.
152    ///
153    /// Note that it is different from `.to_sign()` function. Effectively, already
154    /// SCALE-encoded [`NetworkSpecs`] are encoded second time as an opaque
155    /// `Vec<u8>`. This is done to have encoded piece length announced at the
156    /// beginning of the `u8` set, to simplify cutting the received message in Vault.
157    pub fn to_transfer(&self) -> Vec<u8> {
158        self.encode()
159    }
160}

162/// `load_types` QR code content  
163///
164/// Messages `load_types` are used to add or update Vault types information.
165///
166/// Externally acquired types information is needed only for
167/// [`RuntimeMetadata`](https://docs.rs/frame-metadata/15.0.0/frame_metadata/enum.RuntimeMetadata.html)
168/// `V13` and below. After `V14` all types information is contained within the metadata.
169///
170/// This kind of messages is expected to be used seldom, if ever.  
171#[derive(Decode, Encode)]
172pub struct ContentLoadTypes(Vec<u8>);
173
174#[derive(Decode, Encode)]
175struct DecodedContentLoadTypes {
176    types: Vec<TypeEntry>,
177}
178
179impl ContentLoadTypes {
180    /// Generate [`ContentLoadTypes`] from types information `&[TypeEntry]`.  
181    pub fn generate(types: &[TypeEntry]) -> Self {
182        Self(
183            DecodedContentLoadTypes {
184                types: types.to_owned(),
185            }
186            .encode(),
187        )
188    }
189
190    /// Transform `&[u8]` slice into [`ContentLoadTypes`].  
191    pub fn from_slice(slice: &[u8]) -> Self {
192        Self(slice.to_vec())
193    }
194
195    /// Get types information `Vec<TypeEntry>` from [`ContentLoadTypes`].  
196    pub fn types(&self) -> Result<Vec<TypeEntry>> {
197        Ok(<DecodedContentLoadTypes>::decode(&mut &self.0[..])?.types)
198    }
199
200    /// Write [`ContentLoadTypes`] into file that could be signed by the verifier.  
201    #[cfg(feature = "active")]
202    pub fn write<P>(&self, path: P) -> Result<()>
203    where
204        P: AsRef<Path>,
205    {
206        Ok(std::fs::write(path, self.to_sign())?)
207    }
208
209    /// Transform [`ContentLoadTypes`] into `Vec<u8>` to be put in the database.  
210    pub fn store(&self) -> Vec<u8> {
211        self.0.to_vec()
212    }
213
214    /// Transform [`ContentLoadTypes`] into `Vec<u8>` that could be signed by the verifier.  
215    pub fn to_sign(&self) -> Vec<u8> {
216        self.0.to_vec()
217    }
218
219    /// Transform [`ContentLoadTypes`] into `Vec<u8>` that is concatenated with
220    /// other parts of the QR code.  
221    ///
222    /// Note that it is different from `.to_sign()` function. Effectively, already
223    /// SCALE-encoded `Vec<TypeEntry>` is encoded second time as an opaque
224    /// `Vec<u8>`. This is done to have encoded piece length announced at the
225    /// beginning of the `u8` set, to simplify cutting the received message in Vault.
226    pub fn to_transfer(&self) -> Vec<u8> {
227        self.encode()
228    }
229
230    /// Generate types information hash and corresponding id pic  
231    ///
232    /// Types information hash is calculated for `Vec<u8>` of encoded types information,
233    /// as it would be stored in the database  
234    pub fn show(&self) -> (String, Identicon) {
235        use sp_core::blake2_256;
236
237        let types_hash = blake2_256(&self.store()).as_ref().to_vec();
238        let types_id_pic = pic_types(&types_hash);
239        (hex::encode(types_hash), types_id_pic)
240    }
