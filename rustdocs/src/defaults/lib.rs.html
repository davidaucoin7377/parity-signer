1//! Default data for [Vault](https://github.com/paritytech/parity-signer)
2//! and Vault-supporting ecosystem.
3//!
4//! This crate deals with data loaded into Vault on build and retained on reset.
5//!
6//! Vault supports by default three networks: Polkadot, Kusama, and Westend,
7//! both on the cold and the hot side. More default networks could be considered
8//! in the future.
9//!
10//! # Defaults in the cold database
11//!
12//! Default build of the cold database for Vault includes:
13//!
14//! - general verifier `Verifier` set to `Some(VerifierValue)` with Parity
15//!   public key as a verifier value
16//! - network specs `OrderedNetworkSpecs` for default networks
17//! - verifiers for default networks, set to
18//!   `CurrentVerifier::Valid(ValidCurrentVerifier::General)`
19//! - two latest metadata versions for each of the default networks
20//! - default types information
21//!
22//! Latest metadata entries get updated as soon as they are published and could
23//! be fetched via RPC calls. For this, new metadata entry is added into
24//! `release_metadata` folder, and older one is removed.
25//!
26//! # Defaults in the hot database
27//!
28//! Default hot database contains:
29//!
30//! - network specs `NetworkSpecs` for default networks
31//! - address book containing the data for default networks
32//! - default types information
33//!
34//! # Features
35//! Feature `"signer"` corresponds to everything related to Vault air-gapped
36//! device. Currently the only Vault-related default is `Verifier` with Parity
37//! public key, that could be set up as a general verifier after Vault wipe by
38//! the user.
39//!
40//! Feature `"active"` corresponds to all Vault-related things happening
41//! **without** air-gap. This includes generating default:
42//! - network specs sets for cold and hot databases
43//! - verifiers for cold databases
44//! - metadata sets for cold databases
45//! - address book for hot database
46//! - types information for cold and hot databases
47//!
48//! Feature `"test"` includes both `"signer"` and `"active"` features, along
49//! with some testing, and is the default one.

265/// Read metadata files from given directory. Is used to populate different
266/// variants of cold database (release and tests)
267#[cfg(feature = "active")]
268fn metadata(dir: &str) -> Result<Vec<MetaValues>> {
269    let mut out: Vec<MetaValues> = Vec::new();
270    let path_set = std::fs::read_dir(dir)?;
271    for x in path_set.flatten() {
272        if let Some(filename) = x.path().to_str() {
273            let meta_str = std::fs::read_to_string(x.path())?;
274            let new = MetaValues::from_str_metadata(meta_str.trim())?;
275            let mut found = false;
276            for a in default_network_info() {
277                if new.name == a.name {
278                    found = true;
279                    if let Some(prefix_from_meta) = new.optional_base58prefix {
280                        if prefix_from_meta != a.base58prefix {
281                            return Err(Error::Base58PrefixSpecsMismatch {
282                                specs: a.base58prefix,
283                                meta: prefix_from_meta,
284                            });
285                        }
286                    }
287                    break;
288                }
289            }
290            if !found {
291                return Err(Error::OrphanMetadata {
292                    name: new.name,
293                    filename: filename.to_string(),
294                });
295            }
296            out.push(new)
297        }
298    }
299    Ok(out)
300}

321#[cfg(feature = "active")]
322lazy_static! {
323    static ref REG_STRUCTS_WITH_NAMES: Regex = Regex::new(r"(pub )?struct (?P<name>.*?)( )?\{(?P<description>(\n +(pub )?\w+: .*(,)?)*\n)\}").expect("checked construction");
324    static ref REG_STRUCTS_NO_NAMES: Regex = Regex::new(r"(pub )?struct (?P<name>.*?)( )?\((pub )?(?P<description>.*)\)").expect("checked construction");
325    static ref REG_STRUCT_FIELDS: Regex = Regex::new(r"(?m)^ +(pub )?(?P<field_name>[\w]+): (?P<field_type>[^ ,]*\(.*\)[^ ,]*|[^ ,]*<.*>[^ ,]*|[^ ,]*\[.*\][^ ,]*|[^ ,]+)(,)?$").expect("checked construction");
326    static ref REG_ENUM: Regex = Regex::new(r"(pub )?enum (?P<name>.*?)( )?\{(?P<description>(\n +\w+( )?(\(.*\))?(\{[^}]*\})?(,)?)*)\n\}").expect("checked construction");
327    static ref REG_ENUM_VARIANTS: Regex = Regex::new(r"(?m)^ +(?P<variant_name>\w+)(( )?(?P<variant_type>(\(.*\))|(\{[^}]*\})))?(,)?$").expect("checked construction");
328    static ref REG_ENUM_SIMPLE: Regex = Regex::new(r"(?m)^(?P<simple_type>([^, ]*\[.*\][^, ]*)|([^, ]*<.*>[^, ]*)|([^, ]*\([^)]*\)[^, ]*)|([^, ]+))$").expect("checked construction");
329    static ref REG_ENUM_TUPLE: Regex = Regex::new(r"(?P<type_in_tuple>([^, ]*\[.*\][^, ]*)|([^, ]*<.*>[^, ]*)|([^, ]*\([^)]*\)[^, ]*)|([^, ]+))(, )?").expect("checked construction");
330    static ref REG_ENUM_STRUCT: Regex = Regex::new(r"(\n)?( +)?(?P<struct_field_name>\w+): (?P<struct_field_type>)(?P<fieldtype>[^ ,]*\(.*\)[^ ,]*|[^ ,]*<.*>[^ ,]*|[^ ,]*\[.*\][^ ,]*|[^ ,]+)(,)?").expect("checked construction");
331    static ref REG_TYPES: Regex = Regex::new(r#"(?m)(pub )?type (?P<name>.*) = (?P<description>.*);$"#).expect("checked construction");
332}

334/// Generate default types as [`TypeEntry`] set
335///
336/// Type definitions stored in `../defaults/default_types/full_types_information`
337/// are recorded with standard Rust syntax, processed with regular expressions
338/// to categorize, and collected as `TypeEntry` set.
339///
340/// Types information is necessary only to parse transactions produced with
341/// `RuntimeVersion` of the network metadata below `V14`. Therefore, it is
342/// obsolete for default networks.
343///
344/// Types information currently on file was collected for older metadata
345/// versions of Westend, Polkadot, Kusama, and Rococo, when they were still
346/// using metadata runtime version `V12` and `V13`. Type definitions were gathered
347/// (mostly) from substrate crates and in some cases from js client code, when
348/// it was not possible to find explicit descriptions in substrate crates.
349///
350/// This types collection is totally not expected to be exhaustive for any
351/// network a user may encounter, although the most common types are likely
352/// described. If user gets an error that certain type is unknown, this type
353/// description could be added into types file by user, and types update could
354/// be generated by user.
355#[cfg(feature = "active")]
356pub fn default_types_vec() -> Result<Vec<TypeEntry>> {
357    let filename = "../defaults/default_types/full_types_information";
358    let type_info = fs::read_to_string(filename)?;
359
360    let mut types_prep: Vec<TypeEntry> = Vec::new();
361
362    for caps1 in REG_STRUCTS_WITH_NAMES.captures_iter(&type_info) {
363        let struct_name = (caps1["name"]).to_string();
364        let struct_description = (caps1["description"]).to_string();
365        let mut struct_fields: Vec<StructField> = Vec::new();
366        for caps2 in REG_STRUCT_FIELDS.captures_iter(&struct_description) {
367            let new = StructField {
368                field_name: Some((caps2["field_name"]).to_string()),
369                field_type: (caps2["field_type"]).to_string(),
370            };
371            struct_fields.push(new);
372        }
373        let new_entry = TypeEntry {
374            name: struct_name,
375            description: Description::Struct(struct_fields),
376        };
377        types_prep.push(new_entry);
378    }
379    for caps in REG_STRUCTS_NO_NAMES.captures_iter(&type_info) {
380        let only_field = StructField {
381            field_name: None,
382            field_type: (caps["description"]).to_string(),
383        };
384        let new_entry = TypeEntry {
385            name: (caps["name"]).to_string(),
386            description: Description::Struct(vec![only_field]),
387        };
388        types_prep.push(new_entry);
389    }
390    for caps1 in REG_ENUM.captures_iter(&type_info) {
391        let enum_name = (caps1["name"]).to_string();
392        let enum_description = (caps1["description"]).to_string();
393        let enum_variants = enum_description
394            .lines()
395            .filter(|line| REG_ENUM_VARIANTS.is_match(line))
396            .map(|line| {
397                let caps2 = REG_ENUM_VARIANTS
398                    .captures(line)
399                    .expect("just checked it is match");
400                let variant_name = (caps2["variant_name"]).to_string();
401                let variant_type = match caps2.name("variant_type") {
402                    None => EnumVariantType::None,
403                    Some(a) => {
404                        let x = a.as_str().to_string();
405                        if x.starts_with('(') {
406                            // either a single type or a tuple
407                            match REG_ENUM_SIMPLE.captures(&x[1..x.len() - 1]) {
408                                // single type
409                                Some(b) => EnumVariantType::Type((b["simple_type"]).to_string()),
410                                // tuple
411                                None => EnumVariantType::Type(x),
412                            }
413                        } else {
414                            // struct with struct-like fields
415                            let mut type_is_struct: Vec<StructField> = Vec::new();
416                            for caps3 in REG_ENUM_STRUCT.captures_iter(&x) {
417                                let new = StructField {
418                                    field_name: Some((caps3["field_name"]).to_string()),
419                                    field_type: (caps3["field_type"]).to_string(),
420                                };
421                                type_is_struct.push(new);
422                            }
423                            EnumVariantType::Struct(type_is_struct)
424                        }
425                    }
426                };
427                EnumVariant {
428                    variant_name,
429                    variant_type,
430                }
431            })
432            .collect();
433        let new_entry = TypeEntry {
434            name: enum_name,
435            description: Description::Enum(enum_variants),
436        };
437        types_prep.push(new_entry);
438    }
439    for caps in REG_TYPES.captures_iter(&type_info) {
440        let new_entry = TypeEntry {
441            name: (caps["name"]).to_string(),
442            description: Description::Type((caps["description"]).to_string()),
443        };
444        types_prep.push(new_entry);
445    }
446
447    Ok(types_prep)
448}
