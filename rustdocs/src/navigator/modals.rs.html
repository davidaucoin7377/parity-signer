1//! List of all modals
2
3use definitions::{keyring::NetworkSpecsKey, navigation::MSCContent};
4use sp_runtime::MultiSignature;
5use transaction_signing::SignatureType;
6
7#[derive(PartialEq, Eq, Debug, Clone)]
8pub enum Modal {
9    Backup(String),
10    NewSeedMenu,
11    NewSeedBackup(String),
12    SeedMenu,
13    NetworkSelector(NetworkSpecsKey),
14    PasswordConfirm,
15    SignatureReady(Vec<(MultiSignature, SignatureType)>),
16    EnterPassword,
17    LogComment,
18    LogRight,
19    NetworkDetailsMenu,
20    ManageMetadata(u32),
21    SufficientCryptoReady((Vec<u8>, MSCContent)),
22    KeyDetailsAction,
23    TypesInfo,
24    SelectSeed,
25    Empty,
26}
27
28impl Modal {
29    pub fn get_name(&self) -> String {
30        match self {
31            Modal::Backup(_) => String::from("Backup"),
32            Modal::NewSeedMenu => String::from("NewSeedMenu"),
33            Modal::NewSeedBackup(_) => String::from("NewSeedBackup"),
34            Modal::SeedMenu => String::from("SeedMenu"),
35            Modal::NetworkSelector(_) => String::from("NetworkSelector"),
36            Modal::PasswordConfirm => String::from("PasswordConfirm"),
37            Modal::SignatureReady(_) => String::from("SignatureReady"),
38            Modal::EnterPassword => String::from("EnterPassword"),
39            Modal::LogComment => String::from("LogComment"),
40            Modal::LogRight => String::from("LogRight"),
41            Modal::NetworkDetailsMenu => String::from("NetworkDetailsMenu"),
42            Modal::ManageMetadata(_) => String::from("ManageMetadata"),
43            Modal::SufficientCryptoReady(_) => String::from("SufficientCryptoReady"),
44            Modal::KeyDetailsAction => String::from("KeyDetailsAction"),
45            Modal::TypesInfo => String::from("TypesInfo"),
46            Modal::SelectSeed => String::from("SelectSeed"),
47            Modal::Empty => String::from("Empty"),
48        }
49    }
50}
