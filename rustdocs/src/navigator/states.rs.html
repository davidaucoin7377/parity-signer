10const MAX_COUNT_SET: u8 = 3;

12/// The result of a step within the signing protocol
13/// between the user and the Vault.
14#[derive(Debug, Clone, PartialEq, Eq)]
15pub enum SignResult {
16    /// A password for one of the passworded keys is requested.
17    RequestPassword { idx: usize, counter: u8 },
18
19    /// All signatures are ready.
20    Ready {
21        signatures: Vec<(MultiSignature, SignatureType)>,
22    },
23}
24
25/// State of transaction screen.
26///
27/// In general case Vault may sign a bulk of transactions
28/// (more than one) and any subset of the transactions within
29/// a bulk may be signed by the passworded keys. This structure
30/// implements an interactive protocol between Vault and the user
31/// where user repeatedly enters all necessary passwords until all
32/// transactions are successfully signed or a password entry limit
33/// occurs.
34///
35/// If the same passworded key is used to sign more than one
36/// transaction in the bulk the password for this key is only
37/// requested once.
38///
39/// In case when no transactions in the bulk are signed by passworded
40/// key the result becomes available to the user right away minimizing
41/// the amount of user actions necessary.
42#[derive(Clone, Debug)]
43pub struct TransactionState {
44    /// The vector of seeds per-transcation being signed.
45    /// Since it is possible that signer is signing more than
46    /// one transaction (a bulk) this is a `Vec`.
47    seeds: Vec<String>,
48
49    /// Passwords for the accounts. Key is the ss58 addr.
50    passwords: HashMap<String, String>,
51
52    /// The `TransactionAction` being processed.
53    action: TransactionAction,
54
55    /// User-provided comments for each transaction.
56    comments: Vec<String>,
57
58    /// Failed password entries counter for the tx
59    /// currently being signed if any.
60    counter: u8,
61
62    /// The index of the transaction being currently signed.
63    currently_signing: usize,
64
65    /// Accumulates already-produced signatures.
66    signatures: Vec<(MultiSignature, SignatureType)>,
67}
68
69impl TransactionState {
70    pub fn current_password_author_info(
71        &self,
72    ) -> Option<(MAddressCard, Option<OrderedNetworkSpecs>)> {
73        match &self.action {
74            TransactionAction::Sign { actions, .. } => Some((
75                actions[self.currently_signing].author_info.clone(),
76                actions[self.currently_signing].get_network_spec(),
77            )),
78            _ => None,
79        }
80    }
81
82    pub fn new(database: &sled::Db, details_str: &str) -> Result<Self> {
83        Ok(Self {
84            seeds: vec![],
85            action: produce_output(database, details_str)?,
86            counter: 1,
87            passwords: HashMap::new(),
88            comments: vec![],
89            currently_signing: 0,
90            signatures: vec![],
91        })
92    }
93
94    pub fn update_seeds(&mut self, seeds: &str) {
95        if self.seeds.is_empty() {
96            self.seeds = seeds.lines().map(|seed| seed.to_string()).collect();
97        }
98    }
99
100    pub fn update_comments(&mut self, comments: &str) {
101        if self.comments.is_empty() {
102            self.comments = comments
103                .lines()
104                .map(|comment| comment.to_string())
105                .collect();
106        }
107    }
108
109    pub fn update_checksum_sign(&mut self, new_checksum: u32) {
110        if let TransactionAction::Sign { checksum, .. } = &mut self.action {
111            *checksum = new_checksum;
112        }
113    }
114
115    pub fn action(&self) -> &TransactionAction {
116        &self.action
117    }
118
119    /// Try to further progress the signing of transactions.
120    pub fn handle_sign(&mut self, database: &sled::Db) -> Result<SignResult> {
121        if let TransactionAction::Sign { actions, checksum } = &mut self.action {
122            if self.seeds.len() != actions.len() {
123                return Err(Error::SeedsNumMismatch(self.seeds.len(), actions.len()));
124            }
125
126            loop {
127                // Get the tx currently being signed.
128                let action = &actions[self.currently_signing];
129
130                // Get the password; if there is none, request one.
131                let password = if action.has_pwd {
132                    match self.passwords.get(&action.author_info.base58) {
133                        Some(pwd) => pwd,
134                        None => {
135                            return Ok(SignResult::RequestPassword {
136                                idx: self.currently_signing,
137                                counter: self.counter,
138                            });
139                        }
140                    }
141                } else {
142                    ""
143                };
144
145                // Try to sign it.
146                match transaction_signing::create_signature(
147                    database,
148                    &self.seeds[self.currently_signing],
149                    password,
150                    self.comments
151                        .get(self.currently_signing)
152                        .map(|s| s.as_str())
153                        .unwrap_or_else(|| ""),
154                    *checksum,
155                    self.currently_signing,
156                    action.get_encryption(),
157                ) {
158                    Ok(signature_and_checksum) => {
159                        // If signed successfully progress to the
160                        // next transaction in the bulk.
161                        self.currently_signing += 1;
162                        self.counter = 1;
163                        *checksum = signature_and_checksum.new_checksum();
164                        self.signatures.push((
165                            signature_and_checksum.signature().clone(),
166                            signature_and_checksum.signature_type(),
167                        ));
168
169                        // If this is the last tx, return
170                        //
171                        if self.currently_signing == self.seeds.len() {
172                            return Ok(SignResult::Ready {
173                                signatures: self.signatures.clone(),
174                            });
175                        }
176                    }
177                    Err(e) => {
178                        match e {
179                            SignError::WrongPasswordNewChecksum(new_checksum) => {
180                                // Password was not correct, re-request it.
181                                self.update_checksum_sign(new_checksum);
182                                self.counter += 1;
183
184                                return Ok(SignResult::RequestPassword {
185                                    idx: self.currently_signing,
186                                    counter: self.counter,
187                                });
188                            }
189                            _ => {
190                                // Other signing error happened.
191                                return Err(Error::TransactionSigning(e));
192                            }
193                        }
194                    }
195                }
196            }
197        } else {
198            Err(Error::TxActionNotSign)
199        }
200    }
201
202    pub fn password_entered(&mut self, pwd: &str) {
203        if let TransactionAction::Sign { actions, .. } = &self.action {
204            let base58 = actions[self.currently_signing].author_info.base58.clone();
205            self.passwords.insert(base58, pwd.to_string());
206        }
207    }
208
209    pub fn ok(&self) -> bool {
210        self.counter < MAX_COUNT_SET
211    }
212
213    pub fn counter(&self) -> u8 {
214        self.counter
215    }
