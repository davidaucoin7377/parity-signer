62impl ParserCard {
63    pub fn show_no_docs(&self, indent: u32) -> String {
64        match &self {
65            ParserCard::Pallet(pallet_name) => readable(indent, "pallet", pallet_name),
66            ParserCard::Method {
67                method_name,
68                docs: _,
69            } => readable(indent, "method", method_name),
70            ParserCard::Varname(varname) => readable(indent, "varname", varname),
71            ParserCard::Default(decoded_string) => readable(indent, "default", decoded_string),
72            ParserCard::Text(decoded_text) => readable(indent, "text", decoded_text),
73            ParserCard::Id { id, base58prefix } => readable(
74                indent,
75                "Id",
76                &id.to_ss58check_with_version(Ss58AddressFormat::custom(*base58prefix)),
77            ),
78            ParserCard::Id20 {
79                id,
80                base58prefix: _,
81            } => readable(indent, "Id", &format!("0x{}", hex::encode(id))),
82            ParserCard::None => readable(indent, "none", ""),
83            ParserCard::IdentityField(variant) => readable(indent, "identity_field", variant),
84            ParserCard::BitVec(bv) => readable(indent, "bitvec", bv),
85            ParserCard::Balance { number, units } => {
86                readable(indent, "balance", &format!("{number} {units}"))
87            }
88            ParserCard::FieldName {
89                name,
90                docs_field_name: _,
91                path_type: _,
92                docs_type: _,
93            } => readable(indent, "field_name", name),
94            ParserCard::FieldNumber {
95                number,
96                docs_field_number: _,
97                path_type: _,
98                docs_type: _,
99            } => readable(indent, "field_number", &number.to_string()),
100            ParserCard::EnumVariantName {
101                name,
102                docs_enum_variant: _,
103            } => readable(indent, "enum_variant_name", name),
104            ParserCard::Era(era) => match era {
105                Era::Immortal => readable(indent, "era", "Immortal"),
106                Era::Mortal(period, phase) => readable(
107                    indent,
108                    "era",
109                    &format!("Mortal, phase: {phase}, period: {period}"),
110                ),
111            },
112            ParserCard::Nonce(nonce) => readable(indent, "nonce", nonce),
113            ParserCard::BlockHash(block_hash) => {
114                readable(indent, "block_hash", &hex::encode(block_hash))
115            }
116            ParserCard::Tip { number, units } => {
117                readable(indent, "tip", &format!("{number} {units}"))
118            }
119            ParserCard::NetworkNameVersion { name, version } => {
120                readable(indent, "network", &format!("{name}{version}"))
121            }
122            ParserCard::TxVersion(x) => readable(indent, "tx_version", x),
123        }
124    }
125}
126
127fn readable(indent: u32, card_type: &str, card_payload: &str) -> String {
128    format!(
129        "{}{}: {}",
130        "  ".repeat(indent as usize),
131        card_type,
132        card_payload
133    )
134}
