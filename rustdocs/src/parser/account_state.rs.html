16/// A decoding state that detects account ID byte sequences and outputs a single
17/// card containing the corresponding human-readable address.
18///
19/// This state is designed to recognize account IDs typically represented as
20/// `[u8; 32]` (e.g. for Substrate-based addresses) or `[u8; 20]` (e.g. Ethereum-style addresses).
21///
22/// When such a sequence is detected, the state replaces the default byte output
23/// with a formatted address string to improve readability.
24#[derive(Debug, Clone)]
25pub struct AccountState {
26    len: Option<usize>,
27    partial: Vec<u8>,
28    extra_info: ExtraInfo,
29}
30
31impl AccountState {
32    pub fn new(extra_info: ExtraInfo) -> Self {
33        Self {
34            len: None,
35            partial: vec![],
36            extra_info,
37        }
38    }
39}
40
41impl AccountState {
42    fn get_default_output(&self, indent: u32) -> OutputCard {
43        let hex_string = encode(&self.partial);
44
45        OutputCard {
46            card: ParserCard::Default(hex_string),
47            indent,
48        }
49    }
50}
51
52impl State for AccountState {
53    fn clone_box(&self) -> Box<dyn State> {
54        Box::new(self.clone())
55    }
56
57    fn process_field(
58        &self,
59        _input: &StateInputCompoundItem,
60        indent: u32,
61    ) -> Result<StateOutput, StateError> {
62        Ok(StateOutput::with(
63            Box::new(AccountState::new(self.extra_info.clone())),
64            indent,
65        ))
66    }
67
68    fn process_array(
69        &self,
70        input: &StateInputCompound,
71        indent: u32,
72    ) -> Result<StateOutput, StateError> {
73        Ok(StateOutput::with(
74            Box::new(AccountState {
75                len: Some(input.items_count),
76                partial: vec![],
77                extra_info: self.extra_info.clone(),
78            }),
79            indent,
80        ))
81    }
82
83    fn process_array_item(
84        &self,
85        input: &StateInputCompoundItem,
86        indent: u32,
87    ) -> Result<StateOutput, StateError> {
88        Ok(StateOutput::with(
89            Box::new(AccountState {
90                len: Some(input.items_count),
91                partial: self.partial.clone(),
92                extra_info: self.extra_info.clone(),
93            }),
94            indent,
95        ))
96    }
97
98    fn process_u8(&self, input: u8, indent: u32) -> Result<StateOutput, StateError> {
99        let mut new_partial = self.partial.clone();
100        new_partial.push(input);
101
102        let current_len = match self.len {
103            Some(l) => l,
104            _ => {
105                let card = self.get_default_output(indent);
106                return Ok(StateOutput {
107                    next_state: Box::new(DefaultState),
108                    cards: vec![card],
109                    indent,
110                });
111            }
112        };
113
114        if new_partial.len() < current_len {
115            return Ok(StateOutput::with(
116                Box::new(AccountState {
117                    len: self.len,
118                    partial: new_partial,
119                    extra_info: self.extra_info.clone(),
120                }),
121                indent,
122            ));
123        }
124
125        let card = match current_len {
126            32 => {
127                let account_id: [u8; 32] = new_partial
128                    .try_into()
129                    .map_err(|_err| StateError::BadInput("32 bytes expected".to_string()))?;
130
131                OutputCard {
132                    card: ParserCard::Id {
133                        id: AccountId32::new(account_id),
134                        base58prefix: self.extra_info.base58_prefix,
135                    },
136                    indent,
137                }
138            }
139            20 => {
140                let account_id: [u8; 20] = new_partial
141                    .try_into()
142                    .map_err(|_err| StateError::BadInput("20 bytes expected".to_string()))?;
143
144                OutputCard {
145                    card: ParserCard::Id20 {
146                        id: account_id,
147                        base58prefix: self.extra_info.base58_prefix,
148                    },
149                    indent,
150                }
151            }
152            _ => self.get_default_output(indent),
153        };
154
155        Ok(StateOutput {
156            next_state: Box::new(DefaultState),
157            cards: vec![card],
158            indent,
159        })
160    }
161}
