48/// Function to decode types that are variants of `TypeDefPrimitive` enum.
49///
50/// The function decodes only given type found_ty, removes already decoded part of input data `Vec<u8>`,
51/// and returns whatever remains as `DecodedOut` field `remaining_vector`, which is processed later separately.
52///
53/// The function takes as arguments
54/// - `found_ty` (`TypeDefPrimitive`, found in the previous iteration)
55/// - `data` (remaining `Vec<u8>` of data),
56/// - `indent` used for creating properly formatted js cards.
57///
58/// The function outputs the `DecodedOut` value in case of success.
59fn decode_type_def_primitive(
60    found_ty: &TypeDefPrimitive,
61    possible_ext: &mut Option<&mut Ext>,
62    compact_flag: bool,
63    balance_flag: bool,
64    data: &[u8],
65    indent: u32,
66    short_specs: &ShortSpecs,
67) -> Result<DecodedOut> {
68    match found_ty {
69        TypeDefPrimitive::Bool => {
70            reject_flags(compact_flag, balance_flag)?;
71            decode_known_length::<bool>(data, "bool", indent)
72        }
73        TypeDefPrimitive::Char => {
74            reject_flags(compact_flag, balance_flag)?;
75            decode_char(data, indent)
76        }
77        TypeDefPrimitive::Str => {
78            reject_flags(compact_flag, balance_flag)?;
79            decode_str(data, indent)
80        }
81        TypeDefPrimitive::U8 => decode_primitive_with_flags::<u8>(
82            data,
83            possible_ext,
84            compact_flag,
85            balance_flag,
86            "u8",
87            indent,
88            short_specs,
89        ),
90        TypeDefPrimitive::U16 => decode_primitive_with_flags::<u16>(
91            data,
92            possible_ext,
93            compact_flag,
94            balance_flag,
95            "u16",
96            indent,
97            short_specs,
98        ),
99        TypeDefPrimitive::U32 => decode_primitive_with_flags::<u32>(
100            data,
101            possible_ext,
102            compact_flag,
103            balance_flag,
104            "u32",
105            indent,
106            short_specs,
107        ),
108        TypeDefPrimitive::U64 => decode_primitive_with_flags::<u64>(
109            data,
110            possible_ext,
111            compact_flag,
112            balance_flag,
113            "u64",
114            indent,
115            short_specs,
116        ),
117        TypeDefPrimitive::U128 => decode_primitive_with_flags::<u128>(
118            data,
119            possible_ext,
120            compact_flag,
121            balance_flag,
122            "u128",
123            indent,
124            short_specs,
125        ),
126        TypeDefPrimitive::U256 => {
127            reject_flags(compact_flag, balance_flag)?;
128            decode_big256(data, false, indent)
129        }
130        TypeDefPrimitive::I8 => {
131            reject_flags(compact_flag, balance_flag)?;
132            decode_known_length::<i8>(data, "i8", indent)
133        }
134        TypeDefPrimitive::I16 => {
135            reject_flags(compact_flag, balance_flag)?;
136            decode_known_length::<i16>(data, "i16", indent)
137        }
138        TypeDefPrimitive::I32 => {
139            reject_flags(compact_flag, balance_flag)?;
140            decode_known_length::<i32>(data, "i32", indent)
141        }
142        TypeDefPrimitive::I64 => {
143            reject_flags(compact_flag, balance_flag)?;
144            decode_known_length::<i64>(data, "i64", indent)
145        }
146        TypeDefPrimitive::I128 => {
147            reject_flags(compact_flag, balance_flag)?;
148            decode_known_length::<i128>(data, "i128", indent)
149        }
150        TypeDefPrimitive::I256 => {
151            reject_flags(compact_flag, balance_flag)?;
152            decode_big256(data, true, indent)
153        }
154    }
155}

157fn reject_flags(compact_flag: bool, balance_flag: bool) -> Result<()> {
158    if compact_flag {
159        return Err(Error::Decoding(
160            ParserDecodingError::UnexpectedCompactInsides,
161        ));
162    }
163    if balance_flag {
164        return Err(Error::Decoding(ParserDecodingError::BalanceNotDescribed));
165    }
166    Ok(())
167}

169/// Function to decode `char`.
170/// `char` is always 4 byte element, so the needed length pre-determined.
171///
172/// The function decodes only `char` part, removes already decoded part of input data `Vec<u8>`,
173/// and returns whatever remains as `DecodedOut` field `remaining_vector`, which is processed later separately.
174///
175/// The function takes as arguments
176/// - data (remaining `Vec<u8>` of data),
177/// - indent used for creating properly formatted output cards.
178///
179/// The function outputs the `DecodedOut` value in case of success.
180fn decode_char(data: &[u8], indent: u32) -> Result<DecodedOut> {
181    let mut slice_to_char = data.get(0..4).ok_or(ParserDecodingError::DataTooShort)?;
182    let a = <u32>::decode(&mut slice_to_char)
183        .map_err(|_| ParserDecodingError::PrimitiveFailure("char".to_string()))?;
184    let b = char::from_u32(a)
185        .ok_or_else(|| ParserDecodingError::PrimitiveFailure("char".to_string()))?;
186    let fancy_out = vec![OutputCard {
187        card: ParserCard::Default(b.to_string()),
188        indent,
189    }];
190    let remaining_vector = (data[4..]).to_vec();
191    Ok(DecodedOut {
192        remaining_vector,
193        fancy_out,
194    })
195}

197/// Function to decode `str`.
198/// `str` is encoded as a vector of utf-converteable elements, and is therefore
199/// preluded by the number of elements as compact.
200///
201/// The function decodes only `str` part, removes already decoded part of input data `Vec<u8>`,
202/// and returns whatever remains as `DecodedOut` field `remaining_vector`, which is processed later separately.
203///
204/// The function takes as arguments
205/// - data (remaining `Vec<u8>` of data),
206/// - indent used for creating properly formatted output cards.
207///
208/// The function outputs the `DecodedOut` value in case of success.
209fn decode_str(data: &[u8], indent: u32) -> Result<DecodedOut> {
210    let pre_str = get_compact::<u32>(data)?;
211    let str_length = pre_str.compact_found as usize;
212    match pre_str.start_next_unit {
213        Some(start) => match data.get(start..start + str_length) {
214            Some(a) => {
215                let text = String::from_utf8(a.to_vec())
216                    .map_err(|_| ParserDecodingError::PrimitiveFailure("str".to_string()))?;
217                let fancy_out = vec![OutputCard {
218                    card: ParserCard::Text(text),
219                    indent,
220                }];
221                let remaining_vector = data[start + str_length..].to_vec();
222                Ok(DecodedOut {
223                    remaining_vector,
224                    fancy_out,
225                })
226            }
227            None => Err(Error::Decoding(ParserDecodingError::DataTooShort)),
228        },
229        None => {
230            if str_length != 0 {
231                Err(Error::Decoding(ParserDecodingError::DataTooShort))
232            } else {
233                let fancy_out = vec![OutputCard {
234                    card: ParserCard::Text(String::new()),
235                    indent,
236                }];
237                let remaining_vector = Vec::new();
238                Ok(DecodedOut {
239                    remaining_vector,
240                    fancy_out,
241                })
242            }
243        }
244    }
245}

247/// Function to decode `U256` and `I256`.
248/// Both `U256` and `I256` are always 32 byte element, so the needed length pre-determined.
249/// Note: both types are marked as non-existing in rust in scale-info.
250/// For decoding the elements, `le` functions from `num_bigint` crate are used.
251///
252/// The function decodes only `U256` or `I256` part, removes already decoded part of input data `Vec<u8>`,
253/// and returns whatever remains as `DecodedOut` field `remaining_vector`, which is processed later separately.
254///
255/// The function takes as arguments
256/// - data (remaining `Vec<u8>` of data),
257/// - boolean flag if the expected value is signed (signed corresponds to `I256`, unsigned corresponds to `U256`),
258/// - indent used for creating properly formatted output cards.
259///
260/// The function outputs the `DecodedOut` value in case of success.
261fn decode_big256(data: &[u8], signed: bool, indent: u32) -> Result<DecodedOut> {
262    match data.get(0..32) {
263        Some(slice_to_big256) => {
264            let fancy_out = {
265                if signed {
266                    vec![OutputCard {
267                        card: ParserCard::Default(
268                            BigInt::from_signed_bytes_le(slice_to_big256).to_string(),
269                        ),
270                        indent,
271                    }]
272                }
273                // I256
274                else {
275                    vec![OutputCard {
276                        card: ParserCard::Default(
277                            BigUint::from_bytes_le(slice_to_big256).to_string(),
278                        ),
279                        indent,
280                    }]
281                } // `U256`
282            };
283            let remaining_vector = (data[32..]).to_vec();
284            Ok(DecodedOut {
285                remaining_vector,
286                fancy_out,
287            })
288        }
289        None => Err(Error::Decoding(ParserDecodingError::DataTooShort)),
290    }
291}

301fn check_special(current_type: &Type<PortableForm>) -> SpecialType {
302    match current_type.path.ident() {
303        Some(a) => match a.as_str() {
304            "AccountId32" => SpecialType::AccountId,
305            "AccountId20" => SpecialType::AccountId20,
306            "Call" => SpecialType::Call,
307            _ => SpecialType::None,
308        },
309        None => SpecialType::None,
310    }
311}

327#[allow(clippy::too_many_arguments)]
328pub(crate) fn decoding_sci_complete(
329    current_type: &Type<PortableForm>,
330    possible_ext: &mut Option<&mut Ext>,
331    compact_flag: bool,
332    balance_flag: bool,
333    call_expectation: &CallExpectation,
334    data: Vec<u8>,
335    meta_v14: &RuntimeMetadataV14,
336    indent: u32,
337    short_specs: &ShortSpecs,
338) -> Result<DecodedOut> {
339    if let Some(ext) = possible_ext {
340        ext.check_special(current_type)
341    }
342    let special_type = check_special(current_type);
343    let call_expectation = {
344        if let SpecialType::Call = special_type {
345            call_expectation.add()
346        } else {
347            CallExpectation::None
348        }
349    };
350    match special_type {
351        SpecialType::AccountId => special_case_account_id(data, indent, short_specs),
352        SpecialType::AccountId20 => special_case_account_id_20(data, indent, short_specs),
353        _ => {
354            if let Some(ext) = possible_ext {
355                if let SpecialExt::Era = ext.specialty {
356                    if ext.found_ext.era.is_some() {
357                        return Err(Error::FundamentallyBadV14Metadata(
358                            ParserMetadataError::EraTwice,
359                        ));
360                    } else {
361                        return special_case_era(data, &mut ext.found_ext, indent);
362                    }
363                }
364            }
365            match &current_type.type_def {
366                TypeDef::Composite(x) => decode_type_def_composite(
367                    x,
368                    possible_ext,
369                    compact_flag,
370                    balance_flag,
371                    data,
372                    meta_v14,
373                    indent,
374                    short_specs,
375                ),
376                TypeDef::Variant(x) => {
377                    reject_flags(compact_flag, balance_flag)?;
378                    decode_type_def_variant(
379                        x,
380                        possible_ext,
381                        &call_expectation,
382                        data,
383                        meta_v14,
384                        indent,
385                        short_specs,
386                    )
387                }
388                TypeDef::Sequence(x) => {
389                    if compact_flag {
390                        return Err(Error::Decoding(
391                            ParserDecodingError::UnexpectedCompactInsides,
392                        ));
393                    }
394                    let inner_type = meta_v14
395                        .types
396                        .resolve(x.type_param.id)
397                        .ok_or(ParserDecodingError::V14TypeNotResolved)?;
398                    decode_type_def_sequence(
399                        inner_type,
400                        possible_ext,
401                        balance_flag,
402                        &call_expectation,
403                        data,
404                        meta_v14,
405                        indent,
406                        short_specs,
407                    )
408                }
409                TypeDef::Array(x) => {
410                    if let Some(ext) = possible_ext {
411                        if let SpecialExt::Hash(ref hash) = ext.specialty {
412                            match hash {
413                                Hash::GenesisHash => {
414                                    if ext.found_ext.genesis_hash.is_some() {
415                                        return Err(Error::FundamentallyBadV14Metadata(
416                                            ParserMetadataError::GenesisHashTwice,
417                                        ));
418                                    }
419                                }
420                                Hash::BlockHash => {
421                                    if ext.found_ext.block_hash.is_some() {
422                                        return Err(Error::FundamentallyBadV14Metadata(
423                                            ParserMetadataError::BlockHashTwice,
424                                        ));
425                                    }
426                                }
427                            }
428                            return special_case_hash(
429                                data,
430                                &mut ext.found_ext,
431                                indent,
432                                short_specs,
433                                hash,
434                            );
435                        }
436                    }
437                    if compact_flag {
438                        return Err(Error::Decoding(
439                            ParserDecodingError::UnexpectedCompactInsides,
440                        ));
441                    }
442                    let inner_type = meta_v14
443                        .types
444                        .resolve(x.type_param.id)
445                        .ok_or(ParserDecodingError::V14TypeNotResolved)?;
446                    decode_type_def_array(
447                        inner_type,
448                        x.len,
449                        possible_ext,
450                        balance_flag,
451                        data,
452                        meta_v14,
453                        indent,
454                        short_specs,
455                    )
456                }
457                TypeDef::Tuple(x) => {
458                    if compact_flag {
459                        return Err(Error::Decoding(
460                            ParserDecodingError::UnexpectedCompactInsides,
461                        ));
462                    }
463                    let id_set = x.fields.iter().map(|a| a.id).collect();
464                    decode_type_def_tuple(
465                        id_set,
466                        possible_ext,
467                        balance_flag,
468                        data,
469                        meta_v14,
470                        indent,
471                        short_specs,
472                    )
473                }
474                TypeDef::Primitive(x) => decode_type_def_primitive(
475                    x,
476                    possible_ext,
477                    compact_flag,
478                    balance_flag,
479                    &data,
480                    indent,
481                    short_specs,
482                ),
483                TypeDef::Compact(x) => {
484                    let inner_type = meta_v14
485                        .types
486                        .resolve(x.type_param.id)
487                        .ok_or(ParserDecodingError::V14TypeNotResolved)?;
488                    let compact_flag = true;
489                    decoding_sci_complete(
490                        inner_type,
491                        possible_ext,
492                        compact_flag,
493                        balance_flag,
494                        &CallExpectation::None,
495                        data,
496                        meta_v14,
497                        indent,
498                        short_specs,
499                    )
500                }
501                TypeDef::BitSequence(x) => {
502                    reject_flags(compact_flag, balance_flag)?;
503                    decode_type_def_bit_sequence(x, data, meta_v14, indent)
504                }
505            }
506        }
507    }
508}

510pub(crate) fn decoding_sci_entry_point(
511    mut data: Vec<u8>,
512    meta_v14: &RuntimeMetadataV14,
513    mut indent: u32,
514    short_specs: &ShortSpecs,
515) -> Result<DecodedOut> {
516    let pallet_index: u8 = *data.first().ok_or(ParserDecodingError::DataTooShort)?;
517
518    let mut found_call_type: Option<u32> = None;
519    let mut found_pallet_name: Option<String> = None;
520    for x in meta_v14.pallets.iter() {
521        if x.index == pallet_index {
522            found_pallet_name = Some(x.name.to_string());
523            if let Some(a) = &x.calls {
524                found_call_type = Some(a.ty.id);
525            }
526            break;
527        }
528    }
529    let pallet_name = found_pallet_name.ok_or(ParserDecodingError::PalletNotFound(pallet_index))?;
530    let type_id =
531        found_call_type.ok_or_else(|| ParserDecodingError::NoCallsInPallet(pallet_name.clone()))?;
532    let (current_type, _, _) = type_path_docs(meta_v14, type_id)?;
533
534    let mut fancy_out = vec![OutputCard {
535        card: ParserCard::Pallet(pallet_name),
536        indent,
537    }];
538    indent += 1;
539    data = data[1..].to_vec();
540
541    let compact_flag = false;
542    let balance_flag = false;
543    let decoded_out = decoding_sci_complete(
544        &current_type,
545        &mut None,
546        compact_flag,
547        balance_flag,
548        &CallExpectation::Pallet,
549        data,
550        meta_v14,
551        indent,
552        short_specs,
553    )?;
554    fancy_out.extend_from_slice(&decoded_out.fancy_out);
555
556    Ok(DecodedOut {
557        remaining_vector: decoded_out.remaining_vector,
558        fancy_out,
559    })
560}

562#[allow(clippy::too_many_arguments)]
563fn decode_type_def_sequence(
564    inner_type: &Type<PortableForm>,
565    possible_ext: &mut Option<&mut Ext>,
566    balance_flag: bool,
567    call_expectation: &CallExpectation,
568    mut data: Vec<u8>,
569    meta_v14: &RuntimeMetadataV14,
570    indent: u32,
571    short_specs: &ShortSpecs,
572) -> Result<DecodedOut> {
573    let pre_vector = get_compact::<u32>(&data)?;
574    let mut fancy_output_prep: Vec<OutputCard> = Vec::new();
575    let elements_of_vector = pre_vector.compact_found;
576    match pre_vector.start_next_unit {
577        Some(start) => {
578            data = data[start..].to_vec();
579            for _i in 0..elements_of_vector {
580                let compact_flag = false;
581                let after_run = decoding_sci_complete(
582                    inner_type,
583                    possible_ext,
584                    compact_flag,
585                    balance_flag,
586                    call_expectation,
587                    data,
588                    meta_v14,
589                    indent,
590                    short_specs,
591                )?;
592                fancy_output_prep.extend_from_slice(&after_run.fancy_out);
593                data = after_run.remaining_vector;
594            }
595            Ok(DecodedOut {
596                remaining_vector: data,
597                fancy_out: fancy_output_prep,
598            })
599        }
600        None => {
601            if elements_of_vector != 0 {
602                Err(Error::Decoding(ParserDecodingError::DataTooShort))
603            } else {
604                Ok(DecodedOut {
605                    remaining_vector: Vec::new(),
606                    fancy_out: vec![OutputCard {
607                        card: ParserCard::Default(String::new()),
608                        indent,
609                    }],
610                })
611            }
612        }
613    }
614}

650fn decode_type_def_tuple(
651    id_set: Vec<u32>,
652    possible_ext: &mut Option<&mut Ext>,
653    balance_flag: bool,
654    mut data: Vec<u8>,
655    meta_v14: &RuntimeMetadataV14,
656    indent: u32,
657    short_specs: &ShortSpecs,
658) -> Result<DecodedOut> {
659    let mut fancy_out: Vec<OutputCard> = Vec::new();
660    for (i, type_id) in id_set.iter().enumerate() {
661        let (inner_type, path, docs) = type_path_docs(meta_v14, *type_id)?;
662        fancy_out.push(OutputCard {
663            card: ParserCard::FieldNumber {
664                number: i + 1,
665                docs_field_number: String::new(),
666                path_type: path,
667                docs_type: docs,
668            },
669            indent,
670        });
671        let compact_flag = false;
672        let after_run = decoding_sci_complete(
673            &inner_type,
674            possible_ext,
675            compact_flag,
676            balance_flag,
677            &CallExpectation::None,
678            data,
679            meta_v14,
680            indent,
681            short_specs,
682        )?;
683        fancy_out.extend_from_slice(&after_run.fancy_out);
684        data = after_run.remaining_vector;
685    }
686    Ok(DecodedOut {
687        remaining_vector: data,
688        fancy_out,
689    })
690}

692struct IsOptionBool {
693    is_option: bool,
694    is_bool: bool,
695}
696
697fn is_option_bool(
698    found_ty: &TypeDefVariant<PortableForm>,
699    meta_v14: &RuntimeMetadataV14,
700) -> IsOptionBool {
701    let mut got_len = false;
702    let mut got_none = false;
703    let mut got_some = false;
704    let mut is_bool = false;
705    if found_ty.variants.len() == 2 {
706        got_len = true;
707        for x in found_ty.variants.iter() {
708            if x.name.as_str() == "None" {
709                got_none = true;
710            }
711            if x.name.as_str() == "Some" {
712                got_some = true;
713                let fields = &x.fields;
714                if fields.len() == 1 {
715                    let option_type_id = fields[0].ty.id;
716                    match meta_v14.types.resolve(option_type_id) {
717                        Some(a) => {
718                            if let TypeDef::Primitive(TypeDefPrimitive::Bool) = a.type_def {
719                                is_bool = true
720                            } else {
721                                break;
722                            }
723                        }
724                        None => break,
725                    }
726                } else {
727                    break;
728                }
729            }
730        }
731    }
732    IsOptionBool {
733        is_option: got_len && got_none && got_some,
734        is_bool,
735    }
736}

738fn decode_type_def_variant(
739    found_ty: &TypeDefVariant<PortableForm>,
740    possible_ext: &mut Option<&mut Ext>,
741    call_expectation: &CallExpectation,
742    mut data: Vec<u8>,
743    meta_v14: &RuntimeMetadataV14,
744    indent: u32,
745    short_specs: &ShortSpecs,
746) -> Result<DecodedOut> {
747    let enum_index = *data.first().ok_or(ParserDecodingError::DataTooShort)?;
748
749    let check = is_option_bool(found_ty, meta_v14);
750    if check.is_option {
751        if check.is_bool {
752            let fancy_out = match enum_index {
753                0 => vec![OutputCard {
754                    card: ParserCard::None,
755                    indent,
756                }],
757                1 => vec![OutputCard {
758                    card: ParserCard::Default(String::from("True")),
759                    indent,
760                }],
761                2 => vec![OutputCard {
762                    card: ParserCard::Default(String::from("False")),
763                    indent,
764                }],
765                _ => {
766                    return Err(Error::Decoding(
767                        ParserDecodingError::UnexpectedOptionVariant,
768                    ))
769                }
770            };
771            let remaining_vector = data[1..].to_vec();
772            Ok(DecodedOut {
773                remaining_vector,
774                fancy_out,
775            })
776        } else {
777            match enum_index {
778                0 => {
779                    let fancy_out = vec![OutputCard {
780                        card: ParserCard::None,
781                        indent,
782                    }];
783                    let remaining_vector = data[1..].to_vec();
784                    Ok(DecodedOut {
785                        remaining_vector,
786                        fancy_out,
787                    })
788                }
789                1 => {
790                    if data.len() == 1 {
791                        return Err(Error::Decoding(ParserDecodingError::DataTooShort));
792                    }
793                    data = data[1..].to_vec();
794                    let found_variant = &found_ty.variants[1];
795                    let compact_flag = false;
796                    let balance_flag = false;
797                    process_fields(
798                        &found_variant.fields,
799                        possible_ext,
800                        &CallExpectation::None,
801                        compact_flag,
802                        balance_flag,
803                        data,
804                        meta_v14,
805                        indent,
806                        short_specs,
807                    )
808                }
809                _ => Err(Error::Decoding(
810                    ParserDecodingError::UnexpectedOptionVariant,
811                )),
812            }
813        }
814    } else {
815        let mut found_variant = None;
816        for x in found_ty.variants.iter() {
817            if x.index == enum_index {
818                found_variant = Some(x);
819                break;
820            }
821        }
822        let found_variant = match found_variant {
823            Some(a) => a,
824            None => return Err(Error::Decoding(ParserDecodingError::UnexpectedEnumVariant)),
825        };
826        let mut variant_docs = String::new();
827        for (i, x) in found_variant.docs.iter().enumerate() {
828            if i > 0 {
829                variant_docs.push('\n');
830            }
831            variant_docs.push_str(x);
832        }
833        let mut fancy_out = match call_expectation {
834            CallExpectation::None => vec![OutputCard {
835                card: ParserCard::EnumVariantName {
836                    name: found_variant.name.to_string(),
837                    docs_enum_variant: variant_docs,
838                },
839                indent,
840            }],
841            CallExpectation::Pallet => vec![OutputCard {
842                card: ParserCard::Pallet(found_variant.name.to_string()),
843                indent,
844            }],
845            CallExpectation::Method => vec![OutputCard {
846                card: ParserCard::Method {
847                    method_name: found_variant.name.to_string(),
848                    docs: variant_docs,
849                },
850                indent,
851            }],
852        };
853        data = data[1..].to_vec();
854
855        let compact_flag = false;
856        let balance_flag = false;
857        let fields_processed = process_fields(
858            &found_variant.fields,
859            possible_ext,
860            call_expectation,
861            compact_flag,
862            balance_flag,
863            data,
864            meta_v14,
865            indent + 1,
866            short_specs,
867        )?;
868        fancy_out.extend_from_slice(&fields_processed.fancy_out);
869        data = fields_processed.remaining_vector;
870
871        Ok(DecodedOut {
872            remaining_vector: data,
873            fancy_out,
874        })
875    }
876}

878#[allow(clippy::too_many_arguments)]
879fn process_fields(
880    fields: &[Field<PortableForm>],
881    possible_ext: &mut Option<&mut Ext>,
882    call_expectation: &CallExpectation,
883    compact_flag: bool,
884    mut balance_flag: bool,
885    mut data: Vec<u8>,
886    meta_v14: &RuntimeMetadataV14,
887    indent: u32,
888    short_specs: &ShortSpecs,
889) -> Result<DecodedOut> {
890    let mut indent_skipped = false;
891    let mut field_is_str = false;
892    let mut fancy_out: Vec<OutputCard> = Vec::new();
893    for (i, x) in fields.iter().enumerate() {
894        let mut field_docs = String::new();
895        for (j, y) in x.docs.iter().enumerate() {
896            if j > 0 {
897                field_docs.push('\n');
898            }
899            field_docs.push_str(y);
900        }
901        let (inner_type, path_type, docs_type) = type_path_docs(meta_v14, x.ty.id)?;
902        match &x.name {
903            Some(field_name) => {
904                fancy_out.push(OutputCard {
905                    card: ParserCard::FieldName {
906                        name: field_name.to_string(),
907                        docs_field_name: field_docs,
908                        path_type,
909                        docs_type,
910                    },
911                    indent,
912                });
913                if (field_name == "remark") || (field_name == "remark_with_event") {
914                    field_is_str = true;
915                }
916            }
917            None => {
918                if fields.len() > 1 {
919                    fancy_out.push(OutputCard {
920                        card: ParserCard::FieldNumber {
921                            number: i,
922                            docs_field_number: field_docs,
923                            path_type,
924                            docs_type,
925                        },
926                        indent,
927                    });
928                } else {
929                    indent_skipped = true;
930                }
931            }
932        }
933        balance_flag = match &x.type_name {
934            Some(a) => field_type_name_is_balance(a),
935            None => balance_flag,
936        };
937        let indent = {
938            if indent_skipped {
939                indent
940            } else {
941                indent + 1
942            }
943        };
944        let after_run = {
945            if field_is_str {
946                decode_str(&data, indent)?
947            } else {
948                decoding_sci_complete(
949                    &inner_type,
950                    possible_ext,
951                    compact_flag,
952                    balance_flag,
953                    call_expectation,
954                    data,
955                    meta_v14,
956                    indent,
957                    short_specs,
958                )?
959            }
960        };
961        fancy_out.extend_from_slice(&after_run.fancy_out);
962        data = after_run.remaining_vector;
963    }
964    Ok(DecodedOut {
965        remaining_vector: data,
966        fancy_out,
967    })
968}

971fn decode_type_def_composite(
972    composite_ty: &TypeDefComposite<PortableForm>,
973    possible_ext: &mut Option<&mut Ext>,
974    compact_flag: bool,
975    balance_flag: bool,

977    meta_v14: &RuntimeMetadataV14,

992        data,
993        meta_v14,

995        short_specs,
996    )

999fn decode_type_def_bit_sequence(
1000    bit_ty: &TypeDefBitSequence<PortableForm>,
1001    data: Vec<u8>,
1002    meta_v14: &RuntimeMetadataV14,
1003    indent: u32,
1004) -> Result<DecodedOut> {
1005    let pre_bitvec = get_compact::<u32>(&data)?;
1006    let actual_length = match pre_bitvec.compact_found % 8 {
1007        0 => pre_bitvec.compact_found / 8,
1008        _ => (pre_bitvec.compact_found / 8) + 1,
1009    };
1010    match pre_bitvec.start_next_unit {
1011        Some(start) => {
1012            let fin = start + (actual_length as usize);
1013            let into_bv_decode = match data.get(..fin) {
1014                Some(a) => a.to_vec(),
1015                None => return Err(Error::Decoding(ParserDecodingError::DataTooShort)),
1016            };
1017            let bitorder_type_id = bit_ty.bit_order_type.id;
1018            let bitorder_type = match meta_v14.types.resolve(bitorder_type_id) {
1019                Some(a) => a,
1020                None => return Err(Error::Decoding(ParserDecodingError::V14TypeNotResolved)),
1021            };
1022            let bitorder = match bitorder_type.type_def {
1023                TypeDef::Composite(_) => match bitorder_type.path.ident() {
1024                    Some(x) => match x.as_str() {
1025                        "Lsb0" => FoundBitOrder::Lsb0,
1026                        "Msb0" => FoundBitOrder::Msb0,
1027                        _ => return Err(Error::Decoding(ParserDecodingError::NotBitOrderType)),
1028                    },
1029                    None => return Err(Error::Decoding(ParserDecodingError::NotBitOrderType)),
1030                },
1031                _ => return Err(Error::Decoding(ParserDecodingError::NotBitOrderType)),
1032            };
1033
1034            let bitstore_type_id = bit_ty.bit_store_type.id;
1035            let bitstore_type = match meta_v14.types.resolve(bitstore_type_id) {
1036                Some(a) => a,
1037                None => return Err(Error::Decoding(ParserDecodingError::V14TypeNotResolved)),
1038            };
1039            let card_prep = match &bitstore_type.type_def {
1040                TypeDef::Primitive(a) => {
1041                    match a {
1042                        TypeDefPrimitive::U8 => process_bitvec::<u8>(bitorder, into_bv_decode)?,
1043                        TypeDefPrimitive::U16 => process_bitvec::<u16>(bitorder, into_bv_decode)?,
1044                        TypeDefPrimitive::U32 => process_bitvec::<u32>(bitorder, into_bv_decode)?,
1045                        // this should not be here, but due to possible architecture limitations u64 will not compile on 32-bit architectures
1046                        // ideally, should be patched by `#[repr(C, align(8))]` thing similar to `BitVec` issue 76
1047                        // TypeDefPrimitive::U64 => process_bitvec::<u64> (bitorder, into_bv_decode)?,
1048                        TypeDefPrimitive::U64 => match bitorder {
1049                            FoundBitOrder::Lsb0 => ugly_patch_u64::<Lsb0>(into_bv_decode)?,
1050                            FoundBitOrder::Msb0 => ugly_patch_u64::<Msb0>(into_bv_decode)?,
1051                        },
1052                        _ => return Err(Error::Decoding(ParserDecodingError::NotBitStoreType)),
1053                    }
1054                }
1055                _ => return Err(Error::Decoding(ParserDecodingError::NotBitStoreType)),
1056            };
1057
1058            let fancy_out = vec![OutputCard {
1059                card: ParserCard::BitVec(card_prep),
1060                indent,
1061            }];
1062            let remaining_vector = data[fin..].to_vec();
1063            Ok(DecodedOut {
1064                remaining_vector,
1065                fancy_out,
1066            })
1067        }
1068        None => {
1069            if actual_length != 0 {
1070                return Err(Error::Decoding(ParserDecodingError::DataTooShort));
1071            }
1072            Ok(DecodedOut {
1073                remaining_vector: Vec::new(),
1074                fancy_out: vec![OutputCard {
1075                    card: ParserCard::Default(String::new()),
1076                    indent,
1077                }],
1078            })
1079        }
1080    }
1081}

1083fn process_bitvec<T: BitStore + Decode>(
1084    bitorder: FoundBitOrder,
1085    into_bv_decode: Vec<u8>,
1086) -> Result<String> {
1087    match bitorder {
1088        FoundBitOrder::Lsb0 => match <BitVec<T, Lsb0>>::decode(&mut &into_bv_decode[..]) {
1089            Ok(b) => Ok(b.to_string()),
1090            Err(_) => Err(Error::Decoding(ParserDecodingError::BitVecFailure)),
1091        },
1092        FoundBitOrder::Msb0 => match <BitVec<T, Msb0>>::decode(&mut &into_bv_decode[..]) {
1093            Ok(b) => Ok(b.to_string()),
1094            Err(_) => Err(Error::Decoding(ParserDecodingError::BitVecFailure)),
1095        },
1096    }
1097}

1099fn ugly_patch_u64<O: BitOrder>(into_bv_decode: Vec<u8>) -> Result<String> {
1100    let bitvec_decoded = match <BitVec<u32, O>>::decode(&mut &into_bv_decode[..]) {
1101        Ok(b) => b,
1102        Err(_) => return Err(Error::Decoding(ParserDecodingError::BitVecFailure)),
1103    };
1104    let vec = bitvec_decoded.into_vec();
1105    let mut out = String::from("[");
1106    for i in 0..vec.len() / 2 {
1107        if i > 0 {
1108            out.push_str(", ");
1109        }
1110        let print1 = BitVec::<u32, O>::from_vec(vec![vec[2 * i]]).to_string();
1111        let print2 = BitVec::<u32, O>::from_vec(vec![vec[2 * i + 1]]).to_string();
1112        let _ = write!(
1113            &mut out,
1114            "{}{}",
1115            &print1[1..print1.len() - 1],
1116            &print2[1..print2.len() - 1]
1117        );
1118    }
1119    out.push(']');
1120    Ok(out)
1121}

1123fn type_path_docs(
1124    meta_v14: &RuntimeMetadataV14,
1125    type_id: u32,
1126) -> Result<(Type<PortableForm>, String, String)> {
1127    let current_type = meta_v14
1128        .types
1129        .resolve(type_id)
1130        .ok_or(ParserDecodingError::V14TypeNotResolved)?;
1131    let mut docs = String::new();
1132    for (i, x) in current_type.docs.iter().enumerate() {
1133        if i > 0 {
1134            docs.push('\n');
1135        }
1136        docs.push_str(x);
1137    }
1138    let mut path = String::new();
1139    for (i, x) in current_type.path.segments.iter().enumerate() {
1140        if i > 0 {
1141            path.push_str(" >> ");
1142        }
1143        path.push_str(x);
1144    }
1145    Ok((current_type.to_owned(), path, docs))
