20pub fn process_derivations(database: &sled::Db, data_hex: &str) -> Result<TransactionAction> {
21    let data = unhex(data_hex)?;
22    let export_info = <ExportAddrs>::decode(&mut &data[3..])?;
23    let import_info = prepare_derivations_preview(database, export_info)?;
24    let derivations_card = Card::Derivations(&import_info).card(&mut 0, 0);
25    Ok(TransactionAction::Derivations {
26        content: Box::new(TransactionCardSet {
27            importing_derivations: Some(vec![derivations_card]),
28            ..Default::default()
29        }),
30    })
31}

33pub fn prepare_derivations_preview(
34    database: &sled::Db,
35    export_info: ExportAddrs,

40    }

43fn prepare_derivations_v1(
44    database: &sled::Db,
45    export_info: ExportAddrsV1,
46) -> Result<Vec<SeedKeysPreview>> {
47    let mut result = Vec::new();
48    let all_addrs = get_all_addresses(database)?;
49    for seed_info in export_info.addrs {
50        let mut derived_keys = vec![];
51        for addr_info in seed_info.derived_keys {
52            let multisigner = base58_or_eth_pubkey_to_multisigner(
53                &addr_info.address_or_pubkey,
54                &addr_info.encryption,
55            )?;
56            let identicon = make_identicon_from_multisigner(
57                &multisigner,
58                addr_info.encryption.identicon_style(),
59            );
60            let network_specs_key =
61                NetworkSpecsKey::from_parts(&addr_info.genesis_hash, &addr_info.encryption);
62            let network_title = get_network_specs(database, &network_specs_key)
63                .map(|specs| specs.specs.title)
64                .ok();
65            let path = addr_info.derivation_path.clone().unwrap_or_default();
66            let status = get_derivation_status(
67                database,
68                &path,
69                &network_title,
70                &Some(network_specs_key),
71                &seed_info.multisigner,
72                &multisigner,
73            )?;
74            derived_keys.push(DerivedKeyPreview {
75                address: addr_info.address_or_pubkey.clone(),
76                derivation_path: addr_info.derivation_path,
77                identicon,
78                has_pwd: None, // unknown at this point
79                genesis_hash: addr_info.genesis_hash,
80                encryption: addr_info.encryption,
81                network_title,
82                status,
83            })
84        }
85        let looking_for_ms = seed_info.multisigner.clone();
86        let name = all_addrs
87            .iter()
88            .find(|(m, a)| m == &looking_for_ms && a.network_id.is_none())
89            .map(|(_, a)| a.seed_name.clone());
90
91        if let Some(name) = name {
92            result.push(SeedKeysPreview {
93                name,
94                multisigner: seed_info.multisigner,
95                derived_keys,
96            })
97        } else {
98            log::error!("seed not found for {:?}", seed_info.multisigner);
99        }
100    }
101    Ok(result)
102}

111fn get_derivation_status(
112    database: &sled::Db,
113    path: &str,
114    network_title: &Option<String>,
115    network_specs_key: &Option<NetworkSpecsKey>,
116    seed_multisigner: &MultiSigner,
117    key_multisigner: &MultiSigner,
118) -> Result<DerivedKeyStatus> {
119    let mut seed_found = false;
120    // FIXME: nested loop, should be optimized
121    for (m, address_details) in get_all_addresses(database)?.into_iter() {
122        if m == *seed_multisigner {
123            seed_found = true;
124        }
125        if m == *key_multisigner && address_details.network_id == *network_specs_key {
126            return Ok(DerivedKeyStatus::AlreadyExists);
127        }
128    }
129
130    let mut errors = vec![];
131    if is_passworded(path).is_err() {
132        errors.push(DerivedKeyError::BadFormat);
133    }
134    if network_title.is_none() {
135        errors.push(DerivedKeyError::NetworkMissing);
136    }
137    if !seed_found {
138        errors.push(DerivedKeyError::KeySetMissing);
139    }
140    if !errors.is_empty() {
141        return Ok(DerivedKeyStatus::Invalid { errors });
142    }
143    Ok(DerivedKeyStatus::Importable)
144}
