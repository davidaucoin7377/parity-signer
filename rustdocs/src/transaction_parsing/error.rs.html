151    /// - `02` for `Ecdsa`
152    ///
153    /// In signable transaction the encryption algorithm corresponds to the
154    /// encryption associated with the address that generated the transaction
155    /// and can sign it (and thus to the encryption supported by the network
156    /// in which the transaction is generated).

599fn display_parsing_errors(network_name: &str, errors: &[(u32, parser::Error)]) -> String {
600    let mut insert = String::new();
601    for (i, (version, parser_error)) in errors.iter().enumerate() {
602        if i > 0 {
603            insert.push(' ')
604        }
605        let _ = write!(
606            &mut insert,
607            "Parsing with {network_name}{version} metadata: {parser_error}",
608        );
609    }
610
611    insert
612}

614fn display_general_verifier_for_content(g: &GeneralVerifierForContent) -> String {
615    match g {
616        GeneralVerifierForContent::Network { name } => format!("{name} network information"),
617        GeneralVerifierForContent::Types => String::from("types information"),
618    }
