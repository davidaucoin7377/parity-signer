1//! Atomic transactions in cold and hot databases
2//!
3//! Additions and removals of entries in cold and hot database occur through
4//! atomic [transactions](sled::transaction).
5//! Each tree gets updated with its own [`Batch`], updates occur within a single
6//! transaction.
7//!
8//! For transactions scanned into Vault, currently a temporary database entry
9//! is made to store transaction details while they are displayed to user.

44/// Cold database transaction data containing [`Batch`] elements that will be
45/// applied to each [`Tree`](sled::Tree).
46///
47/// Cold database tree names and content information could be found in
48/// [`constants`] crate. All trees are routinely updated as Vault is used.
49///
50/// [`TrDbCold`] is applied to the cold database in an atomic transaction.
51///
52/// [`TrDbCold`] is used both by the Vault side (for all database-related
53/// actions) and the active side (to generate and populate the cold database).
54///
55/// Note that all the checking is done as the [`TrDbCold`] is generated,
56/// `apply` method does not do any checks on its own.
57#[derive(Debug)]
58pub struct TrDbCold {
59    /// `Batch` to be applied to [`ADDRTREE`] tree
60    for_addresses: Batch,
61
62    /// `Batch` to be applied to [`HISTORY`] tree
63    for_history: Batch,
64
65    /// `Batch` to be applied to [`METATREE`] tree
66    for_metadata: Batch,
67
68    /// `Batch` to be applied to [`SPECSTREE`] tree
69    for_network_specs: Batch,
70
71    /// `Batch` to be applied to [`SETTREE`] tree
72    for_settings: Batch,
73
74    /// `Batch` to be applied to [`TRANSACTION`] tree
75    for_transaction: Batch,
76
77    /// `Batch` to be applied to [`VERIFIERS`] tree
78    for_verifiers: Batch,
79}

144    /// Apply constructed set of batches within [`TrDbCold`] to the database
145    /// with a given name, in a single transaction.
146    ///
147    /// Note that both `ErrorSource` variants are available.
148    pub fn apply(&self, database: &sled::Db) -> Result<()> {
149        let addresses = open_tree(database, ADDRTREE)?;
150        let history = open_tree(database, HISTORY)?;
151        let metadata = open_tree(database, METATREE)?;
152        let network_specs = open_tree(database, SPECSTREE)?;
153        let settings = open_tree(database, SETTREE)?;
154        let transaction = open_tree(database, TRANSACTION)?;
155        let verifiers = open_tree(database, VERIFIERS)?;
156        let s = (
157            &addresses,
158            &history,
159            &metadata,
160            &network_specs,
161            &settings,
162            &transaction,
163            &verifiers,
164        );
165        let res: TransactionResult<(), sled::Error> = s.transaction(
166            |(
167                tx_addresses,
168                tx_history,
169                tx_metadata,
170                tx_network_specs,
171                tx_settings,
172                tx_transaction,
173                tx_verifiers,
174            )| {
175                tx_addresses.apply_batch(&self.for_addresses)?;
176                tx_addresses.flush();
177                tx_history.apply_batch(&self.for_history)?;
178                tx_history.flush();
179                tx_metadata.apply_batch(&self.for_metadata)?;
180                tx_metadata.flush();
181                tx_network_specs.apply_batch(&self.for_network_specs)?;
182                tx_network_specs.flush();
183                tx_settings.apply_batch(&self.for_settings)?;
184                tx_settings.flush();
185                tx_transaction.apply_batch(&self.for_transaction)?;
186                tx_transaction.flush();
187                tx_verifiers.apply_batch(&self.for_verifiers)?;
188                tx_verifiers.flush();
189                Ok(())
190            },
191        );
192
193        Ok(res?)
194    }
195}

336/// SCALE-encodeable draft for [`Batch`], that will be a part of database atomic
337/// transaction.
338///
339/// [`Batch`] does not support SCALE-encoding, so [`BatchStub`] is constructed
340/// from a set of keys to be removed from the database and a set of (key, value)
341/// pairs to be added into the database. Keys and values are SCALE-compatible
342/// `Vec<u8>`.
343///
344/// When applying [`BatchStub`], i.e. transforming it into [`Batch`], the
345/// removals are always applied before additions, to avoid accidental replacing
346/// of just added value.
347#[derive(Debug, Decode, Encode)]
348struct BatchStub {
349    /// Vector of keys to be removed from the database.
350    removals: Vec<Vec<u8>>,
351
352    /// Vector of (key, value) pairs to be added into the database.
353    additions: Vec<(Vec<u8>, Vec<u8>)>,
354}
355
356impl BatchStub {
357    /// Generate empty [`BatchStub`].
358    fn empty() -> Self {
359        Self {
360            removals: Vec::new(),
361            additions: Vec::new(),
362        }
363    }
364
365    /// Transform [`BatchStub`] into [`Batch`], removals first.
366    fn make_batch(&self) -> Batch {
367        self.extend_batch(Batch::default())
368    }
369
370    /// Add elements from [`BatchStub`], removals first, in queue after
371    /// instructions already present in input [`Batch`]
372    fn extend_batch(&self, batch: Batch) -> Batch {
373        let mut out = batch;
374        for key in self.removals.iter() {
375            out.remove(&key[..])
376        }
377        for (key, value) in self.additions.iter() {
378            out.insert(&key[..], &value[..])
379        }
380        out
381    }
382
383    /// Add a new addition element into [`BatchStub`] `additions` queue.
384    fn new_addition(mut self, key: Vec<u8>, value: Vec<u8>) -> Self {
385        self.additions.push((key, value));
386        self
387    }
388
389    /// Add a new removal element into [`BatchStub`] `removals` queue.
390    fn new_removal(mut self, key: Vec<u8>) -> Self {
391        self.removals.push(key);
392        self
393    }
394}

396/// Draft for cold database atomic transaction, constructed for Vault update
397/// transaction (`add_specs`, `load_metadata`, `load_types`).
398///
399/// [`TrDbColdStub`] is stored SCALE-encoded in [`TRANSACTION`] tree
400/// of the cold database under key [`STUB`] while the update is considered by
401/// the user. Draft is applied atomically to the cold database if the update is
402/// accepted.
403///
404/// Accepting an update could result in adding or removing database data.
405///
406/// [`TrDbColdStub`] contains [`Event`] set for [`HISTORY`] tree update and
407/// `BatchStub` update drafts with corresponding removals and additions for
408/// database trees:
409///
410/// - [`ADDRTREE`]
411/// - [`METATREE`]
412/// - [`SPECSTREE`]
413/// - [`SETTREE`]
414/// - [`VERIFIERS`]
415///
416/// Note that all the checking is done before the [`TrDbColdStub`] is written
417/// into [`TRANSACTION`] tree, `apply` method will check only that the checksum
418/// known to the user is the same as the one database has currently.
419#[derive(Debug, Decode, Encode)]
420pub struct TrDbColdStub {
421    /// `BatchStub` to be transformed into `Batch` for [`ADDRTREE`] tree.
422    addresses_stub: BatchStub,
423
424    /// `Vec<Event>` to be entered into [`HISTORY`] tree, the
425    /// [`Entry`](definitions::history::Entry) with a timestamp is generated
426    /// only when the payload is approved by the user.
427    history_stub: Vec<Event>,
428
429    /// `BatchStub` to be transformed into `Batch` for [`METATREE`] tree.
430    metadata_stub: BatchStub,
431
432    /// `BatchStub` to be transformed into `Batch` for [`SPECSTREE`] tree.
433    network_specs_stub: BatchStub,
434
435    /// `BatchStub` to be transformed into `Batch` for [`SETTREE`] tree.
436    settings_stub: BatchStub,
437
438    /// `BatchStub` to be transformed into `Batch` for [`VERIFIERS`] tree.
439    verifiers_stub: BatchStub,
440}
441
442impl TrDbColdStub {
443    /// Construct new empty [`TrDbColdStub`].
444    pub fn new() -> Self {
445        Self {
446            addresses_stub: BatchStub::empty(),
447            history_stub: Vec::new(),
448            metadata_stub: BatchStub::empty(),
449            network_specs_stub: BatchStub::empty(),
450            settings_stub: BatchStub::empty(),
451            verifiers_stub: BatchStub::empty(),
452        }
453    }
454
455    /// Recover [`TrDbColdStub`] from storage in the cold database.
456    ///
457    /// Function requires correct checksum to make sure the transaction is
458    /// still the one that was shown to the user previously, and no changes to
459    /// the database have occured after the atomic transaction draft was placed
460    /// into storage.
461    ///
462    /// [`TRANSACTION`] tree is cleared in the process.
463    pub fn from_storage(database: &sled::Db, checksum: u32) -> Result<Self> {
464        let stub_encoded = {
465            verify_checksum(database, checksum)?;
466            let transaction = open_tree(database, TRANSACTION)?;
467            transaction.get(STUB)?.ok_or(Error::Stub)?
468        };
469        TrDbCold::new()
470            .set_transaction(make_batch_clear_tree(database, TRANSACTION)?) // clear transaction tree
471            .apply(database)?;
472        Ok(Self::decode(&mut &stub_encoded[..])?)
473    }
474
475    /// Put SCALE-encoded [`TrDbColdStub`] into storage in the [`TRANSACTION`]
476    /// tree of the cold database under the key [`STUB`].
477    ///
478    /// Function returns `u32` checksum. This checksum is needed to recover
479    /// stored [`TrDbColdStub`] using `from_storage` method.
480    ///
481    /// The [`TRANSACTION`] tree is cleared prior to adding data to storage.
482    pub fn store_and_get_checksum(&self, database: &sled::Db) -> Result<u32> {
483        let mut transaction_batch = make_batch_clear_tree(database, TRANSACTION)?;
484        transaction_batch.insert(STUB, self.encode());
485        TrDbCold::new()
486            .set_transaction(transaction_batch) // clear transaction tree
487            .apply(database)?;
488        Ok(database.checksum()?)
489    }

491    /// Add new [`Event`] in `history_stub` field of the [`TrDbColdStub`]
492    pub fn new_history_entry(mut self, event: Event) -> Self {
493        self.history_stub.push(event);
494        self
495    }

497    /// Prepare adding the metadata received as `load_metadata` update into the
498    /// cold database:
499    ///
500    /// - Add a (key, value) pair to the metadata additions queue in
501    ///   `metadata_stub`. Key is [`MetaKey`] in key form, value is metadata in
502    ///   `Vec<u8>` format.
503    /// - Add corresponding `Event::MetadataAdded(_)` into `history_stub`.
504    pub fn add_metadata(mut self, meta_values: &MetaValues) -> Self {
505        let meta_key = MetaKey::from_parts(&meta_values.name, meta_values.version);
506        self.metadata_stub = self
507            .metadata_stub
508            .new_addition(meta_key.key(), meta_values.meta.to_vec());
509        self.history_stub.push(Event::MetadataAdded {
510            meta_values_display: MetaValuesDisplay::get(meta_values),
511        });
512        self
513    }
514
515    /// Prepare removing the metadata from the cold database:
516    ///
517    /// - Add [`MetaKey`] in key form to the metadata removals queue in
518    ///   `metadata_stub`.
519    /// - Add corresponding `Event::MetadataRemoved(_)` into `history_stub`.
520    ///
521    /// Function is used for `Hold` and `GeneralHold` processing when,
522    /// respectively, the network verifier or the general verifier is changed.
523    pub fn remove_metadata(mut self, meta_values: &MetaValues) -> Self {
524        let meta_key = MetaKey::from_parts(&meta_values.name, meta_values.version);
525        self.metadata_stub = self.metadata_stub.new_removal(meta_key.key());
526        self.history_stub.push(Event::MetadataRemoved {
527            meta_values_display: MetaValuesDisplay::get(meta_values),
528        });
529        self
530    }
531
532    /// Prepare adding [`OrderedNetworkSpecs`] into the cold database:
533    ///
534    /// - Transform received in `add_specs` payload [`NetworkSpecs`]
535    ///   into [`OrderedNetworkSpecs`] by adding `order` field. Networks are always added
536    ///   in the end of the network list, with order set to the total number of
537    ///   network specs entries currently in Vault. When a network is removed,
538    ///   the order of the remaining networks gets rearranged, see details in
539    ///   function [`remove_network`](crate::helpers::remove_network).
540    /// - Add a (key, value) pair to the network specs additions queue in
541    ///   `network_specs_stub`. Key is [`NetworkSpecsKey`] in key form, value is
542    ///   SCALE-encoded [`OrderedNetworkSpecs`].
543    /// - Add corresponding `Event::NetworkSpecsAdded(_)` into `history_stub`.
544    /// - Add root address for the network if the
545    ///   [`AddressDetails`](definitions::users::AddressDetails) entry with
546    ///   matching [`Encryption`](definitions::crypto::Encryption) already exists,
547    ///   i.e. add (key, value) pair to the address additions queue in
548    ///   `addresses_stub`. Key is
549    ///   [`AddressKey`](definitions::keyring::AddressKey)
550    ///   in key form, value is
551    ///   SCALE-encoded updated
552    ///   [`AddressDetails`](definitions::users::AddressDetails).
553    /// - If address was added, add corresponding `Event::IdentityAdded(_)`
554    ///   into `history_stub`.
555    ///
556    /// Note that `add_network_specs` does not deal with network verifiers:
557    /// verifier data is not necessarily updated each time the network
558    /// specs are added.
559    pub fn add_network_specs(
560        mut self,
561        database: &sled::Db,
562        network_specs_to_send: &NetworkSpecs,
563        valid_current_verifier: &ValidCurrentVerifier,
564        general_verifier: &Verifier,
565    ) -> Result<Self> {
566        let network_specs_key = NetworkSpecsKey::from_parts(
567            &network_specs_to_send.genesis_hash,
568            &network_specs_to_send.encryption,
569        );
570        let order = {
571            let chainspecs = open_tree(database, SPECSTREE)?;
572            chainspecs.len()
573        } as u8;
574        let network_specs = network_specs_to_send.to_store(order);
575        self.network_specs_stub = self
576            .network_specs_stub
577            .new_addition(network_specs_key.key(), network_specs.encode());
578        self.history_stub.push(Event::NetworkSpecsAdded {
579            network_specs_display: NetworkSpecsDisplay::get(
580                &network_specs,
581                valid_current_verifier,
582                general_verifier,
583            ),
584        });
585
586        Ok(self)
587    }

589    /// Prepare removing [`OrderedNetworkSpecs`] from the cold database:
590    ///
591    /// - Add [`NetworkSpecsKey`] in key form to the network specs removal queue
592    ///   in `network_specs_stub`.
593    /// - Add corresponding `Event::NetworkSpecsRemoved(_)` into `history_stub`.
594    ///
595    /// Function is used for `Hold` and `GeneralHold` processing when,
596    /// respectively, the network verifier or the general verifier is changed.
597    ///
598    /// Note that function does not deal with the verifiers nor with the
599    /// addresses.
600    ///
601    /// Verifiers remain unchanged during the hold processing.
602    ///
603    /// The addresses are not removed and will be again visible from the user
604    /// interface when the properly verified network specs are loaded in Vault.
605    pub fn remove_network_specs(
606        mut self,
607        network_specs: &OrderedNetworkSpecs,
608        valid_current_verifier: &ValidCurrentVerifier,
609        general_verifier: &Verifier,
610    ) -> Self {
611        let network_specs_key = NetworkSpecsKey::from_parts(
612            &network_specs.specs.genesis_hash,
613            &network_specs.specs.encryption,
614        );
615        self.network_specs_stub = self.network_specs_stub.new_removal(network_specs_key.key());
616        self.history_stub.push(Event::NetworkSpecsRemoved {
617            network_specs_display: NetworkSpecsDisplay::get(
618                network_specs,
619                valid_current_verifier,
620                general_verifier,
621            ),
622        });
623        self
624    }

696                valid_current_verifier,

703    /// Transform [`TrDbColdStub`] into [`TrDbCold`] and apply to the database
704    /// with a given name, in a single transaction.
705    ///
706    /// The [`TRANSACTION`] tree gets cleared in the process.
707    ///
708    /// It is unlikely that this clearing is ever doing anything, as the
709    /// intended use of the [`TrDbColdStub`] is to recover it from the database
710    /// (with clearing the [`TRANSACTION`] tree) and then immediately apply.
711    pub fn apply(self, database: &sled::Db) -> Result<()> {
712        let for_transaction = make_batch_clear_tree(database, TRANSACTION)?;
713        TrDbCold {
714            for_addresses: self.addresses_stub.make_batch(),
715            for_history: events_to_batch(database, self.history_stub)?,
716            for_metadata: self.metadata_stub.make_batch(),
717            for_network_specs: self.network_specs_stub.make_batch(),
718            for_settings: self.settings_stub.make_batch(),
719            for_transaction,
720            for_verifiers: self.verifiers_stub.make_batch(),
721        }
722        .apply(database)
723    }

733/// Temporary storage for signable transaction and associated data.
734///
735/// Signable transaction received by the Vault must always be parsed prior to
736/// signing, and when it is, [`TrDbColdSign`] is generated and the transaction
737/// details are shown to user.
738///
739/// If the user signs the transaction or tries to sign and enters wrong
740/// password, the transaction data will be recorded in Vault history log.
741///
742/// While the user considers the transaction, [`TrDbColdSign`] is stored
743/// SCALE-encoded in [`TRANSACTION`] tree of the cold database under the key
744/// [`SIGN`].
745///
746/// [`TrDbColdSign`] contains:
747///
748/// - [`SignContent`] with data to sign
749/// - name of the network in which the transaction is made
750/// - derivation path of the address used, whether the address has password,
751///   corresponding [`MultiSigner`] value
752/// - relevant history [`Event`] set: warnings that were shown during the
753///   parsing
754#[derive(Debug, Decode, Default, Encode)]
755pub struct TrDbColdSign {
756    /// Bulk of transactions to sign.
757    pub signing_bulk: Vec<TrDbColdSignOne>,
758}
759
760impl TrDbColdSign {
761    /// Recover [`TrDbColdSign`] from storage in the cold database.
762    ///
763    /// Function requires an optional correct checksum to make sure
764    /// the signable transaction is still the one that was shown to
765    /// the user previously, and no changes to the database have occurred.
766    /// While parsing a bulk no checksum is provided and no checks are done.
767    ///
768    /// [`TRANSACTION`] tree is **not** cleared in the process. User is allowed
769    /// to try entering password several times, for all this time the
770    /// transaction remains in the database.
771    pub fn from_storage(database: &sled::Db, checksum: Option<u32>) -> Result<Option<Self>> {
772        let sign_encoded = {
773            if let Some(checksum) = checksum {
774                verify_checksum(database, checksum)?;
775            }
776            let transaction = open_tree(database, TRANSACTION)?;
777            match transaction.get(SIGN)? {
778                Some(a) => a,
779                None => return Ok(None),
780            }
781        };
782        Ok(Some(Self::decode(&mut &sign_encoded[..])?))
783    }
784
785    /// Put SCALE-encoded [`TrDbColdSign`] into storage in the [`TRANSACTION`]
786    /// tree of the cold database under the key [`SIGN`].
787    ///
788    /// Function returns `u32` checksum. This checksum is needed to recover
789    /// stored [`TrDbColdSign`] using `from_storage` method.
790    ///
791    /// The [`TRANSACTION`] tree is cleared prior to adding data to storage.
792    pub fn store_and_get_checksum(&self, database: &sled::Db) -> Result<u32> {
793        let mut transaction_batch = make_batch_clear_tree(database, TRANSACTION)?;
794        transaction_batch.insert(SIGN, self.encode());
795        TrDbCold::new()
796            .set_transaction(transaction_batch) // clear transaction tree
797            .apply(database)?;
798        Ok(database.checksum()?)
799    }
800
801    /// Use [`TrDbColdSign`] to add history log data into the cold database.
802    ///
803    /// Possible history log entries are:
804    ///
805    /// - `Event::TransactionSigned(_)` and `Event::MessageSigned(_)` for the
806    ///   cases when the signature was generated and displayed through the user
807    ///   interface
808    /// - `Event::TransactionSignError(_)` and `Event::MessageSignError(_)` for
809    ///   the cases when the user has entered the wrong password and no signature
810    ///   was generated. Vault current policy is to log all wrong password entry
811    ///   attempts.
812    ///
813    /// Required input:
814    ///
815    /// - `wrong_password` flag; for entries with `true` value the signature
816    ///   was not generated, because user has entered the wrong password;
817    /// - user-added text comment for the transaction
818    /// - index of the transaction in the bulk
819    /// - database name, into which the data is added
820    ///
821    /// Function returns database checksum, to be collected and re-used in case
822    /// of wrong password entry.
823    pub fn apply(
824        self,
825        database: &sled::Db,
826        wrong_password: bool,
827        user_comment: &str,
828        idx: usize,
829    ) -> Result<u32> {
830        let mut history = vec![];
831        let mut for_transaction = Batch::default();
832        let s = &self.signing_bulk[idx];
833        let signed_by = VerifierValue::Standard { m: s.multisigner() };
834        history.append(&mut s.history.clone());
835        match &s.content {
836            SignContent::Transaction { method, extensions } => {
837                let transaction = [method.encode(), extensions.clone()].concat();
838                let sign_display =
839                    SignDisplay::get(&transaction, &s.network_name, &signed_by, user_comment);
840                if wrong_password {
841                    history.push(Event::TransactionSignError { sign_display })
842                } else {
843                    history.push(Event::TransactionSigned { sign_display });
844                    // if this is the last transaction in the bulk and the password was right,
845                    // all is done and the DB can be cleared.
846                    if idx == self.signing_bulk.len() - 1 {
847                        for_transaction = make_batch_clear_tree(database, TRANSACTION)?;
848                    }
849                }
850            }
851            SignContent::Message(message) => {
852                let hex_message = hex::encode(message);
853                let sign_message_display = SignMessageDisplay::get(
854                    &hex_message,
855                    &s.network_name,
856                    &signed_by,
857                    user_comment,
858                );
859                if wrong_password {
860                    history.push(Event::MessageSignError {
861                        sign_message_display,
862                    })
863                } else {
864                    history.push(Event::MessageSigned {
865                        sign_message_display,
866                    });
867                    // if this is the last transaction in the bulk and the password was right,
868                    // all is done and the DB can be cleared.
869                    if idx == self.signing_bulk.len() - 1 {
870                        for_transaction = make_batch_clear_tree(database, TRANSACTION)?;
871                    }
872                }
873            }
874        }
875        TrDbCold::new()
876            .set_history(events_to_batch(database, history)?)
877            .set_transaction(for_transaction)
878            .apply(database)?;
879        Ok(database.checksum()?)
880    }
881}
