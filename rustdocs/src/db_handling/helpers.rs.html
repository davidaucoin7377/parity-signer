39/// Open a tree in the database.

41/// Wrapper for `open_tree()` method for database [`Db`] from [`sled`].
42/// Input is `&[u8]` tree identifier.
43pub fn open_tree(database: &Db, tree_name: &[u8]) -> Result<Tree> {
44    Ok(database.open_tree(tree_name)?)

47/// Check database schema version.
48/// Prevents app stash when user upgrades the app without re-install.
49pub fn assert_db_version(database: &Db) -> Result<()> {
50    let expected = *SchemaVersion::current();
51    let settings = open_tree(database, SETTREE)?;
52    let ivec = settings
53        .get(SCHEMA_VERSION)?
54        .ok_or(Error::DbSchemaMismatch { expected, found: 0 })?;
55    let found = *SchemaVersion::from_ivec(&ivec)?;
56    if expected != found {
57        return Err(Error::DbSchemaMismatch { expected, found });
58    }
59    Ok(())
60}

62/// Assemble a [`Batch`] that removes all elements from a tree.
63pub fn make_batch_clear_tree(database: &sled::Db, tree_name: &[u8]) -> Result<Batch> {
64    let tree = open_tree(database, tree_name)?;
65    let mut out = Batch::default();
66    for (key, _) in tree.iter().flatten() {
67        out.remove(key)
68    }
69    Ok(out)
70}

85/// Try to get [`ValidCurrentVerifier`] from the Vault database for a network,
86/// using [`VerifierKey`].
87///
88/// If the network is not known to the database, i.e. there is no verifier data
89/// and network genesis hash is not encountered elsewhere in the database,
90/// result is `Ok(None)`.
91///
92/// Note that `CurrentVerifier::Dead` or damaged verifier data result in
93/// errors.
94pub fn try_get_valid_current_verifier(
95    database: &sled::Db,
96    verifier_key: &VerifierKey,
97) -> Result<Option<ValidCurrentVerifier>> {
98    let general_verifier = get_general_verifier(database)?;
99    let verifiers = open_tree(database, VERIFIERS)?;
100    match verifiers.get(verifier_key.key())? {
101        // verifier entry is found
102        Some(verifier_encoded) => {
103            match <CurrentVerifier>::decode(&mut &verifier_encoded[..])? {
104                // verifier is a valid one
105                CurrentVerifier::Valid(b) => {
106                    // Custom verifier ([`Verifier`]) can never be entered in
107                    // the database with same value as database general verifier
108                    // ([`Verifier`]), unless both values are `None`.
109                    // If such entry is found, it indicates that the database is
110                    // corrupted.
111                    if let ValidCurrentVerifier::Custom {
112                        v: ref custom_verifier,
113                    } = b
114                    {
115                        if (custom_verifier == &general_verifier)
116                            && (general_verifier != Verifier { v: None })
117                        {
118                            return Err(Error::CustomVerifierIsGeneral(verifier_key.to_owned()));
119                        }
120                    }
121                    Ok(Some(b))
122                }
123            }
124        }
125        // no verifier for network in the database
126        None => {
127            // `VerifierKey` is formed from network genesis hash.
128            // When the entries are added in the database, network specs could
129            // be added only if the verifier is already in the database or is
130            // added at the same time.
131            // If the genesis hash is found in network specs, but no verifier
132            // entry exists, it indicated that the database is corrupted.
133            if let Some(specs_invariants) =
134                genesis_hash_in_specs(database, verifier_key.genesis_hash())?
135            {
136                return Err(Error::UnexpectedGenesisHash {
137                    name: specs_invariants.name,
138                    genesis_hash: specs_invariants.genesis_hash,
139                });
140            }
141            Ok(None)
142        }
143    }
144}

159/// Specs invariants that are expected to stay unchanged for the network over
160/// time and can not be different for same genesis hash and different encryption
161/// algorithms.
162pub struct SpecsInvariants {
163    pub base58prefix: u16,
164
165    /// network with lowest order, for correct navigation when updating the
166    /// network metadata
167    pub first_network_specs_key: NetworkSpecsKey,
168    pub genesis_hash: H256,
169    pub name: String,
170}

172/// Search for network genesis hash in [`OrderedNetworkSpecs`] entries in [`SPECSTREE`]
173/// of the Vault database.
174///
175/// Genesis hash is calculated from network [`VerifierKey`] input.
176// TODO too convoluted, historically so; replace VerifierKey with genesis hash;
177// fixes needed in `add_specs`, `load_metadata` and
178// `try_get_valid_current_verifier` function above
179///
180/// If there are more than one network corresponding to the same genesis hash,
181/// outputs network specs key for the network with the lowest order.
182///
183/// If there are several entries with same genesis hash, all of them must have
184/// identical base58 prefix and network name. Network name is, and base58 prefix
185/// could be a part of the network metadata, and therefore must not depend on
186/// encryption used.
187pub fn genesis_hash_in_specs(
188    database: &sled::Db,
189    genesis_hash: H256,
190) -> Result<Option<SpecsInvariants>> {
191    let chainspecs = open_tree(database, SPECSTREE)?;
192    let mut specs_set: Vec<(NetworkSpecsKey, OrderedNetworkSpecs)> = Vec::new();
193    let mut found_permanent_specs: Option<(u16, String)> = None;
194    for (network_specs_key_vec, network_specs_encoded) in chainspecs.iter().flatten() {
195        let network_specs_key = NetworkSpecsKey::from_ivec(&network_specs_key_vec);
196        let network_specs = OrderedNetworkSpecs::from_entry_with_key_checked(
197            &network_specs_key,
198            network_specs_encoded,
199        )?;
200        if network_specs.specs.genesis_hash.as_bytes() == &genesis_hash[..] {
201            found_permanent_specs = match found_permanent_specs {
202                Some((base58prefix, name)) => {
203                    if base58prefix == network_specs.specs.base58prefix {
204                        if name == network_specs.specs.name {
205                            Some((base58prefix, name))
206                        } else {
207                            return Err(Error::DifferentNamesSameGenesisHash {
208                                name1: name,
209                                name2: network_specs.specs.name.to_string(),
210                                genesis_hash: network_specs.specs.genesis_hash,
211                            });
212                        }
213                    } else {
214                        return Err(Error::DifferentBase58Specs {
215                            genesis_hash: network_specs.specs.genesis_hash,
216                            base58_1: base58prefix,
217                            base58_2: network_specs.specs.base58prefix,
218                        });
219                    }
220                }
221                None => Some((
222                    network_specs.specs.base58prefix,
223                    network_specs.specs.name.to_string(),
224                )),
225            };
226            specs_set.push((network_specs_key, network_specs))
227        }
228    }
229    specs_set.sort_by(|a, b| a.1.order.cmp(&b.1.order));
230    match specs_set.first() {
231        Some(a) => Ok(Some(SpecsInvariants {
232            base58prefix: a.1.specs.base58prefix,
233            first_network_specs_key: a.0.to_owned(),
234            genesis_hash,
235            name: a.1.specs.name.to_string(),
236        })),
237        None => Ok(None),
238    }
239}

241/// Get general verifier [`Verifier`] from the Vault database.
242///
243/// Vault works only with an initiated database, i.e. the one with general
244/// verifier set up. Failure to find general verifier is always an error.
245pub fn get_general_verifier(database: &sled::Db) -> Result<Verifier> {
246    let settings = open_tree(database, SETTREE)?;
247    let verifier_encoded = settings
248        .get(GENERALVERIFIER)?
249        .ok_or(Error::GeneralVerifierNotFound)?;
250    Ok(<Verifier>::decode(&mut &verifier_encoded[..])?)
251}

384/// Get [`MetaValues`] set from Vault database, for networks with a given name.
385///
386/// The resulting set could be an empty one. It is used to display metadata
387/// available for the network and to find the metadata to be deleted, when the
388/// network gets deleted.
389pub(crate) fn get_meta_values_by_name(
390    database: &sled::Db,
391    network_name: &str,
392) -> Result<Vec<MetaValues>> {
393    let metadata = open_tree(database, METATREE)?;
394    let mut out: Vec<MetaValues> = Vec::new();
395    let meta_key_prefix = MetaKeyPrefix::from_name(network_name);
396    for x in metadata.scan_prefix(meta_key_prefix.prefix()).flatten() {
397        let meta_values = MetaValues::from_entry_checked(x)?;
398        if meta_values.name == network_name {
399            out.push(meta_values)
400        }
401    }
402    Ok(out)
403}

447/// Transfer metadata from the hot database into the cold one.
448///
449/// Function scans through [`METATREE`] tree of the hot database and transfers
450/// into [`METATREE`] tree of the cold database the metadata entries for the
451/// networks that already have the network specs [`OrderedNetworkSpecs`] entries in
452/// [`SPECSTREE`] of the cold database.
453///
454/// Applicable only on the active side.
455#[cfg(feature = "active")]
456pub fn transfer_metadata_to_cold(database_hot: &sled::Db, database_cold: &sled::Db) -> Result<()> {
457    let mut for_metadata = Batch::default();
458    {
459        let metadata_hot = open_tree(database_hot, METATREE)?;
460        let chainspecs_cold = open_tree(database_cold, SPECSTREE)?;
461        for x in chainspecs_cold.iter().flatten() {
462            let network_specs = NetworkSpecs::from_entry_checked(x)?;
463            for (key, value) in metadata_hot
464                .scan_prefix(MetaKeyPrefix::from_name(&network_specs.name).prefix())
465                .flatten()
466            {
467                for_metadata.insert(key, value)
468            }
469        }
470    }
471    TrDbCold::new()
472        .set_metadata(for_metadata)
473        .apply(database_cold)
474}

476/// Remove the network from the database.
477///
478/// Function inputs [`NetworkSpecsKey`] of the network, gets network genesis
479/// hash and proceeds to act on **all** networks with same genesis hash.
480///
481/// Removing network is mostly an emergency tool and is not expected to be used
482/// really often.
483///
484/// Removing a network means:
485///
486/// - Remove from [`SPECSTREE`] all [`OrderedNetworkSpecs`] that have genesis hash
487///   associated with given `NetworkSpecsKey`
488/// - Remove from [`METATREE`] all metadata entries corresponding to the network
489///   name, as found in `OrderedNetworkSpecs`
490/// - Remove from [`ADDRTREE`] all addresses in the networks being removed
491/// - Modify `Verifier` data if necessary.
492///
493/// Note that if the network supports multiple encryption algorithms, the
494/// removal of network with one of the encryptions will cause the networks
495/// with other encryptions be removed as well.
496pub fn remove_network(database: &Db, network_specs_key: &NetworkSpecsKey) -> Result<()> {
497    let mut address_batch = Batch::default();
498    let mut meta_batch = Batch::default();
499    let mut network_specs_batch = Batch::default();
500    let mut verifiers_batch = Batch::default();
501    let mut events: Vec<Event> = Vec::new();
502
503    let general_verifier = get_general_verifier(database)?;
504    let network_specs = get_network_specs(database, network_specs_key)?;
505
506    let verifier_key = VerifierKey::from_parts(network_specs.specs.genesis_hash);
507    let valid_current_verifier = get_valid_current_verifier(database, &verifier_key)?;
508
509    // modify verifier as needed
510    if let ValidCurrentVerifier::Custom { ref v } = valid_current_verifier {
511        match v {
512            Verifier { v: None } => (),
513            _ => {
514                verifiers_batch.remove(verifier_key.key());
515            }
516        }
517    }
518
519    // scan through metadata tree to mark for removal all networks with target name
520    for meta_values in get_meta_values_by_name(database, &network_specs.specs.name)?.iter() {
521        let meta_key = MetaKey::from_parts(&meta_values.name, meta_values.version);
522        meta_batch.remove(meta_key.key());
523        events.push(Event::MetadataRemoved {
524            meta_values_display: MetaValuesDisplay::get(meta_values),
525        });
526    }
527
528    {
529        let chainspecs = open_tree(database, SPECSTREE)?;
530        let identities = open_tree(database, ADDRTREE)?;
531
532        // scan through chainspecs tree to mark for removal all networks with target genesis hash
533        let mut keys_to_wipe: Vec<NetworkSpecsKey> = Vec::new();
534        for (network_specs_key_vec, entry) in chainspecs.iter().flatten() {
535            let x_network_specs_key = NetworkSpecsKey::from_ivec(&network_specs_key_vec);
536            let mut x_network_specs =
537                OrderedNetworkSpecs::from_entry_with_key_checked(&x_network_specs_key, entry)?;
538            if x_network_specs.specs.genesis_hash == network_specs.specs.genesis_hash {
539                network_specs_batch.remove(x_network_specs_key.key());
540                events.push(Event::NetworkSpecsRemoved {
541                    network_specs_display: NetworkSpecsDisplay::get(
542                        &x_network_specs,
543                        &valid_current_verifier,
544                        &general_verifier,
545                    ),
546                });
547                keys_to_wipe.push(x_network_specs_key);
548            } else if x_network_specs.order > network_specs.order {
549                x_network_specs.order -= 1;
550                network_specs_batch.insert(x_network_specs_key.key(), x_network_specs.encode());
551            }
552        }
553
554        // scan through address tree to clean up the network_key(s) from identities
555        for (address_key_vec, entry) in identities.iter().flatten() {
556            let address_key = AddressKey::from_ivec(&address_key_vec)?;
557            let (multisigner, address_details) =
558                AddressDetails::process_entry_checked((address_key_vec, entry))?;
559            for key in keys_to_wipe.iter() {
560                if address_details.network_id.as_ref() == Some(key) {
561                    let identity_history = IdentityHistory::get(
562                        &address_details.seed_name,
563                        &address_details.encryption,
564                        &multisigner_to_public(&multisigner),
565                        &address_details.path,
566                        network_specs.specs.genesis_hash,
567                    );
568                    events.push(Event::IdentityRemoved { identity_history });
569                    address_batch.remove(address_key.key())
570                } else {
571                    address_batch.insert(address_key.key(), address_details.encode())
572                }
573            }
574        }
575    }
576    TrDbCold::new()
577        .set_addresses(address_batch) // upd addresses
578        .set_history(events_to_batch(database, events)?) // add corresponding history
579        .set_metadata(meta_batch) // upd metadata
580        .set_network_specs(network_specs_batch) // upd network_specs
581        .set_verifiers(verifiers_batch) // upd network_verifiers
582        .apply(database)
583}

585/// Remove the network metadata entry from the database.
586///
587/// Function inputs [`NetworkSpecsKey`] of the network and `u32` version of the
588/// network metadata, and proceeds to remove a single metadata entry
589/// corresponding to this version.
590///
591/// Metadata in the Vault database is determined by the network name and
592/// network version, and has no information about the
593/// [`Encryption`](definitions::crypto::Encryption) algorithm supported by the
594/// network. Therefore if the network supports more than one encryption
595/// algorithm, removing metadata for one will affect all encryptions.
596pub fn remove_metadata(
597    database: &sled::Db,
598    network_specs_key: &NetworkSpecsKey,
599    network_version: u32,
600) -> Result<()> {
601    let network_specs = get_network_specs(database, network_specs_key)?;
602    let meta_key = MetaKey::from_parts(&network_specs.specs.name, network_version);
603    let mut meta_batch = Batch::default();
604    meta_batch.remove(meta_key.key());
605
606    let meta_values =
607        get_meta_values_by_name_version(database, &network_specs.specs.name, network_version)?;
608    let meta_values_display = MetaValuesDisplay::get(&meta_values);
609    let history_batch = events_to_batch(
610        database,
611        vec![Event::MetadataRemoved {
612            meta_values_display,
613        }],
614    )?;
615    TrDbCold::new()
616        .set_metadata(meta_batch) // remove metadata
617        .set_history(history_batch) // add corresponding history
618        .apply(database)
619}

621/// User-initiated removal of the types information from the Vault database.
622///
623/// Types information is not necessary to process transactions in networks with
624/// metadata having `RuntimeVersionV14`, as the types information after `V14`
625/// is a part of the metadata itself.
626///
627/// Types information is installed in Vault by default and could be removed by
628/// user manually, through this function.
629///
630/// Types information is verified by the general verifier. When the general
631/// verifier gets changed, the types information is also removed from the
632/// Vault database through so-called `GeneralHold` processing, to avoid
633/// confusion regarding what data was verified by whom. Note that this situation
634/// in **not** related to the `remove_types_info` function and is handled
635/// elsewhere.
636pub fn remove_types_info(database: &sled::Db) -> Result<()> {
637    let mut settings_batch = Batch::default();
638    settings_batch.remove(TYPES);
639    let events: Vec<Event> = vec![Event::TypesRemoved {
640        types_display: TypesDisplay::get(
641            &ContentLoadTypes::generate(&get_types(database)?),
642            &get_general_verifier(database)?,
643        ),
644    }];
645    TrDbCold::new()
646        .set_history(events_to_batch(database, events)?)
647        // add history
648        .set_settings(settings_batch)
649        // upd settings
650        .apply(database)
651}

653/// Modify existing batch for [`ADDRTREE`](constants::ADDRTREE) with incoming
654/// vector of additions.
655#[cfg(feature = "active")]
656pub(crate) fn upd_id_batch(mut batch: Batch, adds: Vec<(AddressKey, AddressDetails)>) -> Batch {
657    for (address_key, address_details) in adds.iter() {
658        batch.insert(address_key.key(), address_details.encode());
659    }
660    batch
661}

663/// Verify checksum in Vault database.
664///
665/// Used in retrieving temporary stored data from
666/// [`TRANSACTION`](constants::TRANSACTION) tree of the database.
667// TODO Goes obsolete if the temporary storage goes.
668pub(crate) fn verify_checksum(database: &Db, checksum: u32) -> Result<()> {
669    let real_checksum = database.checksum()?;
670    if checksum != real_checksum {
671        return Err(Error::ChecksumMismatch);
672    }
673    Ok(())
674}

676/// Get the danger status from the Vault database.
677///
678/// Currently, the only flag contributing to the danger status is whether the
679/// device was online. This may change eventually.
680pub fn get_danger_status(database: &sled::Db) -> Result<bool> {
681    let settings = open_tree(database, SETTREE)?;
682    let a = settings.get(DANGER)?.ok_or(Error::DangerStatusNotFound)?;
683    Ok(DangerRecord::from_ivec(&a).device_was_online()?)
684}
