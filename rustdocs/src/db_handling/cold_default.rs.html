1//! Making and restoring **cold** database with default content
2//!
3//! Cold database is the database that is operated from inside the Vault.
4//!
5//! Cold database contains following trees:
6//!
7//! - `ADDRTREE` with public addresses data
8//! - `HISTORY` with Vault history log
9//! - `METATREE` with network metadata
10//! - `SETTREE` with settings: types information, Vault dangerous exposures
11//!   record and Vault database general verifier
12//! - `SPECSTREE` with network specs
13//! - `TRANSACTION` for temporary storage of the transaction data
14//! - `VERIFIERS` with network verifiers data
15//!
16//! For release, the cold database is generated on the hot side and then copied
17//! verbatim into Vault files during the build.
18//!
19//! Before the database could be used, it must be initiated:
20//!
21//! - History log old entries (if any are present) are removed and a new entry
22//!   `Event::DatabaseInitiated` is added
23//! - General verifier is set and this event is recorded in the history log. By
24//!   default, Vault sets up Parity-associated key as a general verifier. This
25//!   could be later on changed by the user.
26//!
27//! Vault then reads and updates the database as it operates.
28//!
29//! There are two ways to reset the database from the Vault user interface.
30//! Either would remove all the keys and restore the database to the release
31//! state. The difference would be only in the general verifier setting:
32//!
33//! - `Wipe all data` would set the general verifier to default `Some(_)`,
34//!   with Parity-associated key inside
35//! - `Remove general certificate` would set the general verifier to `None`.
36//!   User would then be able to set up own general verifier, preferably
37//!   immediately afterwards, by loading to Vault any verified data.
38//!   Setting up a new general verifier would remove all data associated with the
39//!   general verifier from the Vault database to avoid confusion as to who
40//!   verified what information.

74/// Cold database generation purpose, determining the metadata to be loaded.
75///
76/// Default metadata is loaded into the cold database for default networks:
77/// Polkadot, Kusama, Westend. `Purpose` determines the metadata source folder
78/// and the versions to be loaded.
79#[cfg(feature = "active")]
80enum Purpose {
81    /// Two (or fewer) latest released versions of the metadata for each of the
82    /// default networks
83    Release,
84
85    /// Old metadata set, used mostly in `transaction_parsing` tests
86    Test,
87
88    /// Not so old metadata set, used for `navigator` tests
89    TestNavigator,
90}
91
92/// Make [`Batch`] with default networks metadata, for [`METATREE`] tree, in
93/// purged database.
94///
95/// Adds default metadata entries, according to [`Purpose`].
96#[cfg(feature = "active")]
97fn default_cold_metadata(purpose: Purpose) -> Result<Batch> {
98    let mut batch = Batch::default();
99    let metadata_set = match purpose {
100        Purpose::Release => release_metadata()?,
101
102        Purpose::Test => test_metadata()?,
103
104        Purpose::TestNavigator => nav_test_metadata()?,
105    };
106    for x in metadata_set.iter() {
107        let meta_key = MetaKey::from_parts(&x.name, x.version);
108        batch.insert(meta_key.key(), &x.meta[..]);
109    }
110    Ok(batch)
111}

113/// Make [`Batch`] with default networks
114/// [`OrderedNetworkSpecs`](definitions::network_specs::OrderedNetworkSpecs), for
115/// [`SPECSTREE`] tree, in purged database.
116///
117/// Adds default network specs entries.
118#[cfg(feature = "active")]
119fn default_cold_network_specs() -> Batch {
120    let mut batch = Batch::default();
121    for x in default_chainspecs().iter() {
122        let network_specs_key =
123            NetworkSpecsKey::from_parts(&x.specs.genesis_hash, &x.specs.encryption);
124        batch.insert(network_specs_key.key(), x.encode());
125    }
126    batch
127}

129/// Make [`Batch`] with default settings, for [`SETTREE`] tree, in purged
130/// database.
131///
132/// Adds default entries: types information
133/// [`ContentLoadTypes`](definitions::qr_transfers::ContentLoadTypes) and danger
134/// record [`DangerRecord`].
135///
136/// Note that the general verifier is **not** set up here.
137///
138/// General verifier is set up separately, during the database initiation
139/// [`init_db`]. Without general verifier (i.e. a value in the [`SETTREE`] tree
140/// under the key [`GENERALVERIFIER`]) the database is not usable by the Vault.
141#[cfg(feature = "active")]
142fn default_cold_settings_init_later() -> Result<Batch> {
143    let mut batch = Batch::default();
144    let types_prep = default_types_content()?;
145    batch.insert(TYPES, types_prep.store());
146    batch.insert(DANGER, DangerRecord::safe().store());
147    batch.insert(SCHEMA_VERSION, SchemaVersion::store_current());
148    Ok(batch)
149}

151/// Make [`Batch`] with default networks verifiers, for [`VERIFIERS`] tree, in
152/// purged database.
153///
154/// Adds default
155/// [`CurrentVerifier`](definitions::network_specs::CurrentVerifier) entries.
156#[cfg(feature = "active")]
157fn default_cold_verifiers() -> Batch {
158    let mut batch = Batch::default();
159    for (verifier_key, current_verifier) in default_verifiers().iter() {
160        batch.insert(verifier_key.key(), current_verifier.encode());
161    }
162    batch
163}

165/// Make or restore the cold database with default content, according to
166/// [`Purpose`].
167///
168/// Function wipes everything in the database directory and loads into database
169/// defaults for:
170///
171/// - metadata
172/// - network specs
173/// - types information and danger status
174/// - network verifiers
175///
176/// Note that the resulting database is not initiated and is not ready to be
177/// used by the Vault.
178#[cfg(any(feature = "active", feature = "test"))]
179fn cold_database_no_init(database: &sled::Db, purpose: Purpose) -> Result<()> {
180    use constants::{METATREE, SETTREE, SPECSTREE, VERIFIERS};
181
182    database.drop_tree(SPECSTREE)?;
183    database.drop_tree(VERIFIERS)?;
184    database.drop_tree(METATREE)?;
185    database.drop_tree(SETTREE)?;
186    database.clear()?;
187
188    TrDbCold::new()
189        .set_metadata(default_cold_metadata(purpose)?) // set default metadata
190        .set_network_specs(default_cold_network_specs()) // set default network specs
191        .set_settings(default_cold_settings_init_later()?) // set default types and danger status, no general verifier yet
192        .set_verifiers(default_cold_verifiers()) // set default verifiers
193        .apply(database)?;
194
195    Ok(())
196}

198/// Initiate cold database and set up the database general verifier to given
199/// [`Verifier`].
200///
201/// Function simultaneously sets up the general verifier and marks the new start
202/// of the history log.
203///
204/// Could be used both from the Vault side (with `Wipe all data` and with
205/// `Remove general certificate` procedures) and from the active side, when
206/// when preparing the test databases.
207///
208/// After applying this function the database becomes ready to be used by the
209/// Vault.
210pub fn init_db(database: &sled::Db, general_verifier: Verifier) -> Result<()> {
211    let mut settings_batch = Batch::default();
212    settings_batch.insert(GENERALVERIFIER, general_verifier.encode());
213
214    let clear_history_batch = make_batch_clear_tree(database, HISTORY)?;
215    let events = vec![
216        Event::DatabaseInitiated,
217        Event::GeneralVerifierSet {
218            verifier: general_verifier,
219        },
220    ];
221    let start_zero = true;
222    let history_batch = events_in_batch(database, start_zero, clear_history_batch, events)?;
223
224    TrDbCold::new()
225        .set_history(history_batch) // set *start* history
226        .set_settings(settings_batch) // set general_verifier
227        .apply(database)?;
228
229    Ok(())
230}

232/// Initiate Vault database with default general verifier (Parity-associated
233/// key).
234///
235/// Function is applied during the initial start of the Vault and during
236/// `Wipe all data` procedure.
237pub fn signer_init_with_cert(database: &sled::Db) -> Result<()> {
238    init_db(database, default_general_verifier())

241/// Initiate Vault database with general verifier set up to `Verifier(None)`.

248/// Generate initiated test cold database with no network-associated data.
249///
250/// Function wipes everything in the database directory and loads into database
251/// defaults for types information and danger status. Then the database is
252/// initiated with given general verifier.
253pub fn populate_cold_no_networks(database: &sled::Db, general_verifier: Verifier) -> Result<()> {
254    database.clear()?;
255    TrDbCold::new()
256        .set_settings(default_cold_settings_init_later()?) // set general verifier and load default types
257        .apply(database)?;
258    init_db(database, general_verifier)
259}

312/// Generate database with networks specs both for substrate and ethereum
313pub fn populate_all_network_specs(database: &sled::Db) -> Result<()> {
314    use constants::{METATREE, SETTREE, SPECSTREE, VERIFIERS};
315
316    database.drop_tree(SPECSTREE)?;
317    database.drop_tree(VERIFIERS)?;
318    database.drop_tree(METATREE)?;
319    database.drop_tree(SETTREE)?;
320    database.clear()?;
321
322    let mut batch = Batch::default();
323    for x in defaults::substrate_chainspecs_with_ethereum().iter() {
324        let network_specs_key =
325            NetworkSpecsKey::from_parts(&x.specs.genesis_hash, &x.specs.encryption);
326        batch.insert(network_specs_key.key(), x.encode());
327    }
328
329    TrDbCold::new()
330        .set_network_specs(batch) // set default network specs including Mythos
331        .set_settings(default_cold_settings_init_later()?) // set general verifier and load default types
332        .set_verifiers(default_cold_verifiers()) // set default verifiers
333        .apply(database)?;
334    Ok(())
