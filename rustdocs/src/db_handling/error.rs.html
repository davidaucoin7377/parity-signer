13#[derive(Debug, thiserror::Error)]
14pub enum Error {
15    #[error("No networks available")]
16    NoNetworksAvailable,
17
18    #[error(transparent)]
19    DbTransactionError(#[from] sled::transaction::TransactionError),
20
21    #[error(transparent)]
22    Defaults(#[from] defaults::Error),
23
24    #[error(transparent)]
25    Other(#[from] anyhow::Error),
26
27    #[error(transparent)]
28    Codec(#[from] parity_scale_codec::Error),
29
30    #[error(transparent)]
31    TimeFormat(#[from] time::error::Format),
32
33    #[error(transparent)]
34    DefinitionsError(#[from] definitions::error::Error),
35
36    #[error(transparent)]
37    Bip39MnemonicType(#[from] bip39::ErrorKind),
38
39    #[error("Database error. Internal error. {0}")]
40    DbError(#[from] sled::Error),
41
42    #[error("Database schema version mismatch. Expected v{expected}, found v{found}.")]
43    DbSchemaMismatch { expected: u32, found: u32 },
44
45    /// Temporary database entry in `TRANSACTION` tree of the Vault database
46    /// under the key `STUB`, used to store the update data awaiting for the
47    /// user approval.
48    ///
49    /// Missing `Stub` when it is expected always indicates the database
50    /// corruption.
51    #[error(
52        "Unable to decode temporary entry with information \
53        needed for accepting approved information."
54    )]
55    Stub,
56
57    /// Network [`CurrentVerifier`](definitions::network_specs::CurrentVerifier) is
58    /// `ValidCurrentVerifier::Custom(_)`, but the custom verifier value
59    /// coincides with the general verifier.
60    ///
61    /// Associated data is [`VerifierKey`] corresponding to faulty entry.
62    #[error(
63        "Network with genesis hash {} verifier is set as a custom one. \
64        This custom verifier coinsides the database general verifier \
65        and not None. This should not have happened and likely \
66        indicates database corruption.",
67        hex::encode(.0.key()),
68    )]
69    CustomVerifierIsGeneral(VerifierKey),
70
71    /// Network has no entry for
72    /// [`CurrentVerifier`](definitions::network_specs::CurrentVerifier) under
73    /// `verifier_key` in `VERIFIERS` tree of the database, however, the
74    /// corresponding genesis hash is encountered in a
75    /// [`OrderedNetworkSpecs`](definitions::network_specs::OrderedNetworkSpecs) entry under
76    /// `network_specs_key` in `SPECSTREE` tree of the database.
77    /// No network specs record can get into database without the verifier
78    /// entry, and the verifier could not be removed while network specs still
79    /// remain, so this indicates the database got corrupted.
80    #[error(
81        "Network {name} with genesis hash {} has some network specs entries, \
82        while there is no verifier entry.",
83        hex::encode(genesis_hash)
84    )]
85    UnexpectedGenesisHash { name: String, genesis_hash: H256 },
86
87    /// History log [`Entry`](definitions::history::Entry) stored in `HISTORY` tree
88    /// of the Vault database under the key [`Order`]
89    /// could not be decoded.
90    ///
91    /// Associated data is the corresponding [`Order`].
92    ///
93    /// [`Order`]: definitions::keyring::Order
94    #[error("Unable to decode history entry for order {0}")]
95    HistoryEntryNotFound(u32),
96
97    /// Database has two seed addresses (i.e. with empty derivation path and no
98    /// password) for same seed name and [`Encryption`]
99    ///
100    /// This indicates the database corruption, since the encryption method,
101    /// seed name and derivation path strictly determine the public key.
102    #[error(
103        "More than one seed key (i.e. with empty path and without password)
104        found for seed name {seed_name} and encryption {}.",
105        encryption.show(),
106    )]
107    TwoRootKeys {
108        /// seed name
109        seed_name: String,
110
111        /// encryption algorithm for which two seed keys were found
112        encryption: Encryption,
113    },
114
115    /// To generate QR code with public address information export, Vault
116    /// receives both seed name and
117    /// [`MultiSigner`](https://docs.rs/sp-runtime/6.0.0/sp_runtime/enum.MultiSigner.html)
118    /// from the navigation state `Navstate`.
119    /// `MultiSigner` gets transformed into [`AddressKey`] and corresponds to
120    /// [`AddressDetails`](definitions::users::AddressDetails) that are exported.
121    /// `AddressDetails` also contain `seed_name` field, that must coincide
122    /// with the one received directly from the navigator.
123    /// This error appears if the seed names are different.
124    #[error(
125        "Expected seed name {expected_seed_name} for address key {}. \
126        Address details in database have {real_seed_name} name.",
127        hex::encode(address_key.key()),
128    )]
129    SeedNameNotMatching {
130        /// address key for which the export is done
131        address_key: AddressKey,
132
133        /// seed name, from the navigator
134        expected_seed_name: String,
135
136        /// seed name, from the `AddressDetails`
137        real_seed_name: String,
138    },
139
140    /// QR handling error.
141    #[error("QR error {0}.")]
142    Qr(String),
143
144    /// [`NetworkSpecsKey`] of a network in `network_id` field of the
145    /// [`AddressDetails`](definitions::users::AddressDetails) corresponding to
146    /// [`AddressKey`].
147    ///
148    /// This happens when the derivation is created in some other network(s), but
149    /// not in the given network. This way the `AddressKey` is in the database,
150    /// but the address in the network is not.
151    #[error(
152        "Could not find network specs key {} in address details with key {}.",
153        hex::encode(.network_specs_key.key()),
154        hex::encode(.address_key.key())
155    )]
156    NetworkSpecsKeyForAddressNotFound {
157        network_specs_key: NetworkSpecsKey,
158        address_key: AddressKey,
159    },
160
161    /// Got empty seed phrase.
162    #[error("Seed phrase empty.")]
163    EmptySeed,
164
165    /// Got empty seed name.
166    #[error("Seed name is empty.")]
167    EmptySeedName,
168
169    /// Error in [`SecretString`](https://docs.rs/sp-core/6.0.0/sp_core/crypto/type.SecretString.html).
170    ///
171    /// `SecretString` consists of combined seed phrase and derivation.
172    ///
173    /// Associated error content is
174    /// [`SecretStringError`](https://docs.rs/sp-core/6.0.0/sp_core/crypto/enum.SecretStringError.html).
175    #[error("Secret string error: {}", format!("{:?}", .0))]
176    SecretStringError(sp_core::crypto::SecretStringError),
177
178    /// Same public key was produced for a different seed phrase and/or
179    /// derivation path, during database transaction preparation (not yet in
180    /// the database).
181    #[error(
182        "Tried to create colliding addresses within {}.\
183        Address for seed name {seed_name_new} and path {cropped_path_new} \
184        has same public key as address for seed name {seed_name_existing} \
185        and path {cropped_path_existing}.",
186        if *.in_this_network { "same network" } else { "different networks" }
187    )]
188    KeyCollisionBatch {
189        seed_name_existing: String,
190        seed_name_new: String,
191        cropped_path_existing: String,
192        cropped_path_new: String,
193        in_this_network: bool,
194    },
195
196    /// Same public key was produced for a different seed phrase and/or
197    /// derivation path, as already existing in the database.
198    ///
199    /// Address is generated within a network using seed phrase and derivation
200    /// path.
201    ///
202    /// Address is defined by public key and [`NetworkSpecsKey`]. Public key
203    /// is created from seed phrase and derivation with encryption algorithm
204    /// supported by the network.
205    ///
206    /// If two networks are using the same encryption algorithm, generating
207    /// public key in both with same seed phrase and derivation path would
208    /// result in two identical public keys. This is normal and expected
209    /// behavior, and this is the reason why [`AddressDetails`] contain a set
210    /// of allowed networks in `network_id` field.
211    ///
212    /// It is, however, possible, that when generating public key for
213    /// **different** seed names and/or **different** derivation
214    /// paths, resulting public keys accidentally coincide.
215    ///
216    /// This is called here `KeyCollision`, and results in error.
217    #[error("Address key collision for seed name {seed_name}")]
218    KeyCollision { seed_name: String },
219
220    /// Derivation that user tried to create already exists.
221    ///
222    /// Associated error content is:
223    /// - [`MultiSigner`](https://docs.rs/sp-runtime/6.0.0/sp_runtime/enum.MultiSigner.html)
224    ///   of the already existing address
225    /// - [`AddressDetails`] for already existing address
226    /// - [`NetworkSpecsKey`] of the associated network
227    #[error(
228        "Seed {} already has derivation {}{} \
229        for network specs key {}, public key {}.",
230        .address_details.seed_name,
231        .address_details.path,
232        if .address_details.has_pwd { "///<password>" } else { "" },
233        hex::encode(network_specs_key.key()),
234        hex::encode(multisigner_to_public(.multisigner)),
235    )]
236    DerivationExists {
237        multisigner: MultiSigner,
238        address_details: Box<AddressDetails>,
239        network_specs_key: NetworkSpecsKey,
240    },
241    /// Received `derivations` update payload contains an invalid derivation.
242    ///
243    /// Associated data is the derivation that could not be used as a `String`.
244    #[error("Derivation {0} has invalid format.")]
245    InvalidDerivation(String),
246
247    /// User was creating the derivation with password, and thus moved into
248    /// `PasswordConfirm` modal, however, the password was not found when
249    /// cutting password from the path.
250    #[error("Derivation had password, then lost it.")]
251    LostPwd,
252
253    /// Temporary database entry in `TRANSACTION` tree of the Vault database
254    /// under the key `DRV`, used to store the derivation import data.
255    ///
256    /// Missing `Derivations` when it is expected always indicates the database
257    /// corruption.
258    #[error("Derivations not found.")]
259    DerivationsNotFound,
260
261    /// Temporary database entry in `TRANSACTION` tree of the Vault database
262    /// under the key `SIGN`, used to store the signable transaction data
263    /// awaiting for the user approval.
264    ///
265    /// Missing `Sign` when it is expected always indicates the database
266    /// corruption.
267    #[error(
268        "Could not find database temporary entry with information \
269        needed for signing approved transaction."
270    )]
271    Sign,
272
273    #[error("Unable to decode current verifier entry for key {}.", hex::encode(.0.key()))]
274    NoValidCurrentVerifier(VerifierKey),
275
276    /// While searching for all networks with same genesis hash, found that
277    /// there are networks with same genesis hash, but different names.
278    #[error(
279        "Different network names ({name1}, {name2}) in database for same genesis hash {}.",
280        hex::encode(genesis_hash)
281    )]
282    DifferentNamesSameGenesisHash {
283        name1: String,
284        name2: String,
285        genesis_hash: H256,
286    },
287
288    /// Network specs entries have same genesis hash, but different base58 prefix
289    #[error(
290        "Prefix mismatch for same genesis hash {}: {base58_1}, {base58_2}",
291        hex::encode(genesis_hash.as_ref()),
292    )]
293    DifferentBase58Specs {
294        base58_1: u16,
295        base58_2: u16,
296        genesis_hash: H256,
297    },
298
299    /// General verifier [`Verifier`](definitions::network_specs::Verifier) information
300    /// stored in `SETTREE` tree of the database under key `GENERALVERIFIER`.
301    ///
302    /// Missing general verifier always indicates the database corruption.
303    #[error("Could not find general verifier.")]
304    GeneralVerifierNotFound,
305
306    /// [`DangerRecord`](definitions::danger::DangerRecord) information
307    /// stored in `SETTREE` tree of the database under key `DANGER`.
308    ///
309    /// Missing `DangerRecord` always indicates the database corruption.
310    #[error("Types not found.")]
311    TypesNotFound,
312
313    /// [`OrderedNetworkSpecs`](definitions::network_specs::OrderedNetworkSpecs) for a network
314    /// in `SPECSTREE` tree of the Vault database, searched by
315    /// [`NetworkSpecsKey`].
316    ///
317    /// Associated data is the `NetworkSpecsKey` used for the search.
318    #[error("Could not find network specs for network specs key {}.", hex::encode(.0.key()))]
319    NetworkSpecsNotFound(NetworkSpecsKey),
320
321    /// [`AddressDetails`](definitions::users::AddressDetails) for [`AddressKey`] in
322    /// `ADDRTREE` tree of the Vault database.
323    ///
324    /// Associated data is the `AddressKey` used for search.
325    #[error(
326        "Could not find address details for address key {}.", hex::encode(.0.key())
327    )]
328    AddressNotFound(AddressKey),
329
330    /// Network metadata in `METATREE` tree of the Vault database, for network
331    /// name and version combination.
332    #[error("Meta values not found for {name} version {version}")]
333    MetaValuesNotFound {
334        /// network name
335        name: String,
336
337        /// network version
338        version: u32,
339    },
340
341    /// Database checksum does not match the expected value.
342    #[error("Checksum mismatch.")]
343    ChecksumMismatch,
344
345    /// [`DangerRecord`](definitions::danger::DangerRecord) information
346    /// stored in `SETTREE` tree of the database under key `DANGER`.
347    ///
348    /// Missing `DangerRecord` always indicates the database corruption.
349    #[error("Danger status not found.")]
350    DangerStatusNotFound,
351
352    #[error("There are no seeds. Please create a seed first.")]
353    NoKnownSeeds,
354
355    /// Found `secret_exposed` flag mismatch in the database: address is not
356    /// marked as potentially exposed when it must have been.
357    #[error(
358    "Address details entry with public key {} (seed {}, path {}{}) is not marked as potentially exposed, \
359        when it should be.",
360    hex::encode(multisigner_to_public(.multisigner)),
361    .address_details.seed_name,
362    .address_details.path,
363    if .address_details.has_pwd { "///<password>" } else { "" },
364    )]
365    SecretExposedMismatch {
366        multisigner: MultiSigner,
367        address_details: AddressDetails,
368    },
369
370    /// User has entered a wrong password for a passworded address.
371    #[error("Wrong password.")]
372    WrongPassword,
373
374    #[error("Missing information about whether the path {0} is passworded.")]
375    MissingPasswordInfo(String),
376
377    #[error("Key pair with public key {} can't be expressed as a direct derivation from a seed",
378    hex::encode(multisigner_to_public(.multisigner)),
379    )]
380    NoSeedForKeyPair { multisigner: MultiSigner },
381
382    #[error("No seed is found for {} root key",
383    hex::encode(multisigner_to_public(.multisigner)),
384    )]
385    NoSeedFound { multisigner: MultiSigner },
386
387    #[error("No root derivation for seed {0}")]
388    NoRootKeyForSeed(String),
389
390    #[error("Data packing error: {0}")]
391    DataPacking(String),
392}
