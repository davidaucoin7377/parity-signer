1//! Creating, collecting, and removing addresses in cold database
2//!
3//! In Vault and Vault-related ecosystem the address is a key pair within a
4//! certain network. Vault Rust database stores only **non-secret** data
5//! associated with each address generated by user.
6//!
7//! Required data to generate address:
8//!
9//! - seed phrase (secret words set, must be always zeroized after use and is
10//!   never stored in Rust database)
11//! - derivation path (`/soft//hard///password`); password part, i.e. whatever
12//!   follows `///` is not stored in Rust database and must be zeroized after use
13//! - [`NetworkSpecsKey`](definitions::keyring::NetworkSpecsKey)
14//!
15//! The combination of seed phrase and derivation path allows to generate key
16//! pair using the encryption algorithm supported by the network, for which the
17//! address is generated.
18//!
19//! Vault stores only the public key, it is used together with encryption
20//! algorithm [`Encryption`](definitions::crypto::Encryption) to generate
21//! [`AddressKey`](definitions::keyring::AddressKey).
22//!
23//! Same `AddressKey` could correspond to more than one network if the addresses
24//! were produced for same seed phrase and derivation path, and networks use
25//! same encryption algorithm. The networks specs keys of associated networks
26//! are stored in `network_id` field of `AddressDetails`.
27//!
28//! Vault interface shows those addresses as separate entities.
29//!
30//! Non-secret data associated with `AddressKey` is stored in
31//! [`ADDRTREE`](constants::ADDRTREE) tree of the cold database as SCALE-encoded
32//! [`AddressDetails`](definitions::users::AddressDetails).

217/// Export info about keys and their addresses known to Vault
218pub fn export_key_set_addrs(
219    database: &sled::Db,
220    seed_name: &str,
221    exported_set: ExportedSet,
222) -> Result<ExportAddrs> {
223    let keys = get_addresses_by_seed_name(database, seed_name)?;
224    let root_multisigner = keys
225        .iter()
226        .find(|(_, a)| a.is_root())
227        .map(|(m, _)| m.to_owned())
228        .ok_or(Error::NoRootKeyForSeed(seed_name.to_owned()))?;
229
230    let mut derived_keys = vec![];
231
232    for key in keys {
233        if key.1.is_root() {
234            continue;
235        }
236
237        let mut selected = false;
238
239        match &exported_set {
240            ExportedSet::All => selected = true,
241            ExportedSet::Selected {
242                s: selected_derivations,
243            } => {
244                for selected_derivation in selected_derivations {
245                    if let Some(id) = &key.1.network_id {
246                        if selected_derivation.derivation == key.1.path
247                            && selected_derivation.network_specs_key == hex::encode(id.key())
248                        {
249                            selected = true;
250                            break;
251                        }
252                    }
253                }
254            }
255        }
256
257        if !selected {
258            continue;
259        }
260
261        if let Some(id) = &key.1.network_id {
262            let specs = get_network_specs(database, id)?;
263            let address_or_pub_key = print_multisigner_as_base58_or_eth_public_key(
264                &key.0,
265                Some(specs.specs.base58prefix),
266            );
267            derived_keys.push(AddrInfo {
268                address_or_pubkey: address_or_pub_key.clone(),
269                derivation_path: if key.1.path.is_empty() {
270                    None
271                } else {
272                    Some(key.1.path.to_owned())
273                },
274                encryption: key.1.encryption,
275                genesis_hash: specs.specs.genesis_hash,
276            });
277        }
278    }
279
280    Ok(ExportAddrs::V2(ExportAddrsV2::new(SeedInfo {
281        name: seed_name.to_owned(),
282        multisigner: root_multisigner,
283        derived_keys,
284    })))
285}

287pub fn import_all_addrs(
288    database: &sled::Db,
289    seed_derived_keys: Vec<SeedKeysPreview>,
290) -> Result<()> {
291    // Address preparation set, to be modified and used as `create_address`
292    // input.
293    let mut adds: Vec<(AddressKey, AddressDetails)> = vec![];
294
295    // Associated `Event` set
296    let mut events: Vec<Event> = vec![];
297
298    for addr in &seed_derived_keys {
299        for derived_key in &addr.derived_keys {
300            let path = derived_key.derivation_path.clone().unwrap_or_default();
301            let network_specs_key =
302                NetworkSpecsKey::from_parts(&derived_key.genesis_hash, &derived_key.encryption);
303            let network_specs = get_network_specs(database, &network_specs_key)?;
304            match create_derivation_address(
305                database,
306                &adds, // a single address is created, no data to check against here
307                &path,
308                &network_specs.specs,
309                &addr.name,
310                &derived_key.address,
311                derived_key
312                    .has_pwd
313                    .ok_or_else(|| Error::MissingPasswordInfo(path.to_owned()))?,
314            ) {
315                // success, updating address preparation set and `Event` set
316                Ok(prep_data) => {
317                    adds = prep_data.address_prep;
318                    events.extend_from_slice(&prep_data.history_prep);
319                }
320                // exactly same address already exists, ignoring it
321                Err(Error::DerivationExists { .. }) => (),
322
323                // some other error, processed as a real error
324                Err(e) => return Err(e),
325            }
326        }
327    }
328    TrDbCold::new()
329        .set_addresses(upd_id_batch(Batch::default(), adds)) // modify addresses data
330        .set_history(events_to_batch(database, events)?) // add corresponding history
331        .apply(database)
332}

334/// Get public key from seed phrase and derivation path
335fn full_address_to_multisigner(
336    mut full_address: String,
337    encryption: Encryption,
338) -> Result<MultiSigner> {
339    let multisigner_result = match encryption {
340        Encryption::Ed25519 => match ed25519::Pair::from_string(&full_address, None) {
341            Ok(a) => Ok(MultiSigner::Ed25519(a.public())),
342            Err(e) => Err(Error::SecretStringError(e)),
343        },
344        Encryption::Sr25519 => match sr25519::Pair::from_string(&full_address, None) {
345            Ok(a) => Ok(MultiSigner::Sr25519(a.public())),
346            Err(e) => Err(Error::SecretStringError(e)),
347        },
348        Encryption::Ecdsa | Encryption::Ethereum => {
349            match ecdsa::Pair::from_string(&full_address, None) {
350                Ok(a) => Ok(MultiSigner::Ecdsa(a.public())),
351                Err(e) => Err(Error::SecretStringError(e)),
352            }
353        }
354    };
355    full_address.zeroize();
356    multisigner_result
357}

359pub fn validate_key_password(
360    database: &sled::Db,
361    address_key: &AddressKey,
362    seed_phrase: &str,
363    password: &str,
364) -> Result<bool> {
365    let address = get_address_details(database, address_key)?;
366    let full_address = format!("{}{}///{}", seed_phrase, address.path, password);
367    let expected = full_address_to_multisigner(full_address, address.encryption)?;
368    Ok(&expected == address_key.multi_signer())
369}

371/// Return address details that corresponds to a public key. The root public keys are prioritized.
372/// Then network hashes passed as last param allows to choose one address
373/// in case multiple addresses available in different networks
374pub fn find_address_details_for_multisigner(
375    database: &sled::Db,
376    multisigner: &MultiSigner,
377    prioritizing_network_hashes: Vec<H256>,
378) -> Result<Option<AddressDetails>> {
379    let matching_addresses: Vec<AddressDetails> = get_all_addresses(database)?
380        .into_iter()
381        .filter(|(m, _)| m == multisigner)
382        .map(|(_, details)| details)
383        .collect();
384
385    let maybe_root_address = matching_addresses
386        .iter()
387        .find(|details| details.network_id.is_none());
388
389    if let Some(root_address) = maybe_root_address {
390        return Ok(Some(root_address.clone()));
391    }
392
393    if prioritizing_network_hashes.is_empty() {
394        return Ok(matching_addresses.first().cloned());
395    }
396
397    let mut indexed_addresses: HashMap<H256, &AddressDetails> = HashMap::new();
398
399    for addr in matching_addresses.iter() {
400        if let Some(Ok((hash, _))) = addr
401            .network_id
402            .as_ref()
403            .map(|id| id.genesis_hash_encryption())
404        {
405            _ = indexed_addresses.insert(hash, addr)
406        }
407    }
408
409    for genesis_hash in prioritizing_network_hashes {
410        if let Some(addr) = indexed_addresses.get(&genesis_hash) {
411            return Ok(Some((*addr).clone()));
412        }
413    }
414
415    Ok(matching_addresses.first().cloned())
416}

429pub fn process_dynamic_derivations_v1(
430    database: &sled::Db,
431    seeds: HashMap<String, String>,
432    request: DynamicDerivationsAddressRequestV1,
433) -> Result<DDPreview> {
434    let mut is_some_already_imported = false;
435    let mut is_some_network_missing = false;
436
437    let mut new_addrs: Vec<(AddressKey, AddressDetails)> = vec![];
438    let seed_request = &request.addr;
439    let seed_name = find_seed_name_for_multisigner(database, &seed_request.multisigner)?
440        .ok_or_else(|| Error::NoSeedForKeyPair {
441            multisigner: seed_request.multisigner.clone(),
442        })?;
443    let seed_phrase = seeds
444        .get(&seed_name)
445        .ok_or_else(|| Error::NoSeedForKeyPair {
446            multisigner: seed_request.multisigner.clone(),
447        })?;
448    let mut derivations = vec![];
449    for derivation_request in &seed_request.dynamic_derivations {
450        let network_specs_key = NetworkSpecsKey::from_parts(
451            &derivation_request.genesis_hash,
452            &derivation_request.encryption,
453        );
454        let network_specs = match get_network_specs(database, &network_specs_key) {
455            Ok(s) => s,
456            Err(_) => {
457                is_some_network_missing = true;
458                continue;
459            }
460        };
461        let multisigner = match create_address(
462            database,
463            &new_addrs,
464            &derivation_request.derivation_path,
465            Some(&network_specs.specs),
466            &seed_name,
467            seed_phrase,
468        ) {
469            // success
470            Ok(prep_data) => {
471                new_addrs = prep_data.address_prep;
472                new_addrs.last()
473                    .expect("new_addrs is never empty")
474                    .0
475                    .multi_signer()}
476            ,
477            // exactly same address already exists, ignoring it
478            Err(Error::DerivationExists { .. }) => {
479                is_some_already_imported = true;
480                continue;
481            },
482
483            // some other error, processed as a real error
484            Err(e) => return Err(e),
485        };
486        let encryption = derivation_request.encryption;
487        derivations.push(DDDetail {
488            base58: print_multisigner_as_base58_or_eth_address(
489                multisigner,
490                Some(network_specs.specs.base58prefix),
491                encryption,
492            ),
493            path: derivation_request.derivation_path.clone(),
494            network_logo: network_specs.specs.logo,
495            network_specs_key: hex::encode(network_specs_key.key()),
496            identicon: make_identicon_from_multisigner(multisigner, encryption.identicon_style()),
497        })
498    }
499
500    let key_set = DDKeySet {
501        seed_name: seed_name.to_string(),
502        derivations,
503    };
504    let qr = dynamic_derivations_response_qr(&request, seed_phrase)?;
505    Ok(DDPreview {
506        qr,
507        key_set,
508        is_some_already_imported,
509        is_some_network_missing,
510    })
511}

523/// Prepare `DynamicDerivationsAddressResponse` for the given request
524pub fn dynamic_derivations_response(
525    request: &DynamicDerivationsAddressRequestV1,
526    seed_phrase: &str,
527) -> Result<DynamicDerivationsAddressResponse> {
528    let seed_request = &request.addr;
529    let mut derivations = vec![];
530    for derivation_request in &seed_request.dynamic_derivations {
531        let path = derivation_request.derivation_path.as_str();
532        // create fixed-length string to avoid reallocations
533        let mut full_address = String::with_capacity(seed_phrase.len() + path.len());
534        full_address.push_str(seed_phrase);
535        full_address.push_str(path);
536
537        derivations.push(DynamicDerivationResponseInfo {
538            derivation_path: derivation_request.derivation_path.clone(),
539            encryption: derivation_request.encryption,
540            public_key: full_address_to_multisigner(full_address, derivation_request.encryption)?,
541        });
542    }
543    let addr = DynamicDerivationsResponseInfo {
544        multisigner: seed_request.multisigner.clone(),
545        dynamic_derivations: derivations,
546    };
547
548    Ok(DynamicDerivationsAddressResponse::V1(
549        DynamicDerivationsAddressResponseV1 { addr },
550    ))
551}

553/// Helper function to get public key from seed phrase and derivation path
554pub fn derive_single_key(
555    database: &sled::Db,
556    seeds: &HashMap<String, String>,
557    derivation_path: &str,
558    root_key_id: &[u8; 32],
559    network_key: NetworkSpecsKey,
560) -> Result<(MultiSigner, AddressDetails)> {
561    let root_multisigner = MultiSigner::Sr25519(sr25519::Public(*root_key_id));
562
563    let seed_name =
564        find_seed_name_for_multisigner(database, &root_multisigner)?.ok_or_else(|| {
565            Error::NoSeedFound {
566                multisigner: root_multisigner.clone(),
567            }
568        })?;
569    let seed_phrase = seeds.get(&seed_name).ok_or_else(|| Error::NoSeedFound {
570        multisigner: root_multisigner.clone(),
571    })?;
572    // create fixed-length string to avoid reallocations
573    let mut full_address = String::with_capacity(seed_phrase.len() + derivation_path.len());
574    full_address.push_str(seed_phrase);
575    full_address.push_str(derivation_path);
576
577    let (_, encryption) = network_key.genesis_hash_encryption()?;
578    let multi_signer = full_address_to_multisigner(full_address, encryption)?;
579
580    let address_details = AddressDetails {
581        seed_name,
582        path: derivation_path.to_string(),
583        has_pwd: false,
584        network_id: Some(network_key),
585        encryption,
586        secret_exposed: false,
587    };
588    Ok((multi_signer, address_details))
589}

591pub fn inject_derivations_has_pwd(
592    mut seed_derived_keys: Vec<SeedKeysPreview>,
593    seeds: HashMap<String, String>,
594) -> Result<Vec<SeedKeysPreview>> {
595    let mut sr25519_signers = HashMap::new();
596    let mut ed25519_signers = HashMap::new();
597    let mut ecdsa_signers = HashMap::new();
598
599    for (k, v) in &seeds {
600        let sr25519_public = sr25519::Pair::from_phrase(v, None).unwrap().0.public();
601        let ed25519_public = ed25519::Pair::from_phrase(v, None).unwrap().0.public();
602        let ecdsa_public = ecdsa::Pair::from_phrase(v, None).unwrap().0.public();
603        sr25519_signers.insert(sr25519_public, k);
604        ed25519_signers.insert(ed25519_public, k);
605        ecdsa_signers.insert(ecdsa_public, k);
606    }
607
608    for addr in seed_derived_keys.iter_mut() {
609        for derived_key in addr.derived_keys.iter_mut() {
610            let path = derived_key.derivation_path.clone().unwrap_or_default();
611            let seed_name = match addr.multisigner {
612                MultiSigner::Sr25519(p) => sr25519_signers.get(&p),
613                MultiSigner::Ed25519(p) => ed25519_signers.get(&p),
614                MultiSigner::Ecdsa(p) => ecdsa_signers.get(&p),
615            };
616
617            let seed_name = if let Some(seed_name) = seed_name {
618                seed_name
619            } else {
620                continue;
621            };
622
623            let seed_phrase = if let Some(seed_phrase) = seeds.get(seed_name.as_str()) {
624                seed_phrase
625            } else {
626                continue;
627            };
628
629            // create fixed-length string to avoid reallocation
630            let mut full_address = String::with_capacity(seed_phrase.len() + path.len());
631            full_address.push_str(seed_phrase);
632            full_address.push_str(&path);
633
634            let multisigner_pwdless =
635                full_address_to_multisigner(full_address, derived_key.encryption)?;
636            let multisigner =
637                base58_or_eth_pubkey_to_multisigner(&derived_key.address, &derived_key.encryption)?;
638            derived_key.has_pwd = Some(multisigner_pwdless != multisigner);
639        }
640    }
641    Ok(seed_derived_keys)
642}

644/// Get all existing addresses from the database.
645#[cfg(feature = "active")]
646pub fn get_all_addresses(database: &sled::Db) -> Result<Vec<(MultiSigner, AddressDetails)>> {
647    let identities = open_tree(database, ADDRTREE)?;
648    let mut out: Vec<(MultiSigner, AddressDetails)> = Vec::new();
649    for (address_key_vec, address_entry) in identities.iter().flatten() {
650        let address_key = AddressKey::from_ivec(&address_key_vec)?;
651        let (multisigner, address_details) =
652            AddressDetails::process_entry_with_key_checked(&address_key, address_entry)?;
653        out.push((multisigner, address_details));
654    }
655    Ok(out)
656}

730///     <tr><td><code>//Alice//1///&ltpassword1&gt</code></td><td>+</td><td><code>//Alice///&ltpassword0&gt</code></td></tr>

733#[cfg(feature = "active")]

735    path: &str,
736    path_is_passworded: bool,
737    exposed_path: &str,
738    exposed_is_passworded: bool,
739) -> bool {
740    if (path == exposed_path) || path.starts_with(&format!("{exposed_path}/")) {

742    } else {
743        false
744    }

804/// Prepare data for address generation using seed phrase, derivation path, and
805/// network specs.
806///
807/// Function generates addresses individually, and checks the database to
808/// avoid collisions and duplicates. However, sometimes within a single database
809/// transaction multiple addresses are generated, and each new address must be
810/// checked with already semi-prepared new addresses set, to avoid collisions,
811/// duplicates and overwriting.
812///
813/// Therefore, function inputs a set of `(AddressKey, AddressDetails)` values
814/// already processed in preparation for the current database transaction and if
815/// corresponding `AddressKey` is already there, adds new `NetworkSpecsKey` into
816/// `AddressDetails` field `network_id` instead of creating a new entry.
817///
818/// The output [`PrepData`] contains:
819///
820/// - `address_prep` set of `(AddressKey, AddressDetails)` values that
821///   eventually will be transformed into [`Batch`](sled::Batch) to update the
822///   `ADDRTREE` of the cold database
823/// - output `Event` set that will be eventually added into `HISTORY` as a part
824///   of single `Entry`(definitions::history::Entry)
825///
826/// Function creates only **one** address, and output `PrepData` **updates**
827/// address preparation set received as input.
828///
829/// Function gets used both on Active side (when generating test cold database
830/// with well-known addresses) and on Vault side (when real addresses are
831/// actually created by the user).
832///
833/// Secrets passed into this function are:
834///
835/// - seed phrase
836/// - path, possibly with secret password part (`///<...>`).
837///
838/// Both are input as `&str`. Seed phrase is combined with derivation path into
839/// a string, a reference to which is sent into inner logic of
840/// [`sp_core::crypto`]. Combined secret string is then zeroized here regardless
841/// of the address generation success.
842#[cfg(feature = "active")]
843pub(crate) fn create_address(
844    database: &sled::Db,
845    input_batch_prep: &[(AddressKey, AddressDetails)],
846    path: &str,
847    network_specs: Option<&NetworkSpecs>,
848    seed_name: &str,
849    seed_phrase: &str,
850) -> Result<PrepData> {
851    // Check that the seed phrase is not empty.
852    // In upstream, empty seed phrase means default Alice seed phrase.
853    if seed_phrase.is_empty() {
854        return Err(Error::EmptySeed);
855    }
856    // create fixed-length string to avoid reallocations
857    let mut full_address = String::with_capacity(seed_phrase.len() + path.len());
858    full_address.push_str(seed_phrase);
859    full_address.push_str(path);
860
861    let encryption = network_specs
862        .map(|ns| ns.encryption)
863        .unwrap_or(Encryption::Sr25519);
864
865    let multisigner = full_address_to_multisigner(full_address, encryption)?;
866
867    // TODO regex elements may keep the line with password somewhere, how to
868    // zeroize then? checked regex crate and it appears that only references are
869    // moved around; need to double-check later;
870    let (cropped_path, has_pwd) = match REG_PATH.captures(path) {
871        Some(caps) => match caps.name("path") {
872            Some(a) => (a.as_str(), caps.name("password").is_some()),
873            None => ("", caps.name("password").is_some()),
874        },
875        None => ("", false),
876    };
877
878    do_create_address(
879        database,
880        input_batch_prep,
881        cropped_path,
882        network_specs,
883        seed_name,
884        multisigner,
885        has_pwd,
886    )
887}

912#[allow(clippy::too_many_arguments)]
913fn do_create_address(
914    database: &sled::Db,
915    input_batch_prep: &[(AddressKey, AddressDetails)],
916    cropped_path: &str,
917    network_specs: Option<&NetworkSpecs>,
918    seed_name: &str,
919    multisigner: MultiSigner,
920    has_pwd: bool,
921) -> Result<PrepData> {
922    // Check that the seed name is not empty.
923    if seed_name.is_empty() {
924        return Err(Error::EmptySeedName);
925    }
926    let mut address_prep = input_batch_prep.to_vec();
927    let mut history_prep = vec![];
928    let mut address_key: Option<AddressKey> = None;
929    let mut network_specs_key: Option<NetworkSpecsKey> = None;
930    if let Some(network_specs) = network_specs {
931        network_specs_key = Some(NetworkSpecsKey::from_parts(
932            &network_specs.genesis_hash,
933            &network_specs.encryption,
934        ));
935
936        let public_key = multisigner_to_public(&multisigner);
937        address_key = Some(AddressKey::new(
938            multisigner.clone(),
939            Some(network_specs.genesis_hash),
940        ));
941
942        // prepare history log here
943        let identity_history = IdentityHistory::get(
944            seed_name,
945            &network_specs.encryption,
946            &public_key,
947            cropped_path,
948            network_specs.genesis_hash,
949        );
950        history_prep.push(Event::IdentityAdded { identity_history });
951    }
952    if address_key.is_none() {
953        address_key = Some(AddressKey::new(multisigner.clone(), None))
954    }
955
956    // check if the same address key already participates in current database
957    // transaction
958
959    // number in `(AddressKey, AddressDetails)` set with same `AddressKey`, if
960    // found
961    let number_in_current = None;
962
963    match number_in_current {
964        // `AddressKey` already participates in transaction, just add
965        // `NetworkSpecsKey` into corresponding `AddressDetails` `network_id`
966        // set
967        Some(i) => {
968            let mod_entry = address_prep.remove(i);
969            address_prep.push(mod_entry);
970            Ok(PrepData {
971                address_prep,
972                history_prep,
973            })
974        }
975
976        // `AddressKey` is not yet in the transaction
977        None => {
978            // The only way to create address with `secret_exposed` marker is if
979            // the parent with exposed secret is in the database.
980            //
981            // If the address key already exists in the database, the
982            // `secret_exposed` flag will not be changed by extending key to
983            // another network.
984            let secret_exposed =
985                has_parent_with_exposed_secret(cropped_path, has_pwd, seed_name, database)?;
986
987            // check if the `AddressKey` is already in the database
988            let identities = open_tree(database, ADDRTREE)?;
989            if let Some(address_key) = address_key {
990                match identities.get(address_key.key()) {
991                    // `AddressKey` is in the database
992                    Ok(Some(address_entry)) => {
993                        if let Some(network_specs_key) = network_specs_key {
994                            let address_details = AddressDetails::from_entry_with_key_checked(
995                                &address_key,
996                                address_entry,
997                            )?;
998
999                            Err(Error::DerivationExists {
1000                                multisigner,
1001                                address_details: Box::new(address_details),
1002                                network_specs_key,
1003                            })
1004                        } else {
1005                            Ok(PrepData {
1006                                address_prep,
1007                                history_prep,
1008                            })
1009                        }
1010                    }
1011
1012                    // `AddressKey` is not in the database either.
1013                    // Make altogether new entry.
1014                    Ok(None) => {
1015                        let address_details = AddressDetails {
1016                            seed_name: seed_name.to_string(),
1017                            path: cropped_path.to_string(),
1018                            has_pwd,
1019                            network_id: network_specs_key,
1020                            encryption: network_specs
1021                                .map(|ns| ns.encryption)
1022                                .unwrap_or(Encryption::Sr25519),
1023                            secret_exposed,
1024                        };
1025                        address_prep.push((address_key, address_details));
1026                        Ok(PrepData {
1027                            address_prep,
1028                            history_prep,
1029                        })
1030                    }
1031
1032                    // database error
1033                    Err(e) => Err(e.into()),
1034                }
1035            } else {
1036                Ok(PrepData {
1037                    address_prep,
1038                    history_prep,
1039                })
1040            }
1041        }
1042    }
1043}

1045/// Prepare data for default addresses generation for a seed.
1046///
1047/// Each network has a default derivation path, recorded in [`NetworkSpecs`]
1048/// field `path_id`, normally `path_id` is `//<network_name>`.
1049///
1050/// This function prepares the generation of addresses with default derivation
1051/// paths for each of the available networks. Additionally, if the
1052/// `make_seed_keys` flag is set to `true` the function also prepares seed key
1053/// generation.
1054///
1055/// The output [`PrepData`] contains:
1056///
1057/// - `address_prep` set of `(AddressKey, AddressDetails)` values that
1058///   eventually will be transformed into [`Batch`](sled::Batch) to update the
1059///   `ADDRTREE` of the cold database
1060/// - output `Event` set that will be eventually added into `HISTORY` as a part
1061///   of single `Entry`(definitions::history::Entry)
1062///
1063/// All output `PrepData` elements are generated within this function.
1064/// It does not input addresses previously processed for the same transaction
1065/// and therefore does not check for collisions, so this function always goes
1066/// first, when there are no address additions planned yet.
1067///
1068/// This function inputs secret seed phrase as `&str`. It is passed as `&str`
1069/// into `create_address` and used there.
1070#[cfg(feature = "active")]
1071fn populate_addresses(
1072    database: &sled::Db,
1073    seed_name: &str,
1074    seed_phrase: &str,
1075    make_seed_keys: bool,
1076) -> Result<PrepData> {
1077    // Set of `(AddressKey, AddressDetails)` to be added into the database.
1078    // Set is updated and is used on each iteration of `create_address` to check
1079    // for collisions.
1080    let mut address_prep: Vec<(AddressKey, AddressDetails)> = Vec::new();
1081
1082    // Set of associated `Event::IdentityAdded(_)` entries.
1083    // Gets updated after each successful `create_address` run.
1084    let mut history_prep: Vec<Event> = Vec::new();
1085
1086    // Make seed keys if requested.
1087    // Seed keys **must** be possible to generate,
1088    // if a seed key has a collision with some other key, it is an error
1089    if make_seed_keys {
1090        let prep_data = create_address(database, &address_prep, "", None, seed_name, seed_phrase)?;
1091        address_prep = prep_data.address_prep;
1092        history_prep.extend_from_slice(&prep_data.history_prep);
1093    }
1094    Ok(PrepData {
1095        address_prep,
1096        history_prep,
1097    })
1098}

1191/// Remove a set of addresses within a single network from the Vault database.
1192///
1193/// Address is determined by [`MultiSigner`] and [`NetworkSpecsKey`] of the
1194/// network. Function inputs network `NetworkSpecsKey` and a set of
1195/// `MultiSigner` values for which the address within a network must be deleted.
1196///
1197/// `NetworkSpecsKey` of the network is removed from `network_id` set in
1198/// [`AddressDetails`] associated with each of the provided `MultiSigner`
1199/// values. If no networks associated with [`AddressKey`] remain, i.e.
1200/// `network_id` set becomes empty, whole associated entry is removed.
1201pub fn remove_keys_set(
1202    database: &sled::Db,
1203    multiselect: &[MultiSigner],
1204    network_specs_key: &NetworkSpecsKey,
1205) -> Result<()> {
1206    let mut id_batch = Batch::default();
1207    let mut events: Vec<Event> = Vec::new();
1208    let network_specs = get_network_specs(database, network_specs_key)?;
1209    for multisigner in multiselect.iter() {
1210        let public_key = multisigner_to_public(multisigner);
1211        let address_key =
1212            AddressKey::new(multisigner.clone(), Some(network_specs.specs.genesis_hash));
1213        let address_details = get_address_details(database, &address_key)?;
1214        let identity_history = IdentityHistory::get(
1215            &address_details.seed_name,
1216            &network_specs.specs.encryption,
1217            &public_key,
1218            &address_details.path,
1219            network_specs.specs.genesis_hash,
1220        );
1221        events.push(Event::IdentityRemoved { identity_history });
1222        if address_details.network_id.as_ref() == Some(network_specs_key) {
1223            id_batch.remove(address_key.key())
1224        } else {
1225            id_batch.insert(address_key.key(), address_details.encode())
1226        }
1227    }
1228    TrDbCold::new()
1229        .set_addresses(id_batch) // modify existing address entries
1230        .set_history(events_to_batch(database, events)?) // add corresponding history
1231        .apply(database)
1232}

1234/// Add a set of new derived addresses: N+1, N+2, etc into Vault database.
1235///
1236/// The number of the added addresses is set by `increment` input.
1237///
1238/// Input [`MultiSigner`] and [`NetworkSpecsKey`] determine the address, and the
1239/// associated [`AddressDetails`] are found in the database. The derivations are
1240/// based on the `path` from the `AddressDetails`.
1241///
1242/// If the database has addressed in given network with path that could be
1243/// interpreted as `<path>//<number>`, the first derivation of the increment set
1244/// is `<path>//<number>+1`. If no such addresses are existing in the database,
1245/// the first derivation of the increment set is `<path>//0`.
1246///
1247/// Further derivations of the increment set have number part increased by 1
1248/// until all requested derivations are generated.
1249///
1250/// Note: this is intended only for addresses not protected by the password
1251/// (i.e. having derivation path **without** `///<password>` part). If the
1252/// increment set is created on a passworded address, only public derivation
1253/// path part will be used for increment set generation.
1254///
1255/// This function inputs secret seed phrase as `&str`. It is passed as `&str`
1256/// into `create_address` and used there.
1257///
1258/// ## Example
1259///
1260/// User creates increment set of 3 on derivation `//user`. Resulting
1261/// derivations are:
1262///
1263/// - `//user`
1264/// - `//user//0`
1265/// - `//user//1`
1266/// - `//user//2`
1267///
1268/// User creates increment set of 2 on `//user` again, resulting derivations
1269/// are:
1270///
1271/// - `//user`
1272/// - `//user//0`
1273/// - `//user//1`
1274/// - `//user//2`
1275/// - `//user//3`
1276/// - `//user//4`
1277///
1278/// If the increment set of 2 is created on `//user//0`, resulting derivations
1279/// would be:
1280///
1281/// - `//user`
1282/// - `//user//0`
1283/// - `//user//0//0`
1284/// - `//user//0//1`
1285/// - `//user//1`
1286/// - `//user//2`
1287/// - `//user//3`
1288/// - `//user//4`
1289pub fn create_increment_set(
1290    database: &sled::Db,
1291    increment: u32,
1292    multisigner: &MultiSigner,
1293    network_specs_key: &NetworkSpecsKey,
1294    seed_phrase: &str,
1295) -> Result<()> {
1296    let network_specs = get_network_specs(database, network_specs_key)?;
1297    let address_details = get_address_details(
1298        database,
1299        &AddressKey::new(multisigner.clone(), Some(network_specs.specs.genesis_hash)),
1300    )?;
1301
1302    let existing_identities =
1303        addresses_set_seed_name_network(database, &address_details.seed_name, network_specs_key)?;
1304    let mut last_index = 0;
1305    for (_, details) in existing_identities.iter() {
1306        if let Some(("", suffix)) = details.path.split_once(&address_details.path) {
1307            if let Some(could_be_number) = suffix.get(2..) {
1308                if let Ok(index) = could_be_number.parse::<u32>() {
1309                    last_index = std::cmp::max(index + 1, last_index);
1310                }
1311            }
1312        }
1313    }
1314    let mut identity_adds: Vec<(AddressKey, AddressDetails)> = Vec::new();
1315    let mut current_events: Vec<Event> = Vec::new();
1316    for i in 0..increment {
1317        let path = address_details.path.to_string() + "//" + &(last_index + i).to_string();
1318        let prep_data = create_address(
1319            database,
1320            &identity_adds,
1321            &path,
1322            Some(&network_specs.specs),
1323            &address_details.seed_name,
1324            seed_phrase,
1325        )?;
1326        identity_adds = prep_data.address_prep;
1327        current_events.extend_from_slice(&prep_data.history_prep);
1328    }
1329    let id_batch = upd_id_batch(Batch::default(), identity_adds);
1330    TrDbCold::new()
1331        .set_addresses(id_batch) // add created addresses
1332        .set_history(events_to_batch(database, current_events)?) // add corresponding history
1333        .apply(database)
1334}

1351/// Proposed derivation status.
1352pub enum DerivationCheck {

1354    /// generation.
1355    BadFormat,
1356
1357    /// Derivation has password (`///<password>` part).
1358    ///
1359    /// Dynamic check in the database for already existing address is disabled.

1362    /// Derivation has no password.
1363    ///
1364    /// Dynamic check in the database for already existing address is enabled.

1371/// Preliminary checks if the user-proposed derivation could be used to create
1372/// the address.
1373///
1374/// This function **does not** use secret phrase, and only searches the
1375/// [`AddressDetails`] from the database for already existing addresses with
1376/// same derivation path, in case there is no password in the proposed
1377/// derivation.
1378///
1379/// If the proposed derivation is passworded, no checking could be done without
1380/// using the seed phrase, and UI allows to proceed to try creating new address.
1381/// If address already exists, an error would appear on the next stage.
1382///
1383/// No checking for collisions is done here (i.e. no checking if the same
1384/// [`AddressKey`] will be produced for <secret phrase 1> + <derivation path 1>
1385/// as for already known <secret phrase 2> + <derivation path 2>), because seed
1386/// phrase is not known. If collision occurs, an error would appear on the next
1387/// stage.
1388///
1389/// Note that the only errors in this function could appear from database
1390/// malfunction when the database is being searched for the exact match, in case
1391/// of password-free valid derivation. Bad format of the derivation is **not**
1392/// an error, UI just does not allow to proceed.
1393pub fn derivation_check(
1394    database: &sled::Db,
1395    seed_name: &str,
1396    path: &str,
1397    network_specs_key: &NetworkSpecsKey,
1398) -> Result<DerivationCheck> {
1399    match is_passworded(path) {
1400        // Proposed derivation has password, no checks could be made, proceed.
1401        Ok(true) => Ok(DerivationCheck::Password),
1402
1403        // Proposed derivation has no password, checking the database for exact
1404        // coincidence.
1405        Ok(false) => {
1406            let mut found_exact = None;
1407            for (multisigner, address_details) in get_all_addresses(database)?.into_iter() {
1408                if (address_details.seed_name == seed_name) // seed name
1409                    && (address_details.path == path) // derivation path, cropped part without password
1410                    && (address_details.network_id.as_ref() == Some(network_specs_key)) // in this network
1411                    && (!address_details.has_pwd)
1412                // has no password to begin with
1413                {
1414                    found_exact = Some((multisigner, address_details));
1415                    break;
1416                }
1417            }
1418            Ok(DerivationCheck::NoPassword(found_exact))
1419        }
1420
1421        // Proposed derivation is not suitable, UI would not allow to proceed.
1422        // Note that this is **not** an error.
1423        Err(_) => Ok(DerivationCheck::BadFormat),
1424    }
1425}

1427/// Separate derivation path with password into non-secret part and password.
1428///
1429/// Function is expected to deal **only** with derivations having a password
1430/// (i.e. with `///<password>` part of the derivation path). Password-free
1431/// derivations do not get to the point in `navigator` where this function is
1432/// called. Calling this function on password-free derivation is an error.
1433///
1434/// Function receives derivation path with secret password part, and outputs
1435/// hard and soft derivations part and secret password part of the derivation
1436/// path separately.
1437///
1438// TODO regex and secrets, see `create_address` comments.
1439pub fn cut_path(path: &str) -> Result<(String, String)> {
1440    let caps = REG_PATH
1441        .captures(path)
1442        .ok_or_else(|| Error::InvalidDerivation(path.to_string()))?;
1443    let cropped_path = caps
1444        .name("path")
1445        .map(|a| a.as_str().to_string())
1446        .ok_or_else(|| Error::InvalidDerivation(path.to_string()))?;
1447    match caps.name("password") {
1448        Some(pwd) => Ok((cropped_path, pwd.as_str().to_string())),
1449        None => Err(Error::LostPwd),
1450    }
1451}

1453/// Create a new address in the Vault database.
1454///
1455/// If the associated [`AddressKey`] already exists in the [`ADDRTREE`] tree of
1456/// the Vault database, a new network [`NetworkSpecsKey`] is added into
1457/// `network_id` set of the `AddressDetails`. If the `AddressKey` is new,
1458/// a new `ADDRTREE` entry is made.
1459///
1460/// Secrets passed into this function are:
1461///
1462/// - seed phrase
1463/// - path, possibly with secret password part (`///<...>`).
1464///
1465/// Both are input as `&str` and go directly in `create_address` (both) and
1466/// `derivation_check` (path only).
1467pub fn try_create_address(
1468    database: &sled::Db,
1469    seed_name: &str,
1470    seed_phrase: &str,
1471    path: &str,
1472    network_specs_key: &NetworkSpecsKey,
1473) -> Result<()> {
1474    match derivation_check(database, seed_name, path, network_specs_key)? {
1475        // UI should prevent user from getting into `try_create_address` if
1476        // derivation has a bad format
1477        DerivationCheck::BadFormat => Err(Error::InvalidDerivation(path.to_string())),
1478
1479        // UI should prevent user from getting into `try_create_address` if
1480        // derivation already exists
1481        DerivationCheck::NoPassword(Some((multisigner, address_details))) => {
1482            Err(Error::DerivationExists {
1483                multisigner,
1484                address_details: Box::new(address_details),
1485                network_specs_key: network_specs_key.to_owned(),
1486            })
1487        }
1488
1489        // Valid derivation:
1490        // - without a password and not in the database
1491        // - with password
1492        // Vault can try adding either. Collisions (for both cases) and exact
1493        // coincidence (for passworded derivation) could not have been checked
1494        // preliminarily and would result in `create_address` errors here.
1495        _ => {
1496            let network_specs = get_network_specs(database, network_specs_key)?;
1497            let prep_data = create_address(
1498                database,
1499                &Vec::new(), // a single address is created, no data to check against here
1500                path,
1501                Some(&network_specs.specs),
1502                seed_name,
1503                seed_phrase,
1504            )?;
1505            let id_batch = upd_id_batch(Batch::default(), prep_data.address_prep);
1506            TrDbCold::new()
1507                .set_addresses(id_batch) // add created address
1508                .set_history(events_to_batch(database, prep_data.history_prep)?) // add corresponding history
1509                .apply(database)
1510        }
1511    }
1512}

1641fn prepare_secret_key_for_export(
1642    multisigner: &MultiSigner,
1643    full_address: &str,
1644    pwd: Option<&str>,
1645) -> Result<[u8; 32]> {
1646    match multisigner {
1647        MultiSigner::Ed25519(public) => {
1648            let ed25519_pair =
1649                ed25519::Pair::from_string(full_address, pwd).map_err(Error::SecretStringError)?;
1650            if public != &ed25519_pair.public() {
1651                return Err(Error::WrongPassword);
1652            }
1653            Ok(ed25519_pair.seed().to_owned())
1654        }
1655        MultiSigner::Sr25519(public) => {
1656            let (sr25519_pair, seed) = sr25519::Pair::from_string_with_seed(full_address, pwd)
1657                .map_err(Error::SecretStringError)?;
1658            if public != &sr25519_pair.public() {
1659                return Err(Error::WrongPassword);
1660            }
1661            Ok(seed.ok_or_else(|| Error::NoSeedForKeyPair {
1662                multisigner: multisigner.clone(),
1663            })?)
1664        }
1665        MultiSigner::Ecdsa(public) => {
1666            let ecdsa_pair =
1667                ecdsa::Pair::from_string(full_address, pwd).map_err(Error::SecretStringError)?;
1668            if public != &ecdsa_pair.public() {
1669                return Err(Error::WrongPassword);
1670            }
1671            Ok(ecdsa_pair.seed())
1672        }
1673    }
1674}

1676/// Prepare **secret** key export screen struct [`MKeyDetails`].
1677///
1678/// For QR code the secret key information is put in format
1679/// `secret:0x{}:{network genesis hash}` transformed into bytes, to be
1680/// compatible with `polkadot-js` interface.
1681///
1682/// There is no direct mentioning of [`Encryption`] in the QR code. It is
1683/// that user knows the key in which network and with what encryption they are
1684/// exporting.
1685///
1686/// The QR code here contains sensitive information, and is made with special
1687/// coloration, so that the difference with safe QR codes is immediately visible
1688/// on screen.
1689pub fn export_secret_key(
1690    database: &sled::Db,
1691    public_key: &str,
1692    expected_seed_name: &str,
1693    network_specs_key_hex: &str,
1694    seed_phrase: &str,
1695    mut key_password: Option<String>,
1696) -> Result<MKeyDetails> {
1697    let public_key = &unhex(public_key)?;
1698    let network_specs_key = &NetworkSpecsKey::from_hex(network_specs_key_hex)?;
1699    let network_specs = get_network_specs(database, network_specs_key)?;
1700    let multisigner = &get_multisigner(public_key, &network_specs.specs.encryption)?;
1701    let address_key = AddressKey::new(multisigner.clone(), Some(network_specs.specs.genesis_hash));
1702    let address_details = get_address_details(database, &address_key)?;
1703    if address_details.seed_name != expected_seed_name {
1704        return Err(Error::SeedNameNotMatching {
1705            address_key,
1706            expected_seed_name: expected_seed_name.to_string(),
1707            real_seed_name: address_details.seed_name,
1708        });
1709    }
1710    if address_details.network_id.as_ref() != Some(network_specs_key) {
1711        return Err(Error::NetworkSpecsKeyForAddressNotFound {
1712            network_specs_key: network_specs_key.to_owned(),
1713            address_key,
1714        });
1715    }
1716    let public_key = multisigner_to_public(multisigner);
1717
1718    let style = address_details.identicon_style();
1719    let address = Address {
1720        path: address_details.path.to_string(),
1721        has_pwd: address_details.has_pwd,
1722        identicon: make_identicon_from_multisigner(multisigner, style),
1723        seed_name: address_details.seed_name.to_string(),
1724        secret_exposed: true,
1725    };
1726
1727    let network_info = MSCNetworkInfo {
1728        network_title: network_specs.specs.name,
1729        network_logo: network_specs.specs.logo,
1730        network_specs_key: network_specs_key_hex.to_owned(),
1731    };
1732
1733    let database_addresses = get_addresses_by_seed_name(database, expected_seed_name)?;
1734
1735    let exposed_addresses = exposed_set(
1736        &address_details.path,
1737        address_details.has_pwd,
1738        database_addresses,
1739    );
1740
1741    let mut identity_batch = Batch::default();
1742
1743    for (x_multisigner, x_address_details) in exposed_addresses.into_iter() {
1744        let mut new_address_details = x_address_details;
1745        new_address_details.secret_exposed = true;
1746        if let Some(id) = &new_address_details.network_id {
1747            let network_specs = get_network_specs(database, id)?;
1748            identity_batch.insert(
1749                AddressKey::new(
1750                    x_multisigner.clone(),
1751                    Some(network_specs.specs.genesis_hash),
1752                )
1753                .key(),
1754                new_address_details.encode(),
1755            )
1756        }
1757    }
1758
1759    let history_batch = events_to_batch(
1760        database,
1761        vec![Event::SecretWasExported {
1762            identity_history: IdentityHistory::get(
1763                &address_details.seed_name,
1764                &address_details.encryption,
1765                &public_key,
1766                &address_details.path,
1767                network_specs.specs.genesis_hash,
1768            ),
1769        }],
1770    )?;
1771
1772    let qr = generate_secret_qr(
1773        multisigner,
1774        &address_details,
1775        &network_specs.specs.genesis_hash,
1776        seed_phrase,
1777        key_password.as_deref(),
1778    )?;
1779    key_password.zeroize();
1780
1781    TrDbCold::new()
1782        .set_addresses(identity_batch) // modify addresses
1783        .set_history(history_batch) // add corresponding history
1784        .apply(database)?;
1785
1786    Ok(MKeyDetails {
1787        qr,
1788        pubkey: hex::encode(public_key),
1789        network_info,
1790        base58: print_multisigner_as_base58_or_eth_address(
1791            multisigner,
1792            Some(network_specs.specs.base58prefix),
1793            address_details.encryption,
1794        ),
1795        address,
1796    })
1797}

1799fn generate_secret_qr(
1800    multisigner: &MultiSigner,
1801    address_details: &AddressDetails,
1802    genesis_hash: &H256,
1803    seed_phrase: &str,
1804    pwd: Option<&str>,
1805) -> Result<QrData> {
1806    // create fixed-length string to avoid reallocations
1807
1808    let mut full_address = String::with_capacity(seed_phrase.len() + address_details.path.len());
1809    full_address.push_str(seed_phrase);
1810    full_address.push_str(&address_details.path);
1811
1812    let mut secret = match prepare_secret_key_for_export(multisigner, &full_address, pwd) {
1813        Ok(a) => {
1814            full_address.zeroize();
1815            a
1816        }
1817        Err(e) => {
1818            full_address.zeroize();
1819            return Err(e);
1820        }
1821    };
1822
1823    let qr = QrData::Sensitive {
1824        data: format!(
1825            "secret:0x{}:{}",
1826            hex::encode(secret),
1827            hex::encode(genesis_hash)
1828        )
1829        .as_bytes()
1830        .to_vec(),
1831    };
1832    secret.zeroize();
1833    Ok(qr)
1834}
