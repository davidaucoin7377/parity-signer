1//! Displaying and updating history log
2//!
3//! The Vault keeps a history log of all events that change the database and
4//! affect security. It is stored in [`HISTORY`] tree of the cold database.
5//!
6//! Each history log [`Entry`] contains [`Event`] set and a timestamp. Database
7//! key for [`Entry`] value is [`Order`], SCALE-encoded number of the entry.
8//!
9//! In addition to keeping the log, Vault also displays [`HISTORY`] tree
10//! checksum for user to possibly keep the track of.

12// some functions are not called at the moment from the user interface - kept

32/// Print total number of pages, for maximum [`HISTORY_PAGE_SIZE`](constants::HISTORY_PAGE_SIZE) number of
33/// entries per page.
34pub fn history_total_pages(database: &sled::Db) -> Result<u32> {
35    use constants::HISTORY_PAGE_SIZE;
36
37    let history = get_history(database)?;
38    let total_pages = {
39        if history.len() % HISTORY_PAGE_SIZE == 0 {
40            history.len() / HISTORY_PAGE_SIZE
41        } else {
42            history.len() / HISTORY_PAGE_SIZE + 1
43        }
44    };
45    Ok(total_pages as u32)
46}

48/// Get history log contents from the database.
49pub fn get_history(database: &sled::Db) -> Result<Vec<(Order, Entry)>> {
50    let history = open_tree(database, HISTORY)?;
51    let mut out: Vec<(Order, Entry)> = Vec::new();
52    for (order_encoded, history_entry_encoded) in history.iter().flatten() {
53        let order = Order::from_ivec(&order_encoded)?;
54        let history_entry = <Entry>::decode(&mut &history_entry_encoded[..])?;
55        out.push((order, history_entry));
56    }
57    out.sort_by(|a, b| b.0.stamp().cmp(&a.0.stamp()));
58    Ok(out)
59}

61/// Get from the database a history log [`Entry`] by `u32` order identifier
62/// received from the frontend.
63pub fn get_history_entry_by_order(database: &sled::Db, order: u32) -> Result<Entry> {
64    let history = open_tree(database, HISTORY)?;
65    let mut found = None;
66    let order = Order::from_number(order);
67    for (order_encoded, history_entry_encoded) in history.iter().flatten() {
68        let order_found = Order::from_ivec(&order_encoded)?;
69        if order_found == order {
70            found = Some(<Entry>::decode(&mut &history_entry_encoded[..])?);
71            break;
72        }
73    }
74    found.ok_or_else(|| Error::HistoryEntryNotFound(order.stamp()))
75}

77/// Clear Vault history and make a log [`Entry`] that history was cleared.
78pub fn clear_history(database: &sled::Db) -> Result<()> {
79    let batch = make_batch_clear_tree(database, HISTORY)?;
80    let events = vec![Event::HistoryCleared];
81    let for_history = events_in_batch(database, true, batch, events)?;
82    TrDbCold::new().set_history(for_history).apply(database)?;
83
84    Ok(())
85}

87/// Timestamp [`Event`] set and make with it a new [`Batch`], that could be
88/// applied to the [`HISTORY`] tree.
89pub fn events_to_batch(database: &sled::Db, events: Vec<Event>) -> Result<Batch> {
90    events_in_batch(database, false, Batch::default(), events)
91}
92
93/// Timestamp [`Event`] set and add it to existing [`Batch`], that could be
94/// applied to the [`HISTORY`] tree.
95///
96/// Note that existing [`Batch`] must contain no [`Entry`] additions, only
97/// removals are possible. Only one [`Event`] set, transformed into [`Entry`]
98/// with a single timestamp could be added in a single database transaction.
99pub(crate) fn events_in_batch(
100    database: &sled::Db,
101    start_zero: bool,
102    mut out_prep: Batch,
103    events: Vec<Event>,
104) -> Result<Batch> {
105    let history = open_tree(database, HISTORY)?;
106    let order = {
107        if start_zero {
108            Order::from_number(0u32)
109        } else {
110            Order::from_number(history.len() as u32)
111        }
112    };
113    let timestamp = time::OffsetDateTime::now_utc().format(&time::macros::format_description!(
114        "[year]-[month]-[day] [hour]:[minute]:[second].[subsecond]"
115    ))?;
116    let history_entry = Entry { timestamp, events };
117    out_prep.insert(order.store(), history_entry.encode());
118    Ok(out_prep)
119}

121/// Enter [`Event`] set into the database as a single database transaction.
122pub fn enter_events(database: &sled::Db, events: Vec<Event>) -> Result<()> {
123    TrDbCold::new()
124        .set_history(events_to_batch(database, events)?)
125        .apply(database)
126}

128/// Enter user-generated [`Event`] into the database.
129pub fn history_entry_user(database: &sled::Db, string_from_user: &str) -> Result<()> {
130    let events = vec![Event::UserEntry {
131        user_entry: string_from_user.to_string(),
132    }];
133    enter_events(database, events)

144/// Process the fact that the Vault device was online.
145///
146/// - Add history log entry with `Event::DeviceWasOnline`.
147/// - Update [`DangerRecord`] stored in [`SETTREE`](constants::SETTREE) with
148///   `device_was_online = true` flag.
149///
150/// Unacknowledged non-safe [`DangerRecord`] block the use of Vault in the
151/// frontend.
152pub fn device_was_online(database: &sled::Db) -> Result<()> {
153    let events = vec![Event::DeviceWasOnline];
154    let mut settings_batch = Batch::default();
155    settings_batch.insert(DANGER, DangerRecord::set_was_online().store());
156    TrDbCold::new()
157        .set_history(events_to_batch(database, events)?)
158        .set_settings(settings_batch)
159        .apply(database)
160}
161
162/// Acknowledge that the Vault device was online and reset the
163/// [`DangerRecord`] back to safe.
164///
165/// - Add history log entry with `Event::ResetDangerRecord`.
166/// - Reset [`DangerRecord`] stored in [`SETTREE`](constants::SETTREE) to
167///   `safe`, i.e. with `device_was_online = false` flag.
168///
169/// Acknowledged and reset [`DangerRecord`] allow to resume the use of Vault in
170/// the frontend. Use it wisely.
171pub fn reset_danger_status_to_safe(database: &sled::Db) -> Result<()> {
172    let events = vec![Event::ResetDangerRecord];
173    let mut settings_batch = Batch::default();
174    settings_batch.insert(DANGER, DangerRecord::safe().store());
175    TrDbCold::new()
176        .set_history(events_to_batch(database, events)?)
177        .set_settings(settings_batch)
178        .apply(database)
179}
